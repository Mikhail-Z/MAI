)
,

.
Несмотря

,

,

.

.

.

:

:
Корпус

,

,

.
Пластик
,

в

,

,
это

touch
,

,

,

.
В

,

лицевой

.
На

,

из

.
Ход

,

.
Клавиша
располагается
на

,

по
центру
,

,

ее

,

,

отметить
,

.
Здесь

,

боковой

,

.
На
правой
боковой

интерфейсный

,

,

на
корпусе
.
А

и

пусты
.
На
лицевой
поверхности
телефона
располагается
объектив
камеры
,
который
совсем
немного
утоплен
в
корпус
,
а
вокруг
него

и

,
о

.
По
центру
лицевой
панели
производитель

по
размеру
(
20х25
мм
)
TFT-дисплей
,
который

128х160
точек
и

отображать
до
256000
цветов
.
Экран

под

,

его

его

.
Интересно
,
что

о

,
что

дисплей
цветной
,

и
не
получится
,

.
Увидеть

,

,

.
Благодаря

,

,

.
На

,

.
Часы

.
Для

,

.
Во

.
Также

)

.

.

.
Экран

,

,

.
На

,

Клавиатура
Под
дисплеем
располагаются
четыре
клавиши
быстрого
запуска
приложений
(
будильник
,
календарь
,
изображения
,
избранные
)
,
переназначить
действия
,
закрепленные
за
клавишами
,
нельзя
.
Честно
сказать
,
немного
непонятно
,
для
чего
эти
клавиши
вообще
были
сделаны
,
так
как
пользоваться
ими
не
очень
удобно
,
к
тому
же
они
не
имеют
подсветки
.
С
другой
стороны
,
возможно
,
кому-то
наличие
данных
клавиш
и
придется
по
душе
,
и
они
будут
им
полезны
.
Основная
клавиатура
располагается
на
нижней
половине
устройства
,
и
она
поделена
на
два
блока
.
Все
клавиши
выполнены
из
пластика
.
В
навигационный
блок
входит
круглая
навигационная
клавиша
с
вписанной
в
нее
кнопкой
«
Ок
»
,
две
функциональные
клавиши
,
кнопка
начала
и
окончания
вызова
,
кнопка
активации
камеры
,
а
также
кнопка
отмены
действия
.
Все
клавиши
хотя
и
располагаются
вплотную
друг
к
другу
,
но
это
никак
не
влияет
на
удобство
работы
с
ними
,
так
как
их
размеры
достаточно
крупные
,
чтобы
ошибочные
нажатия
были
практически
исключены
.
В
цифровом
блоке
клавиши
также
располагаются
вплотную
друг
к
другу
,
но
,
опять
же
,
из-за
их
крупных
размеров
работать
с
ними
очень
удобно
,
и
об
ошибочных
нажатиях
речи
не
идет
.
Сравнение
клавиатур
с

:
Подсветка

,

,

,

,

.
Цифровые

,

,

,

,

,

убедиться
,
взглянув
на

.
Символы

клавишах
очень
хорошо

,
с

никаких

у

людей
,

.

Аккумулятор
С
телефоном
поставляется
литий-ионнная
батарея
емкостью
800
мАч
.
По
данным
производителя
,
с
ней
аппарат
способен
проработать
до
2
часов
в
режиме
разговора
и
до
200
часов
в
режиме
ожидания
.
В
условиях
сетей
московских
операторов
аппарат
работал
в
среднем
2
суток
при
10-15
минутах
разговора
в
день
,
около
2-2.5
часов
прослушивания
радио
и
плеера
в
день
и
до
15
минут
использования
других
функций
.
Время
работы
не
поражает
воображение
,
хотя
если
вы
не
будете
использовать
плеер
и
радио
,
то
аппарат
спокойно
проработает
около
3
суток
,
хотя
,
конечно
,
это
и
не
очень
много
.
Время
полного
заряда
батареи
составляет
около
2.5
часов
.
К
содержанию
>
>
>
Память
,
подключение
к
ПК
Под
аккумулятором
располагается
слот
для
SIM-карты
,
а
рядом
-
слот
для
карт
памяти
формата
microSD
.
При
подключении
телефона
к
ПК
установленная
карта
памяти
будет
доступна
как
съемный
диск
без
установки
дополнительных
драйверов
.
Переключение
работы
аппарата
в
режим
«
Съемный
диск
»
или
«
Модем
»
производится
вручную
через
меню
телефона
.
Подключение
к
ПК
осуществляется
по
USB
версии
1.1
,
так
что
для
загрузки
большого
объема
информации
на
карту
памяти
проще
будет
воспользоваться
кард-ридером
,
чем
загружать
все
через
телефон
.
К
содержанию
>
>
>
Меню
Главное
меню
телефона
может
быть
представлено
либо
в
виде
вертикальных
списков
,
либо
в
виде
матрицы
3х4
.
Подменю
аппарата
представлено
в
виде
вертикальных
списков
.
В
телефоне
имеется
быстрая
навигация
с
помощью
цифровых
последовательностей
,
также
имеется
возможность
перемещаться
в
подменю
в
соседние
разделы
с
помощью
горизонтальных
отклонений
навигационной
клавиши
.
Для
того
чтобы
быстро
узнать
,
какие
разделы
соседствуют
с
тем
,
в
котором
находитесь
вы
в
данный
момент
,
в
верхней
части
дисплея
располагаются
названия
этих
пунктов
меню
вместе
со
стрелками
,
указывающими
направление
(
право
или
лево
)
.
В
режиме
ожидания
можно
назначить
доступ
к
четырем
пунктам
меню
на
нажатия
навигационной
клавиши
.
А
через
меню
настроек
дисплея
можно
включить
визуальную
подсказку
,
которая
будет
всегда
отображаться
на
главном
экране
,
с
помощью
иконок
показывая
,
какие
разделы
присвоены
клавише
.
Как
уже
говорилось
выше
,
под
дисплеем
имеются
также
и
клавиши
для
быстрого
доступа
,
но
переназначить
их
нельзя
.
Прорисовка
меню
хорошая
,
хотя
стоит
отметить
,
что
они
немного
мелковаты
,
и
в
телефоне
не
хватает
еще
одного
типа
отображения
меню
,
а
именно
постраничного
.
Просто
если
вы
выставите
в
настройках
телефона
шрифт
даже
на
самый
крупный
,
он
изменится
везде
,
кроме
главного
меню
.
И
поэтому
не
думаю
,
что
людям
,
которые
будут
использовать
телефон
со
шрифтом
,
выставленным
в
настройках
на
максимум
,
будет
комфортно
пользоваться
такими
мелкими
иконками
и
подписями
мелким
шрифтом
в
главном
меню
аппарата
.
В
аппарате
около
14
Мб
памяти
,

в
качестве
плеера
,
конечно
же
,
можно
,
только
не
стоит
рассчитывать
даже
на
высокое
качество
воспроизведения
,
к
тому
же
,
громкость
при
прослушивании
через
наушники
невысокая
,
и
ее
не
хватает
при
прослушивании
плеера
в
метро
.
FM-радио
.
В
памяти
телефона
можно
сохранять
до
12
радиостанций
,
по
одной
на
каждую
из
клавиш
цифрового
блока
,
для
быстрого
доступа
к
ним
.
Интересно
,
что
в
нашем
образце
не
было
возможности
работы
радио
в
фоновом
режиме
,
что
достаточно
странно
.
В
закрытом
состоянии
телефона
на
внешнем
экране
отображается
частота
выбранной
радиостанции
.
Да
,
для
сохраненных
радиостанций
нет
возможности
записать
их
название
,
отображается
только
частота
.
Радио
работает
только
при
подключенной
проводной
гарнитуре
,
провод
которой
выступает
в
качестве
антенны
.
При
этом
при
подключенной
гарнитуре
звук
можно
вывести
на
внешний
динамик
или
же
передать
его
на
Bluetooth-гарнитуру
.
Качество
приема
радио
достаточно
хорошее
.
Диктофон
.
Может
работать
только
в
фоновом
режиме
,
запись
разговора
с
его
помощью
невозможна
.
Продолжительность
записи
ограничена
только
свободной
памятью
аппарата
,
но
возможно
выставить
и
ограничение
,
для
того
чтобы
запись
можно
было
в
последующем
отправить
по
MMS
.

содержанию

Камера
В
телефоне
установлена
2
Мп
камера
без
автофокуса
.
В
качестве
видоискателя
может
выступать
только
внутренний
дисплей
.
Меню
настроек
достаточно
удобно
и
функционально
,
оно
отображается
в
виде
горизонтальной
строки
с
иконками
пунктов
,
настройки
выбранного
пункта
выводятся
в
виде
выпадающего
вертикального
списка
.
В
настройках
вы
можете
выбрать
разрешение
(
1600х1200
,
1280х1024
,
640х480
или
320х240
)
,
качество
(
нормальное
,
хорошее
,
отличное
)
,
включить
таймер
и
режим
серийной
съемки
(
3
или
6

)
.
Также
имеется
возможность
выбрать
один
из
трех
эффектов
(
сепия
,
негатив
,
черно-белый
)
,
выставить
баланс
белого
и
выбрать
мелодию
затвора
или
же
просто
выключить
звук
.
Качество

,
сделанных
встроенной
камерой

,
не

.

они

,

можно

Впечатления
Аппарат
обеспечивает
качество
связи
на
уровне
большинства
современных
GSM-телефонов
,
даже
чуть
лучше
.
Громкости
динамика
и
чувствительности
микрофона
хватает
для
разговора
практически
в
любых
условиях
.
В
аппарате
реализована
64-тональная
полифония
,
предустановленные
мелодии
звучат
средне
,
но
ничто
вам
не
мешает
установить
собственную
композицию
в
формате
MP3
,
тогда
ситуация
значительно
улучшится
.
Громкость
вызова
,
несмотря
на
отдельный
динамик
,
средняя
,
а
вот
вибровызов
достаточно
мощный
,
его
легко
почувствовать
,
даже
если
аппарат
находится
в
сумке
.
В
аппарате
реализована
функция
громкой
связи
,
которая
легко
активируется
и
деактивируется
во
время
разговора
одинарным
нажатием
на
клавишу
«
Ок
»
.
Работа
громкой
связи
в
аппарате
не
вызывает
нареканий
.
С
одной
стороны
,
модель
LG
KF300
-
это
вполне
обычный
малоинтересный
для
большинства
пользователей
аппарат
,
в
котором
вроде
бы
есть
все
необходимые
функции
,
будь
то
камера
,
плеер
или
радио
,
но
вот
реализованы
они
все
достаточно
посредственно
.
С
другой
стороны
,
аппарат
предлагает
такие
возможности
,
которые
не
встретишь
в
большинстве
аппаратов
,
это
крупные
клавиши
с
крупными
же
символами
на
них
,
возможность
выбрать
в
меню
просто
огромный
шрифт
,
ну
и
как
бонус
-
крупные
часы
,
отображаемые
на
внешнем
экране
.
По
последним
пунктам
модель
не
имеет
конкурентов
,
только
если
не
брать
в
расчет
специальные
аппараты
для
пожилых
людей
,
но
купить
такие
трубки
затруднительно
,
особенно
в
России
,
в
то
время
как
LG
KF300
доступен
в
большинстве
салонов
связи
.
Кстати
,
на
нашем
форуме
в
разделе
помощи
в
выборе
телефона
нередко
появляются
темы
с
просьбой
помочь
выбрать
аппарат
для
себя
или
же
для
родственников/друзей/коллег
,
главным
критерием
в
выборе
которого
являются
именно
крупный
шрифт
в
меню
и
удобные
клавиши
при
наборе
.
Так
вот
,
я
думаю
,
для
тех
,
кто
ищет
аппарат
по
этим
параметрам
,
LG
KF300
будет
идеальным
вариантом
.
Модель
уже
появилась
в
продаже
по
ориентировочной
цене
около
6000
рублей
,
что
вполне
оправданно
.
