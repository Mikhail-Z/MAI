.

.
Дизайн
,

,

,

.

,

.
Также

,

,

.
Уже

,

.
В

,

.
С

,

,

,

,

.
А

,

,

.
Внешне

,

.
Внешне

КПК

WM
и

LG
KC910
Renoir
.
От

,

.
С

,

,

.
Для

,

.
Возможно
,

,

.
Молодым

,

,

.
Производителю

,

.
Нет
,

,

,

,

.
Думаю
,

.
Кстати
,

,

,

аппарата

,

,

.
Габариты

(

,

)

.
В

,

.
В

,

,

,

,

.
Сравнение

:
На

,

,

,

.
Клавиша

,

,

.
На

,

,

,

.
На

,

,

.
Здесь

,

,

.
Немного

,

,

,

,

.
Здесь

,

,

.
Да-да
,

,

,

,

,

.
В

,

.
Стилус

.

,

.

.

,

.

Клавиатура
Клавиш
у
телефона
всего
три
,
они

размеру
,

,

нет
.
Кнопки
имеют
квадратную
форму
,
слегка

,
так
что

с

.
К

.

Экран
В
телефоне
установлен
крупный
3
”
(
38х64
мм
)
сенсорный
экран
,
выполненный
по
TFT-технологии
,
имеющий
разрешение
240х400
точек
и
способный
отображать
до
262000
цветов
.
На
экране
одновременно
может
отображаться
до
8
строк
текста
,
а
также
три
строки
со
служебной
информацией
.
В
сравнении
с
экраном
телефона
LG
KC910
Renoir
дисплей
KP500
выглядит
хуже
,
он
менее
яркий
и
контрастный
,
хотя
и
обладает
схожими
характеристиками
.
Также
дисплей
500-й
модели
хуже
обрабатывает
нажатия
,
у
него
явно
ниже
чувствительность
,
чем
у
экрана
KC910
.
С
дисплеем
KP500
можно
работать
как
используя
стилус
,
так
и
просто
нажимая
пальцами
,
единственное
,
что
к
работе
с
ним
придется
немного
привыкнуть
по
причине
его
не
очень
высокой
чувствительности
.
Для
комфортной
работы
с
телефоном
предусмотрено
три
режима
виброотдачи
,
а
также
звукового
эффекта
при
нажатии
,
которые
можно
настроить
через
меню
.
Режимы
вибрации
при
нажатии
отличаются
своей
продолжительностью
,
для
меня
наиболее
комфортным
оказался
первый
,
самый
короткий
,
режим
,
но
здесь
,
конечно
,
все
индивидуально
.
Кроме
этого
,
возможна
настройка
силы
вибрации
,
здесь
предлагается
выбор
из
пяти
уровней
,
опять
же
,
для
меня
оптимальным
был
третий
уровень
,
хотя
при
работе
в
перчатках
не
хватало
и
пятого
.
Звуковой
эффект
нажатия
можно
совсем
отключить
,
как
,
впрочем
,
и
вибрацию
,
также
имеется
возможность
настроить
громкость
звука
при
нажатии
.
Яркость
подсветки
в
аппарате
регулируется
вручную
и
имеет
5
уровней
.
Углы
обзора
у
дисплея
отличные
как
по
вертикали
,
так
и
по
горизонтали
.
К
содержанию
>
>
>
Батарея
Крышка
аккумуляторного
отсека
располагается
на
задней
стороне
телефона
и
занимает
практически
половину
всей
поверхности
.
Крепление
крышки
не
идеальное
,
присутствует
совсем
небольшой
люфт
,
который
со
временем
может
увеличиться
.
С
аппаратом
поставляется
литий-ионная
батарея
емкостью
900
мАч
.
По
данным
производителя
,
с
ней
аппарат
способен
проработать
до
3.5
часов
в
режиме
разговора
и
до
350
часов
в
режиме
ожидания
.
В
условиях
сетей
московских
операторов
аппарат
работал
в
среднем
около
2.5
суток
при
15-20
минутах
разговора
в
день
,
около
1
часа
прослушивания
плеера
и
радио
,
до
30
минут
серфинга
в
сети
и
около
20
минут
использования
других
функций
(
сообщения
,
камера
и
т.д.
)
.
В
целом
время
работы
телефона
вполне
нормальное
,
особенно
если
сравнивать
с
таковым
у
старшей
модели
KC910
.
Но
вот
конкуренту
Philips
Xenium
X800
KP500
проигрывает
по
этому
параметру
.
Время
полного
заряда
батареи
составляет
около
2
часов
.
К
содержанию
>
>
>
Коммуникационные
возможности
В
аппарате
реализован
Bluetooth
версии
2.1
с
поддержкой
EDR
.
Реализована
поддержка
основных
профилей
,
таких
,
как
:
Headset
,
Handsfree
,
Serial
Port
,
Dial
Up
Networking
,
File
Transfer
,
Object
Push
,
Basic
Printing
,
Phone
book
access
,
A2DP
,
AVRCP
.
Телефон
без
проблем
работает
с
другими
устройствами
,
в
частности
,
в
плане
передачи
данных
в
обоих
направлениях
,
также
не
замечено
и
проблем
при
работе
с
гарнитурами
.
Передача
данных
между
телефоном
и
ПК
происходит
по
USB
версии
2.0
,
так
что
большой
объем
информации
можно
загружать
на
карту
памяти
и
не
прибегая
к
использованию
кард-ридера
.
При
подключении
к
ПК
в
телефоне
возможно
выбрать
режим
работы
,
это
либо
съемный
диск
(
память
телефона
и
карты
памяти
становятся
доступны
на
ПК
без
установки
дополнительных
драйверов
)
,
либо
PC
Suite
,
либо
PC
Internet
(
для
работы
аппарата
в
качестве
модема
)
,
либо
режим
синхронизации
музыки
.
Для
GSM-сетей
предусмотрена
поддержка
GPRS
класс
12
и
EDGE
класс
10
.
К
содержанию
>
>
>
Память
и
карты
памяти
В
аппарате
около
48
Мб
собственной
памяти
,
доступной
пользователю
.
Также
телефон
может
работать
с
картами
памяти
формата
microSD
с
возможностью
их
горячей
замены
.
Гарантирована

Здесь

аппарата
или

.
Просмотр

осуществляется
в
альбомном
режиме
,

,
что

по

вы

на

.
Просмотр
возможен
либо
в

(

)
,
либо

более
крупных

,
когда

выводится
по

.
И
,

,

можно
просматривать
по
одной
на
полный
экран
.
В
галерее
возможно
выбрать
сортировку
по

,

или

.
Также
сразу
возможно
отправить
картинки
по

,

или

.
Кроме

,
предусмотрена

на

.
При

вы

пролистывать

на

иконки
со
стрелками
,
либо
просто
пролистывая
изображения
,
а
именно
быстро
проводя
пальцем
по
экрану
в

.
При

появляются

,
а
также
для

слайд-шоу
,
быстрой
установки
изображений
в

на

,

на
другие

,
а
также
доступа
к
редактору
изображений
.
Редактор
изображений
прост
и
понятен
в

,
имеет
в

все
необходимые

для

на

.
Конечно
,
о

не
идет
,
но
что-то
подправить
или
же

,
текст
,

фото
,
вполне
можно
.
На

,

,

,

.
Аппарат

.
Вся

,

.
Качество

,
про

,

.

.
К

Меню
На
главном

в

с

,
переназначить

к

.
Здесь
находится
все

необходимое
:
вызов
клавиатуры

номера
,
телефонная

,
сообщения
и
доступ
к
главному
меню
.
В

экрана
,

,
находится

,
на

отображается
вся
необходимая

.
Нажатие

выводит

,
на

доступна

о

и

,
уровне

,
причем
в

(

идет

:

,

,

)
.
Здесь

,

,
а

.
Кроме

,

,

,

,

.
Всего

.
В

,

.
В

,
и

.
Еще

,
что

,
и

,

.
Кроме

,
виджеты

,
для

достаточно

телефон
и

.
Также
в
аппарате
есть

,
который

,
на
нем

.
С

,

.
Главное

В

Первый

,

.
В

,

.
Подменю

.
Главное

,

.
Нажатие

,

9

,
которые
можно
сортировать
в

на
свое

.
На
второй

(
и

,

)
,

,

,

.
Всего

,

сможете
открыть
.
Меню

полностью

,

идет
в
комплекте
.
Все
иконки
и

крупные
,
так

в

не

.
Пролистывать

можно
или
с

боковой
полосы
прокрутки
,
которая
тонкая
и
не
очень
удобна
в

,
или

просто
быстро
проведя
по
экрану
в
любом
месте
.
При
запуске
Java-приложения

,

.
Жаль

,

,
что

,

,

с

.
Ввод

.
Для

:

,

,

-

месте
экрана
.
Распознавание
рукописного
ввода
работает

,
так

,

,

.
Более

,
которых

также

.
При

,

.
Так
,

,

.
Виртуальные

,

,

,

.
Второй

.
В

,
сделано

того
,

,

,
так

.
Честно

,

,

,

,

,
и
на

,
не

удобно
,
так

и
нередки
ошибочные

.
В

имеется

,
который

.
Работает

,

,
дольше

изменения

.
Кроме
основных

акселерометр

и
в

предустановленных

предустановленных
,
так
и
из
пользовательских
.
Кроме
этого
,
в
настройках
возможно
выбрать
размер
шрифта
меню
.
К
содержанию
>
>
>
Конкуренты
Конкурент
у
LG
KP500
на
данный
момент
один
,
это
Philips
Xenium
X800
.
Если
сравнить
их
в
лоб
,
то
получается
достаточно
интересная
ситуация
.
Так
,
у
X800
более
качественные
материалы
корпуса
,
лучше
по
качеству
экран
,
также
немного
лучше
камера
,
дольше
время
автономной
работы
и
ниже
стоимость
.
В
то
же
время
у
KP500
более
проработанный
интерфейс
,
который
позволяет
отказаться
от
стилуса
,
более
крупный
экран
,
корпус
у
него
тоньше
,
а
также
присутствует
многозадачность
.
Как
видите
,
у
обоих
аппаратов
достаточно
плюсов
,
так
что
здесь
надо
решить
,
что
для
вас
важнее
,
да
покрутить
оба
аппарата
в
руках
.
К
тому
же
,
немаловажным
фактором
в
принятии
решения
будет
являться
цена
,
которая
у
X800
всегда
будет
ниже
,
так
как
в
компании
Philips
хорошо
отслеживают
ситуацию
на
рынке
,
быстро
опуская
цены
на
свои
продукты
в
случае
необходимости
.
К
содержанию
>
>
>
Впечатления
Аппарат
обеспечивает
хорошее
качество
связи
,
ничем
не
уступая
современным
GSM-телефонам
,
а
даже
немного
превосходя
большинство
из
них
.
Громкости
динамика
и
чувствительности
микрофона
хватает
для
разговора
практически
в
любых
условиях
.
В
аппарате
реализована
64-тональная
полифония
,
в
качестве
сигнала
вызова
можно
выбрать
одну
из
предустановленных
мелодий
или
же
выбрать
свою
композицию
в
формате
MP3
.
Громкость
сигнала
воспроизведения
вызова
у
аппарата
чуть
выше
средней
,
а
динамик
спокойно
ведет
себя
даже
на
максимальной
громкости
.
Вибровызов
средний
по
силе
.
Компании
LG
удалось
выпустить
сравнительно
недорогой
и
интересный
аппарат
с
полноценным
touch-интерфейсом
.
Да
,
конечно
,
чтобы
снизить
конечную
стоимость
продукта
,
производителю
пришлось
пойти
на
компромисс
и
убрать
или
же
снизить
качество
некоторых
функций
.
Первое
,
на
что
обратит
внимание
пользователь
,
-
это
использование
недорого
пластика
в
корпусе
аппарата
,
также
наверняка
ему
бросится
в
глаза
невысокое
качество
камеры
телефона
,
проблемные
стороны
которой
прекрасно
видны
даже
на
экране
телефона
.
Но
,
несмотря
на
отрицательные
моменты
,
в
телефоне
достаточно
и
положительного
заряда
.
Так
,
покупателям
наверняка
придется
по
душе
крупный
сенсорный
дисплей
с
высоким
разрешением
,
недостатки
которого
заметить
можно
,
если
только
сравнить
непосредственно
экран
со
старшей
моделью
KC910
.
Уверен
,
что
пользователям
придется
по
душе
и
интерфейс
аппарата
,
полностью
заточенный
под
управлением
пальцами
,
так
что
необходимость
в
стилусе
отпадает
,
хотя
,
конечно
,
для
некоторых
задач
он
и
пригодится
.
Конечно
,
пользователь
обратит
внимание
и
на
неплохую
функциональность
аппарата
,
ведь
здесь
есть
и
MP3-проигрыватель
,
и
FM-радио
,
и
поддержка
Java-приложений
,
причем
с
возможностью
их
работы
в
фоновом
режиме
.
Стоимость
LG
KP500
на
данный
момент
составляет
около
10000
рублей
(
около
370
долларов
США
)
,
что
в
целом
вполне
оправданно
за
данный
продукт
.
