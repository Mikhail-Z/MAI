Камера
Пятимегапиксельная

без

,

,

,
все

.
Видеть
такое
во

,

странно

.
При

,

,

,
на

ноутбука
-
зачастую

.
Поддерживается
съемка
видео
(
720p
)
,
автофокусировка

медленно
,

.
Плюс
,

Вспышка

.
Есть

,

(

,
не

)
,

,
четырехкратное

.
Еще

,
камера

о
чем
,
очень
жаль
.
На

,

.
