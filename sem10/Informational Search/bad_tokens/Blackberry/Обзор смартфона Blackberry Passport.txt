.
У

.
Это

.
В

и

,

3
(
9900
)
,

,

–

.

Потом

и

,

не

то

,

.
В

не

,

с

.
Учитывая
,
что
компания
выживает
,
не
в

,
можно

радоваться
,

к

.
В

–

,

,

.
В

.
Посмотрите

,

,

.
Из

,

–

,

,

,

.
В

,

,

,

,

.
Для

,

,

.
Сказать
,

,

.
Не

,

,

.
Но

,

.
Давайте

,

и

.

Дизайн
,

,

Пожалуй
,

,

,

.
Итак
,

.
Если

,

,

,

,

,

.
Может
,

,

в
руках
,

не

.
Особенно
тяжело
то
,
что

большое
на

,

,

.

.
В
руке

,

,

.
Вес

С

.

,

.
Но

,

.
Заднюю

,

Предполагается

.
Но

.
Если

.
Они

,

.
Это

,

.
Этот

,

,

.
За

,

,

.
Следы

рук

,

.
На

,

,

.
Для

,

,
что

(

,

)
,
можно

,

.
Я

.
Теперь

.
На

,
ее

,

.
На

–

Такое

.
Интересно
,

,

.

,

.

.

.

.
В

,

.

увидите

игровое
поле
,
причем
обрезает
его
как
по
боковым
сторонам
,
так
и
сверху
и
снизу
.
Но
разработчики
предусмотрели
это
,
можно
масштабировать
экран
приложений
,
появляются
полосы
,
но
зато
вся
картинка
видна
.
Удобно
,
что
для
каждого
приложения
в
памяти
сохраняются
настройки
масштаба
,
который
вы
задали
движением
пальца
.
Технические
характеристики
экрана
следующие
–
1440х1440
точек
,
IPS
,
поддерживает
до
10
одновременных
нажатий
.
Качество
картинки
хорошее
,
запас
яркости
достаточный
,
на
максимуме
даже
чересчур
.
На
солнце
экран
остается
читаемым
.
Мне
этот
аппарат
понравился
по
качеству
картинки
,
никаких
вопросов
не
вызывает
,
но
при
просмотре
видео
или
фотографий
остаются
черные
полосы
,
то
есть
часть
полезной
площади
пропадает
.
Это
телефон
для
работы
,
с
другими
задачами
он
справляется
заметно
хуже
из-за
геометрии
экрана
и

.

Клавиатура
Здесь
не
совсем
обычная
клавиатура
,
начну
с
того
,
что
есть
только
три
ряда
клавиш
,
в
других
Blackberry
их
больше
.
Меньше
число
клавиш
заставило
поместить
на
некоторые
кнопки
по
две
русских
буквы
,
это
не
всегда
удобно
,
но
если
вы
пользуетесь
Т9
,
то
вполне
комфортно
.
Другой
момент
–
это
то
,
что
клавиатура
не
просто
механическая
,
она
поддерживает
распознавание
жестов
.
Например
,
начиная
набирать
слово
,
вы
видите
подсказку
на
экране
,
и
проведя
пальцем
вверх
,
выбираете
его
(
три
зоны
,
три
слова
соответственно
)
.
Более
того
,
на
клавиатуре
поддерживаются
простые
движения
,
чтобы
стереть
написанное
слово
,
достаточно
провести
справа
налево
по
клавишам
.
Удобно
?
Довольно-таки
.
Посмотрите
,
как
это
работает
в
официальном
видео
.
Думаю
,
что
я
,
как
и
многие
другие
люди
,
использовавшие
Blackberry
в
прошлом
,
уже
отвыкли
от
механических
клавиатур
,
было
тяжело
снова
привыкать
.
На
это
нужно
больше
времени
,
чем
я
провел
с
Passport
–
честно
признаюсь
,
что
привыкнуть
сложно
,
надо
себя
заставлять
.
На
сенсорной
клавиатуре
я
набираю
заметно
быстрее
,
и
мне
это
удобнее
.
Поэтому
данное
решение
точно
для
тех
,
кто
любит
механические
клавиши
–
таких
пользователей
на
планете
осталось
немного
,
что
показывает
статистика
продаж
того
же
Blackberry
.
Обещают
,
что
этот
аппарат
появится
в
России
,
тогда
мы
сможем
увидеть
официальную
локализацию
,
но
пока
посмотрите
,
как
выглядит
лазерная
гравировка
на
клавишах
,
она
довольно
удобная
,
хотя
буквы
лежат
на
поверхности
,
а
не
как
в
оригинальном
варианте
,
внутри
клавиш
.

Аккумулятор
В
2014
году
в
фаблетах
стали
повально
ставить

около
3000
мАч
,
в

аспекте
Passport
находится
в
тренде
–
здесь
Li-Ion
аккумулятор
3450
мАч
.
По
заявлению
производителя
,
он
способен
обеспечить
время
работы
в
режиме
разговора
до
23
часов
(
3G
)
,
до
444
часов
в
ожидании
(
3G
)
,
проигрывания
музыки
до
84
часов
,
в
смешанном

аппарат

25
часов
.
Начну
от
обратного
,

полной
зарядки

–
около
3.5
часов
.
При

наполовину
в
течение
суток
,
то
есть
два
дня
работы
вполне
достижимы
.
Но
даже
при
очень
активном
использовании
это
будет
один
день
уверенной
работы
.
Вовсе
неплохо
и
типично
для

,

.

Камера
В

Blackberry
камеры
всегда
были
среднего
уровня
,
лучше
,
чем
у
китайцев
,
но
заметно
хуже
,
чем
у
других
крупных
производителей
.
Уже
многократно
было
так
,
что
нам
обещали
новое
качество
снимков
,
рассказывали
про
технологии
,
но
на
деле
все

очень
грустно
.
В
Passport
ситуация

–

получилась
очень
достойной
,
несмотря
на
ряд
шероховатостей
.
Например
,
у
меня
в
какой-то
момент
взбесился
автофокус
,
он
просто
прекратил
работать
,
пришлось
перезагружать
аппарат
.
Итак
,
камера
13-мегапиксельная
,
BSI
,
f=2.0
,
есть
оптическая
стабилизация
.

видео
возможна
с

до
60

в

.
Посмотрите
на

снимков
,
они
вполне
удачны
как
в

время
,
так
и
в
темное
,
хотя
от
последних
ждешь
большего
,
но
это
очень
и
очень
неплохо
.
Пример
видео
1
(
MP4
,

)

Программные

Blend

В

Blackberry
OS
10.3
,

знакома
всем
пользователям
аппаратов
последних

,
останавливаться
на
ее
подробном
описании
нет

,

отдельный

.
Обзор
операционной
системы
Blackberry
10
Уникальная
функция
Blackberry
Blend
позволяет
превратить
компьютер
или
планшет
в
удаленный
терминал
для
работы
с
вашим
телефоном
,
то
есть
дать
большой
экран
,
на
котором
удобнее
работать
с
данными
.
Учитывая
,
что
в
Passport
экран
большой
,
тут
это
не
так
интересно
,
как
на
других
моделях
.
Корни
этой
технологии
стоит
искать
в
первом
планшете
компании
–
в
Playbook
не
было
почтового
клиента
,
и
работать
с
почтой
можно
было
только
подсоединив
к
планшету
телефон
.
Тогда
эта
концепция
по
понятным
причинам
провалилась
,
но
в
новой
реинкарнации
имеет
все
шансы
стать
полезной
.
Итак
,
вы
можете
подключиться
к
ПК
или
планшету
по
USB
,
Wi-Fi
,
сотовой
сети
,
и
тогда
на
большом
экране
получите
доступ
к
почте
,
календарю
,
сообщениям
BBM
,
SMS
,
контактам
и
мультимедийным
файлам
.
Вносите
правки
,
и
они
тут
же
передаются
на
телефон
.
Можно
скопировать
фотографии
на
компьютер
.
В
этом
же
направлении
идет
Apple
,
совмещая
работу
на
ПК
и
телефон
(
можно
принимать
звонки
и
писать
SMS
)
.
В
Blackberry
сделали
более
продвинутую
,
интересную
систему
.
Посмотрите
,
как
это
работает
,
в
официальном
видео
.
С
предыдущими
моделями
Blend
пока
не
работает
,
только
с
Passport
и
P9983
.

Впечатления
Это
один
из
немногих
аппаратов
,
в
которых
поработали
над
качеством
звука
и
звучания
.
Во-первых
,
увеличили
громкость
динамика
.
Утверждается
,
что
он
звучит
на
18
процентов
громче
,
чем
в
HTC
One
M8
,
и
на
350
%
громче
,
чем
в
Galaxy
S5
.
Звучит
,
как
фантастика
,
надо
слушать
своими
ушами
,
что
в
ближайшие
дни
я
и
планирую
сделать
.
Во-вторых
,
у
каждого
аппарата
есть
комфортная
зона
при
разговоре
в
обычном
стиле
,
когда
вы
прикладываете
телефон
к
уху
–
это
зона
,
в
которой
вы
четко
и
хорошо
слышите
динамик
.
Чем
больше
такая
зона
,
тем
более
небрежно
можно
держать
телефон
или
дергать
его
–
точность
позиционирования
не
играет
роли
.
В
Blackberry
создали
технологию
Natural
Sound
,
которая
на
70
процентов
увеличивает
эту
зону
.
Пожалуй
,
по
громкости
звонка
Passport
может
соперничать
только
с
Vertu
–
чистый
,
красивый
звук
,
но
он
еще
и
безумно
громкий
.
Можно
выкрутить
громкость
наполовину
,
вы
все
равно
будете
его
слышать
.
Виброзвонок
средний
,
но
его
не
пропускаешь
,
вибрации
довольно
заметные
.
Продажи
уже
идут
в
30
странах
мира
с
25
сентября
.
Цена
без
контракта
в
США
составит
599
долларов
(
плюс
налоги
)
,
в
Канаде
она
будет
чуть
больше
,
а
вот
в
Европе
(
например
,
Германии
)
в
пересчете
на
доллары
будет
равна
835
долларам
,
в
Великобритании
–
870
долларам
.
В
России
на
сером
рынке
аппарат
можно
купить
за
45-50
тысяч
рублей
,
и
говорят
,
что
интерес
к
новинке
довольно
заметный
.
Начну
с
того
,
что
эта
модель
точно
не
для
всех
.
Привыкнуть
к
клавиатуре
можно
довольно
быстро
,
но
этот
телефон
точно
для
двух
рук
,
как
и
многие
Blackberry
до
того
,
вне
зависимости
от
размера
.
Учитывая
,
что
на
рынке
предостаточно
фаблетов
,
эта
модель
не
вызывает
отторжения
весом
и
размерами
,
а
плюсы
в
виде
времени
работы
,
качества
сборки
и
материалов
очевидны
.
То
,
что
клавиатура
поддерживает
жесты
,
например
,
в
том
же
браузере
,
это
плюс
.
Непривычно
,
но
привыкаешь
,
и
это
становится
удобным
.
Камера
впервые
для
Blackberry
стала
вровень
с
другими
крупными
производителями
.
В
Blackberry
говорят
о
том
,
что
в
2015
году
точно
появится
продолжение
этого
устройства
.
То
есть
концепция
такого
рода
фаблета
получит
развитие
.
Это
очень
необычное
устройство
,
которое
точно
не
предназначено
для
массового
рынка
.
Возможно
,
что
оно
не
придется
по
душе
тем
,
кто
любит
классику
от
Blackberry
.
Но
одно
могу
сказать
наверняка
–
устройство
интересное
и
по
ряду
характеристик
не
имеющее
на
рынке
аналогов
.
Например
,
по
качеству
звучания
,
громкости
.
Очень
надеюсь
,
что
эти
технологии
будут
перенесены
и
в
другие
модели
,
это
даст
толчок
продажам
Blackberry
,
да
и
заставит
почесаться
другие
компании
.
Благодарим
за
предоставленное
устройство
сайт
Blackberries.ru
.
