После
выхода

Dell
Axim
X5
минимальная
цена
на
карманные
компьютеры
Pocket
PC
упала
,
причем
довольно
серьезно
.
Другим
производителям
пришлось
в
спешке
представлять
на
рынок
бюджетные
КПК
стоимостью
не
более
$
300
.
Ответом
от
лидера
рынка
,
компании
HP
,
стала
миниатюрная
модель
iPaq
1910
(
до
сих
пор
она
остается
самой
маленькой
из
всех
Pocket
PC
)
.
Летом
2003
года
состоялась
премьера
наладонника
iPaq
1940
,
продолжения
линейки
,
который
отличался
от
предшественника
поддержкой
bluetooth
,
процессором
от
Samsung
,
увеличенным
объемом
оперативной
памяти
и
поддержкой
SDIO
.
Чуть
позже
HP
представила
модель
iPaq
1930
,
которая
отличается
от
iPaq
1940
отсутствием
поддержки
bluetooth
,
частотой
процессора
и
,
конечно
,
меньшей
ценой
.
Как
уже
было
сказано
выше
,
КПК
имеет
наименьшие
размеры
среди
всех
Pocket
PC
представленных
на
рынке
(
113,3х69,8х12,8
мм
)
.
Внешний
вид
устройства
(
как
и
размер
)
не
изменился
со
времени
iPaq
1910
:
наладонник
имеет
классические
для
iPaq
обтекаемые
формы
и
корпус
,
выполненный
из
серебристого
пластика
.
Качество
сборки
на
высоте
,
детали
подогнаны
плотно
,
если
сильно
сжать
устройство
в
руке
,
то
оно
не
скрипит
.
Единственное
слабое
место
корпуса
–
это
крышка
аккумуляторного
отсека
:
она
сделана
из
очень
тонкого
пластика
и
неплотно
прилегает
,
продавливается
если
на
нее
сильно
надавить
пальцем
.
Вес
устройства
в
124
г
не
ощущается
при
ношении
устройства
практически
где
угодно
,
в
том
числе
и
на
веревке
(
на
КПК
есть
специальное
ушко
для
ее
крепления
)
.
Вверху
на
лицевой
части
расположен
один
светодиод
,
он
же
является
и
кнопкой
питания
устройства
.
Светодиод
мигает
синим
цветом
при
включенном
bluetooth
,
оранжевым
–
при
зарядке
устройства
(
даже
если
bluetooth
включен
)
,
зеленым
–
при
активном
или
пропущенном
напоминании
.
Внизу
расположены
аппаратные
кнопки
,
отвечающие
за
вызов
календаря
,
контактов
,
почтового
клиента
и
кнопка
возврата
на
стартовый
экран
.
Они
расположены
дугообразно
по
две
слева
и
справа
от
джойстика
.
Несмотря
на
то
,
что
разработчики
в
первую
очередь
думали
об
их
дизайне
,
а
уже
во
вторую
–
об
удобстве
,
пользоваться
ими
достаточно
удобно
,
в
том
числе
в
играх
.
Но
из-за
близкого
их
расположения
сохраняется
большая
вероятность
случайного
нажатия
соседней
кнопки
.
Сам
джойстик
маленький
и
круглый
,
имеет
четкий
отклик
при
нажатии
в
разных
направлениях
(
не
такой
мягкий
как
,
например
,
в
модели
HP
iPaq
2210
)
.
Многие
находят
,
что
пользоваться
им
не
слишком
удобно
из-за
того
,
что
нажимать
кнопку
«
Enter
»
,
которая
находится
в
центре
джойстика
,
не
всегда
получается
отдельно
от
курсорных
кнопок
джойстика
.
На
левом
торце
расположены
следующие
элементы
:
ушко
для
крепления
веревки
,
окошко
инфракрасного
порта
,
кнопка
диктофона
(
она
болтается
,
если
потрясти
устройство
)
,
отверстие
для
мягкой
перезагрузки
устройства
.
На
правом
–
только
защелка
для
аккумуляторного
отсека
.
На
верхнем
торце
располагаются
:
вход
для
наушников
(
для
его
использования
придется
использовать
переходник
который
идет
в
комплекте
;
смотрится
это
,
скажем
,
не
слишком
эстетично
)
,
SD/MMC
слот
,
гнездо
для
стилуса
.
Слот
SD/MMC
поддерживает
SDIO
,
то
есть
в
перспективе
устройство
можно
оснастить
периферийным
оборудованием
в
формате
SD
–
например
,
Wi-Fi
адаптером
или
GPS
приемником
.
Правда
,
подобные
аксессуары
дороги
и
их
проблематично
найти
в
продаже
.
Отсутствие
слота
CompactFlash
становится
не
таким
критичным
на
фоне
падения
цены
карточки
SD
объемом
512
Мб
до
$
160-180
.
Стилус
имеет
средние
размеры
,
пользоваться
им
удобно
.
На
нижнем
торце
располагается
вход
для
подключения
кабеля
синхронизации
и
док-станции
.
Использовать
его
можно
исключительно
для
синхронизации
,
подключить
к
нему
периферию
не
получится
–
т.к
.
не
поддерживается
COM
порт
.
Экран
Безусловно
,
самая
спорная
часть
данного
КПК
–
это
экран
.
Оснащение
КПК
новым
процессором
и
bluetooth
при
сохранении
прежней
цены
в
$
300
привело
ухудшению
экрана
(
для
подержания
баланса
цены
и
себестоимости
)
.
Если
iPaq
1910
мог
похвастаться
одним
из
лучших
экранов
на
КПК
,
то
новый
экрана
имеет
желтый
оттенок
,
если
смотреть
на
него
под
углом
.
Работе
с
устройством
это
не
мешает
.
Желтизна
видна
,
если
на
белом
фоне
(
в
меню
)
посмотреть
на
него
немного
снизу
.
Если
,
например
,
проделать
ту
же
процедуру
на
синем
фоне
,
то
экран
выглядит
нормально
.
Мое
мнение
–
желтизна
не
мешает
нормальной
работе
с
устройством
(
если
держать
его
в
руке
,
то
смотришь
на
экран
как
бы
немного
сверху
,
при
таком
положении
никакой
желтизны
нет
)
.
Практика
показала
,
что
если
человеку
не
сказать
,
что
«
экран
желтый
»
,
то
сам
он
этого
не
заметит
в
подавляющем
большинстве
случаев
.
В
остальном
экран
яркий
,
с
отличной
цветопередачей
и

.
Настройки

имеют
четыре
уровня

.
Практически
в
любых

всего
с
ним
работать
на
минимальном
,
первом
уровне
.
Однако
отметим
,
что
он
лишком
яркий
для
работы
с
ним
в
темноте
.
