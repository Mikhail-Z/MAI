:
)

,

Очень

,

.
Думаю

Крайне

Есть
,

,

,

.
Еще

:

Экран

За

,

,
в

,

,
портативными
«

»
,

.
Олеофобное
покрытие
есть
.
Не

,
но

,

M1
Note
–

.
Отпечатки

,

,

.
По

.
Остальная

.
Цветов

:

,

,

,

и

.
Однако

только

.
Остальные
позже
.
Пластик
обычный
,

.
За

на
тыльной

появились

,

.
На

.
Правильно
,

не

с
черным

–

.
Задняя

.
В

,

обычные
,
но

и

.
Речевой

,

сеточкой
.
Его

высокая
,

,
тембр

:
хватает
и

,
и

,

,

.
Если

с
MX4
и
MX4
Pro
,
то
у

,

,

,

.
Справа
–
датчики

и

.
Если

,

.
Во

,

«

»

.
Явный

,

,

,

.
Еще

.
В

.
Она

)
.
Подсветка

.
Во

.
Выполняет

.
Если

Длительное

.
Достаточно

,

,

.
Внизу

.
Спикер

,

.
Впрочем
,

.
Сверху
–
3.5

аудиовыход

,

,

.
Кнопка

.
Она

,

,

,

.
Возможно
,

.
Слева

,

Имеют

»
.

–

.
Глазок

,

На

,

С

Производительность

.
Что

,

.
