,

.

Каждая

Дизайн

Дизайн

Качество

.
На

,

,

,

,

Meilan

.
В

В

В

,

.
Габариты
Аппарат

,

6
''
,

.
Да
,

,

.
Использовать

Кнопки

,

,

,

.
Сверху
–

(

)
.
Под

.
Кнопка

,

.
Экран
В

6
''

.
Дисплей

,

.
Есть

«

,

,
а

.
Олеофобное

,

.
Платформа
,

.
Она

.
Это

,

.
Графика

.
В

3

.
Есть

.
Повторю

,

–

,

,

,
3/32

и

.
Одна

по

и

–

.
Меньше

,

и

модельный

.
В

с

M3
Max
это
3/64
ГБ
.
Камера
Обе

,

,

,

не

с

в

Note
.
Изменился

.
Соответственно
,

в

M3
Max

,

и

.
Я
не

,

же

недорогой

,
просто

на

и

.
Как
и

от

,

есть

(
это

)
,
а

то
,
что

не
умеет

в

(
это
плохо
)
.
фронтальная
камера
Интерфейсы
В
аппарате
есть

двухдиапазонного
Wi-Fi
и

.
В
4G

смартфон
работает
во

,

Band

.
Bluetooth

4.1

LE
(

)
.
Аккумулятор
Емкость

в

Max
–
4100

,
то

же
,

в

.
С

,

,
с

-

,

Max
,
емкость
и

,

.
Как

,

,

!
Аппарат

(

)
.
Заключение
В

Meizu

1
700

.
Стоимость

,

–
25
000-26
000

,

–

.
Как
и

M3e
,

,

,

,

,

,

(

в

,

и

,

)

.
Другой

,

–

,

,

)
.
Пару

»

,

,

»
.
Сейчас

ровно

:

,
есть

в

(

и

)
и

(

)
,

(

)
и

(

)
,

(

)

(

)
,

,

,

,

оно
