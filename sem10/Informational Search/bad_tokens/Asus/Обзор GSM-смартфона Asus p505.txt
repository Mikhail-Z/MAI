.

(

)

.
Видно
,

.
Не

.

.
Многие

.

.

.

.
Можно

.
В

.
Из

,

.
Компания

)
,

Siemens

,

(

,

,

)
.
А

Asus
близка
к

,

–
смартфон
Asus
p505
,

к

.
Итак
,

,

.
Все

,

,

,

,

,

,
производители
сотовых
терминалов
.
Для
новичков
отмены
моделей
и

–
вполне

.
Возрастает
конкуренция
,
все
больше
появляется
интересных
моделей
.
Долгое

Asus
A8100
,

2003

.
В

,
и

Asus
p505
.
Сейчас

по
связи
США
(
FCC
)
,
а
значит
,

ее

месяц
,
максимум
–
два
.
Итак
,
что
такое

p505
?
Это

с

и

Mobile
2003

Pocket
PC

.
Самый

–
HTC
Magician
(

)
,

–
модель

Ericsson
p900/p910i
.
Модель

Asus

находится

,
но

,

p505

Sony
Ericsson
p900
или

,

HTC
Magician
.
У

и

,

.
Размеры

,

,

.
Модель

,

.
Что

,

,

,

,

.
Нельзя

,

.
Скорее
–

.
Одним

,

.
Хотя
,
внешняя

–

(

,

)
,

,

,

.
Корпус

.
Ощущения

.
Качество

.
Разве

,

,

.

.
Габариты

.

.
Поговорим

.
Флип

.
Его

При

,

.
Итак
,

(

)
,

.
Впервые

,

.
Это

,

,

.
На

,

,

профиля
,
текущее
время
(
верхняя
строка
иконок
)
,
название
оператора
,
дата
(
под
верхней
строкой
)
.
На
заставку
можно
поставить
любую

или

gif
,
например
такую
(
среди
предустановленных
)
:
На

можно
вывести

(

,

)
,
вместо

.
Внизу

,

.
Нажав

.
Перемещение

)
.
Клавиши

,

(

)
.
К

,

можно

,
что-то

,

с

(

)
.
То
есть
завести
новый
контакт
,
отредактировать
напоминание
,
ответить
на

нельзя
.
Для

придется
открыть
весь
экран
.
Минус
,
но
не
очень
значительный
.
Можно
–
просматривать
контакты
(
набирать
оттуда
номер
)
,
просматривать
календарь
и
задачи
,
изменить
профиль
,
просмотреть
журнал

,
прочитать
SMS
,
сделать
запись
на
диктофон
,
управлять
встроенной

,
не
так
уж
и
мало
.
Напомним
,
что
на
Pocket
PC

UI
появился
впервые
в
Asus
p505
.
Вероятно
,
что
в
следующих
моделях
он
будет
доработан
,
расширен
,
появится
текстовый
ввод
,
но
и
то
,
что
есть
–
уже
неплохо
.
Кстати
,
модель
Asus
p505
будет
поставляться
и
под
марками
оператора
Vodafone
,
так
что
внешний
вид
интерфейса
в

версии
непринципиально
будет
отличаться
(
цвета
,
расположение

)
.
И
последнее
–
подсветка
клавиатуры
отличная
,
в
темноте
можно
без
проблем
разобрать
все
цифры
(
хотя
,

букв
,

другого
и
не
ожидали
)
.
Открывая

возвращаемся
к

интерфейсу

PC
(
вернее
–
к

,
отсюда
можно

программы

к

)
.
Экран
с
диагональю
2.8
”
(

,

65

,
разрешение
320x240
точек
)
постепенно

PC
,

.
Сравнимый
по

и

Magician
,
вместе
с
тем

лучше
,
чем
у

MPx
и

S2
(

,

,
минимальный

)
.
В

–
в
своем
классе
он

,

моделям

в

p910
по

(
как
с

флипом
,
так
и
с
открытым
)
.
Но
надо
понимать
,
что
сокращение
диагонали
экрана
вызвано
в
первую
очередь
сокращением
линейных
размеров
устройства
(
чем
меньше

поверхности
устройства
,

меньше
экран
)
.
Под
экраном
в
линию
расположены
четыре
кнопки
:
отбой
,
прием
,
календарь
,
контакты
.
Из-за
конструкции
и
расположения
джойстика
–
смартфон
мало
пригоден
для

,

можно
.
Над
экраном
расположены
:
динамик
,
два
световых
индикатора
,
клавиша
включения
(

Пятипозиционный

В

Во

Все

Чуть

.
Матрица

1.3

.
Встроенная

,

,

При

Экран

.
Картинка

.

Обычный

Можно

.
Доступен

.

.

,

.
MMS
.
Снимок

.
Доступные

разрешения
-

160x180

до
320x240

.
Burst
.
Серия

,

.
Доступные

-

80x60

.
Composition
.
Композиционные

.
Доступные

-

160x120

1280x960

.

.
Видео

,

.
Разрешение
:
320x240

.
Caller
ID
.
Фотографии

к

.
Разрешение

.
Из

:

,

,
экспозиция
,

.
В

,

настроек
Asus
p505

от

Asus
A730
.
Качество

среднее
,
оно

Motorola
MPx
,
лучше

MPx220
,

с

(

)
.
На

Asus
p505
–

,

.
Программа

,

,

,

При

.

.
На

Задний

,

.

.

Аккумулятор
Под

Оно

Заявленная
