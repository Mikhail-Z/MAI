AnyDATA
AML-100M

1x
450

.
Габариты

,

:
109х43х18

,

:
95

.
Аппарат
можно
спокойно
носить

кармане

,

.
Кроме
небольших

и
интересным

.
Опознать
в
нем
CDMA
аппарат
довольно
сложно
,
все

впервые
брал
его
в
руки
,
были
уверены
,

GSM
трубка
,
а

,
что
это
не
так
,

удивлялись
.
По
поводу

людей
,

,

50/50
.
Кому-то

понравился
,
а

,

.
Интересно
,

всем
женщинам
телефон
нравился
сразу
.
Корпус
телефона
выполнен
из
пластика
,
сборка

высокого
уровня
,
все
детали
подогнаны
вплотную
друг
к
другу
.
Покрытие
лакированное
,
цвет
корпуса
-
сочетание
черного
с
серебристым
.
Поверхность
черного
цвета
быстро
заляпывается
,
если

лицевой

почти
не
видно
,
то

задней
отпечатки

видны

под
любым
углом
.
На
левой
боковой
поверхности
аппарата
располагаются
две
клавиши
регулировки
громкости
,
ближе
к
нижней

находится
разъем
гарнитуры
,
прикрытый
резиновой
заглушкой
,
которая
закреплена
на
корпусе
.
На
правой
боковой

присутствует
только
клавиша

,
чтобы

случайное
нажатие
,
она
слегка
утоплена
в
корпус
.
Блокировка
происходит
после
нажатия
и
удержания

,
разблокировка
происходит
также
.
На
нижнем
торце
по
традиции
располагается
интерфейсный
разъем
,
прикрытый
резиновой
заглушкой
,
которая
закреплена
на
корпусе
,
а
на
верхнем
торце
находится
только
отверстие
для
крепления
ремешка
.
На
лицевой
панели
находится
небольшой
cSTN
дисплей
разрешением
128х128
точек
(
24х24
мм
)
,
способный
отображать
до
65
000
цветов
.
Честно
сказать
,
качество
дисплея
ниже
среднего
,
даже
дисплей
Ubiquam
U-200
(
о
нем
мы

ранее
)

лучше
.
Клавиши

серебристого
цвета
.
Функциональный
блок

в
себя

и

,

нее

«
ОК
»
.
Навигационная
клавиша

форму
овала
,

среднего
размера
,
работа
с

никаких
проблем
.
Остальные
клавиши
также
имеют
вполне
приемлемые
габариты
,
и
удобны
в
работе
.
Цифровые
клавиши

и
расположены
тремя
горизонтальными
рядами
.
На
первый
взгляд

неудобны
,
но
на
самом
деле
работать
с
ними
можно
вполне
комфортно
,
сказывается
то
,
что

из
корпуса

на
1
миллиметр
.
Символы
нанесены
не
на
сами
клавиши
,
а
на
подложку
.
Подсветка
клавиатуры
синего
цвета
,
более
яркая
в
центре
.
Но
,
не
смотря
на
то
,
что
крайние
символы
подсвечены
хуже
,
они
все
равно
остаются
читаемы
.
На
задней
поверхности
аппарата
располагается
динамик
для
воспроизведения
сигнала
вызова
.
Оставшуюся
часть
занимает
крышка
,
под
которой
находится
батарея
.
Крышка

из
тонкого

,
крепление

,

и

.
Аккумуляторная
батарея

,
емкостью
680
мАч
.
В

московской
сети
Skylink
аппарат
проработал
3
суток
при
20-30
минутах
разговора
в
день
,
если
время
разговоров
возрастет
,
то
телефон
будет
работать
порядка
1.5-2
дней
.
Время
полной
зарядки
батареи
составляет
около
2
часов
.
Меню
Доступ
в
меню
аппарата
осуществляется

на
кнопку
«
ОК
»
из
режима
ожидания
.
Главное
меню
представлено
в

,
одновременно

только

.
Подменю

двумя

рядами
иконок
,

иконок
(

)
.
Из-за

,

,

.
В

.
Шрифт

.
Локализация

выполнена
средне
,
есть

,

объясняется
не

удачным

.
Телефонная
книга
.
В

может

до
500

.
Для
каждого

можно
записать
имя
,
до
4

телефонов
,
создать
небольшой
текстовый

и
поместить
контакт
в

из
групп
.
Всего
в
телефоне
4

,

изменить

и
задать
сигнал
вызова
.
При
вводе
нового
контакта
,
предлагается
выбрать
,

,
в

или

R-UIM

.
Поиск
в
телефонной

осуществляется
по
имени
,
номеру
и
группе
.
Если

ожидания

,

(

)

книге
.
Также
в
телефоне
есть

набор

99
номеров
.
Кроме
этого
,
в
данном
разделе

,
набранных
и

.
Одинаковые

объединяются
,
кроме
номера
(

)

(

,

)
.
Сообщения
.
Доступ

»

.
Всего

100

.
Ввод

,

,

160

(

)
.
Поддерживается

,

9

.
Сигнал

,

одну

2

,

,

,

с

в
2
минуты
.
Звук
.
В

,

.
В

,

вибро
,

.
Всего
в

35
мелодий
звонка
,

то
,
что
ему
больше
нравиться
.
Дисплей
.
В

меню
можно
установить

одну

,

:

,

и

,

.
Кроме

,

в

,

(

)
,

текст
,
который

,
и

,

.
Все

,

,

,

»
.
В
данном
разделе
можно

и

.

.
Всего

,

Black

21
)
,

)

)
.
Графикой

,

.

.

,
время
,

,

)
,

.
В

.
Утренний
будильник
.
Будильник
один
,

,

(

3

)
,

,

и

.
Будильник
.
Здесь

5

,

,

)
,

,

.
Имеется

,

(

,

и

.
Также

,

Диктофон
.
Позволяет

.
К

.
Настройки
.
В

.
Здесь

,

,

.
Впечатления
Несмотря
на

антенну
,

с

не
наблюдалось
,
качество

.
Так

.
Громкости
динамика

,

.
А

,

,

.
Звучание
32-тональной

,

Вибровызов
средний

.
Модель

,

,

впечатление

,

.
Из

,

.
Данный

,

.
Данная

,

,

,

,

.
Официальные

июля
-

,
