Младшей

9

9

Значительно

,

Как

Причина

.
Корпус

Philips
9
@

.
Дизайн

,

Благодаря

На

,

.
В

с
9
@
9g

миниатюрным
,

—
100.7х44х15.7
мм
при
весе
77
грамм
.
Качество

и
сборки
на
высоте
:
все

«
на
совесть
»
,

друг

,

.
Ненужные

.
Боковые

Передняя

Кнопки

.
Функциональные

.
Пользоваться

,

.
Кроме

,
необходимо

к

.
Я
бы
назвала
такое
решение
неудачным
—
все
давно
привыкли
к
джойстикам
и
пятипозиционным

навигации
,
поэтому

управления

всем
.
Что
же
касается

,
к

:
кнопки
крупные

,
четко
отделены
друг
от
друга
,
что
позволяет
избежать
случайных
нажатий
,
и
имеют
небольшой
,
но
ощутимый
ход
.
Дисплей
и
область
вокруг
него
,
как
и
в
модели
9
@
9g
,

.
Правда
,

у
Philips
9
@
9f

.
Он

OLED-технологии

А

погоду

улице
изображение
очень
сильно

и
становится
едва
читаемым
,
однако
я
бы
назвала
это
вполне
оправданной
платой

длительный
срок

.
В

6
строк
текста
+
1

,
при

номера
—
до
4х
строк
+
1
служебная
.
Несмотря
на
то
,
что
по
функциональности
Philips
9
@
9f
уступает
старшей
модели
с

g
,
аккумулятор
у

—
Li-Ion
емкостью

.
По

В

.
Для

.
Меню
Главное
меню

из
9

,

3
ряда
по
3

.
Иконки

,

.
Для
перехода

меню
можно

,

.
Вдобавок

)
.
Значения

2-8

(

не

,

,
только

1

,
за
9
—
вызов
раздела
Java
,
за
0
—
вызов
WAP-браузера
,
а
за
«
решеткой
»
—
блокировка
клавиатуры
.
Телефонная
книга
.
В
записной
книге
нельзя
одновременно
использовать
и
номера
из
памяти
SIM-карты
,
и
контакты
из
телефона
—
только
что-то

.
Для

в
памяти

предусмотрено
1000
ячеек
.
При

заполнять

(
мобильный
,

,
рабочий
,

,
информация
,
номер
пейджера
,
e-mail
или
комментарий
)
,
однако
можно
создавать
несколько
записей
с
одинаковым
именем
.
Сразу
же
предлагается

абонента
в
одну
из
6
групп
.
У
каждой
группы
можно
изменять
название
,
назначать
ей
желаемую
мелодию
и

.
Присваивать
мелодии
и
картинки
отдельным

нельзя
.
Списки
вызовов
.
Список
вызовов
всего
один
,
в
нем

все
последние
звонки
.
По
типам
(
входящий
,
исходящий
,
пропущенный
)
они
различаются
пиктограммами
,
которые
отображаются
в
списке
возле
номера
.
В
памяти
сохраняется
номер
телефона
или
имя
абонента
(
если
он
есть
в
записной
книге
)
,
а
также
дата
и
время
звонка
.
Длительность
вызова
не
запоминается
.
Сообщения
.
Этот
раздел
поделен
на
две
категории
:
SMS
и
MMS
.
Создавать
сообщения

,
интерфейс

.
В
тело
SMS
можно

только
текстовые
смайлы
из
списка
,
а
MMS
,
как
всегда
,
позволяет
прикреплять
анимации
,
картинки
,
мелодии
и
видео
.
Для
SMS
предусмотрено
10
шаблонов
,
у
MMS
шаблоны
пустые
,
но
есть
возможность
создавать
собственные
.
Поддерживается
режим
предикативного
ввода
Т9
.
Игры
и
приложения
.
Здесь
содержатся
игры
и
java-апплеты
.
В
категорию
игр
вынесены
:
простенький
арканоид
под
названием
Brick
game
,
двухмерная
«
змейка
»
и
игра
под
названием
Mastermind
,
в
которой
требуется
за
10
попыток

,
какие
четыре
кубика
были
загаданы
.
Игра
Battlefield
,
которая

во
многих
аппаратах
Philips
,
была

в
раздел
java-приложений
.
Органайзер
.
Органайзер
у
Philips
9
@
9f
простой
,
но
продуманный
—
в

содержится
набор
стандартных

,

.
Календарь

,

.
Есть

план

.
События

,
а

(
название
,

,

,

,

,

,

,

.
Предусмотрена

)
.
Будильников

,

,

,

.
Есть

.
Калькулятор

.
Медиа

.
В

.
Внутри

Просматривать

)
,

.
Для

.
Для

1.8

,

.
Еще
один

—
отсутствие

mp3-файлов

в

.
Philips
9
@
9f
позволяет
воспроизводить
лишь
midi-файлы
.
Профили
.
Профилей
как

у
Philips
9
@
9f
нет
,

лишь
беззвучный

.
Зато
функция
вибровызова

с

.
Использовать

WAP-браузер
.
Возможности
браузера

стандартны
.
Он
позволяет
открывать
как
WAP
,
так
и
небольшие
HTML-страницы
.
Настройки
.
Раздел
настроек
поделен
на
5
разделов
:
звуки
,
дисплей
,
другие
настройки
,
горячие

и

.
Все
настройки
стандартны
.
В
звуках
можно
выбирать
мелодии

и
входящих

,

о

.
В

,
можно
лишь

ее

,
а
также
фоновую

.
Впечатления
По
качеству

Philips
9
@
9f
показала

,
за

не

.
Собеседника

,
в

с

9
@
9g
динамик

более

,

проблем

.
64-голосная

,

.
И

,
