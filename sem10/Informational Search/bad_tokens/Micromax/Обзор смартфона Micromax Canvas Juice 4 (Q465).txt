,

,

,

,

.
Сама

,

от

,

.
Для

Juice
4
попытались

,

,

,
но
сохранили
старую
версию
Android
,
к
тому
же
надо
добавить
,
что
обновлений
в
этом
аппарате
можно
не
ждать
.
Получилось
ассортиментное
предложение
,
на
которое
могут
клюнуть
под
влиянием
продавца
,
но
осознанно
выбирать
его
невозможно
,
так
как
на

есть
множество

другого

за
те
же
деньги
.
Дизайн
,
размеры
,
управляющие
элементы
Модель
выпускается
в
двух
цветовых
решениях
–
золотом
и
черном
,
причем
отличается
в
этих
случаях
оформление
задней
панели
и
датчика
отпечатка
пальца
.
В
России
черный
цвет
в
продаже
мне
не
встречался
,
везде
поставляются
устройства
в
золотой
раскраске
,
такой
же
и
у
меня
в
руках
.
Перед
нами
типичный
представитель
публичного
дизайна
,
этот
облик
растиражирован
множеством
китайских
производителей
в
десятках
моделей
.
В
России
он
встречался
в
моделях
от
Senseit
,
хотя
там
отличались
некоторые
элементы
оформления
,
например
,
ободок
камеры
,
да
и
не
было
датчика
отпечатка
на
задней
панели
.
Для
недорогого
телефона
дизайн
отличен
,
равно
как
и
материалы
корпуса
,
задняя
панель
имеет
вставку
из
металла
.
К
сожалению
,
в
телефоне
есть
один
серьезный
недостаток
,
задняя
крышка
сделана
так
,
что
снять
ее
без
особых
усилий
и
танцев
с
бубном
не
получится
.
Каждый
раз
у
меня
это
занимает
порядка
минуты
,
причем
я
боюсь
сломать
ее
окончательно
.
С
точки
зрения
сборки
это
один
из
самых
серьезных
минусов
аппарата
,
установить
SIM-карту
в
него
очень
сложно
.
Мне
давно
не
встречалось
телефонов
,
где
крышка
снимается
с
таким
трудом
,
хотя
Juice
4
среди
них
был
бы
номером
один
.
В
остальном
все
типично
,
на
правой
боковой
стороне
находится
спаренная
клавиша
регулировки
громкости
,
тут
же
кнопка
включения/выключения
.
Верхний
торец
имеет
разъем
3.5
мм
,
на
нижнем
располагается
единственный
микрофон
,
а
также
разъем
microUSB
.
Датчик
отпечатка
пальца
расположился
на
задней
панели
,
что
характерно
для
многих
китайских
производителей
.
Оформление
датчика
не
вызывает
вопросов
,
он
работает
при
прикосновении
,
нажимать
на
сам
датчик
не
нужно
,
в
том
числе
нет
надобности
включать
телефон
(
в
смысле
,
активировать
дисплей
,
в
полностью
выключенном
аппарате
датчик
не
работает
)
.
У
меня
сложилось
ощущение
,
что
датчик
добавили
как
уникальную
характеристику
,
чтобы
сделать
Juice
4
привлекательнее
.
Это
первый
аппарат
компании
с
датчиком
отпечатка
,
и
можно
сказать
,
что
первый
блин
вышел
комом
.
Проблема
заключается
в
том
,
что
в
большей
части
случаев
прикосновение
просто
не
распознается
,
нужно
вспомнить
,
как
именно
вы
прикладывали
палец
при
регистрации
отпечатка
,
и
тогда
что-то
срабатывает
.
Этот
аппарат
–
также
рекордсмен
по
тому
,
сколько
раз
отпечаток
не
срабатывает
и
выдает
ложные
нажатия
.
Проще
совсем
не
пользоваться
датчиком
,
так
как
он
большую
часть
времени
не
работает
.
Вот
так
недоработка
превращает
уникальную
черту
аппарата
в
его
самую
большую
слабость
.
В
аппарате
один
микрофон
,
нет
системы
шумоподавления
.
На
ветру
микрофон
иногда
задувает
,
так
что
собеседник
может
вас
плохо
слышать
.
Размеры
телефона
–
143.5х71.6х8.65
мм
,
вес
–
145
грамм
,
в
руке
он
лежит
хорошо
,
равно
как
и
в
кармане
с
ним
нет
никаких
проблем
.
Слоты
для
microSIM
карт
находятся
под
задней

,
также

и

для

памяти
.
Первое
,

,
«

»

не

,
всегда

(

,

)
.

>
Дисплей
В
этом

5-дюймовый
IPS
экран
с
HD-разрешением
1280х720
точек
.
В

нет

,

,

,

.
На

.
Это

.
Из

–

,

ее

.
Аккумулятор
Встроенная
Li-Pol

3900

до
700

,
в

35

.
Стандартная

(
5В
,

)

.
Неважно
,

,

.
Сделано

,

.
С

2

30

,
до
пары

экрана
.
Установка
второй

резко

время
работы
до
одного
дня
.
Причина
в
неоптимальном

карт
при
работе
.
При

сильно
ухудшается
качество

для

из
карт
.
Поэтому

об

,
только

вторая

.
Время
проигрывания
видео
при
максимальной
яркости
экрана
–
8.5
часов
,
это
очень
низкий

,
учитывая
емкость
батареи
и
разрешение
экрана
.

Программное

В

,

,

,

не

.
Из

,

,

.

,

,

,

.

По

,

.
В

.

.

.
Громкость

.
Иногда

и

.
Тут

,

не

,

,

,

.
Цена
в
9
990

на

5.1
.
Покупателю

,

,

,

и

процессор
,
он

.
На

,

на

,

есть

,

набор

.
Обзор
смартфонов
LG
Magna
(
H502F
)
и

(
H422
)
Можно

на
ZTE
Blade
X3
,

,
что

,

.
В

,

,

.
Обзор

ZTE
Blade
X3
А

до
10
000

.
Перечислять
их

,
выбор

такого

огромен
,

4

в

7-8

.
Модель
,

,

,
и

ее

,

.
К

Все

,

,
