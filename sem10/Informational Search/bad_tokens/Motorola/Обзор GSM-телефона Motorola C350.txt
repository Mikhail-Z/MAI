,

.
Примером

Motorola
.
Модель
Motorola
C350

С33х
,

.
Азиатский

С350

С370
,

.
Размеры

98
x
42
x
18

,

80

,

,

,

.
Стандартный

Приятной

,

С33х

.
Это

не

,

,

.
На
рынке

,

.
Основное

С350

,

4096

STN-технологии
.
По

,

T720
.
За

меньших

и
,

,
разрешения
(
96х65
точек
)

помещается

трех

,
строки

и

иконками
.
На
солнце

,

,

Сравнение

,

,

,

.
Клавиатура

,

,

.
С

удивление

,

.
Клавиша
Меню
,

,

,

.
К

,

ожидания
.
В
русской

,

и

.
Подсветка

,

видны

,

ярком

они
не
всегда
различимы
,
слишком
яркий
свет
начинает
играть
на
гранях
,
да
и
размер
шрифта
маловат
.
На
нижнем
торце
телефона
можно
видеть

mini-USB
,
он

использоваться

к
ПК
и

,
а

.
Кабель

,

,

.
Рядом

разъем
зарядного

,
а

.
Открыв
заднюю

,

моделей
С33х
.
В

литий-ионный
аккумулятор

650
мАч
выполнен

,
пользователь

его

.
По

180

.
Использование

,
что

время

2

15-20

.
Мы
специально
оставляли
телефон

и
использования

,
он
проработал
трое
с

суток
,
но
как
только
начинает
работать
экран
,
энергопотребление
резко
возрастает
,
эффективное
время
работы
резко
падает
.
На
наш

типичного
пользователя
такого
класса
телефонов
среднее
время
работы
составит
как
раз
2
дня
,
рассчитывать
на
большее
нельзя
.
Время

2
часов
.
В
этом
аппарате
держатель
SIM-карты
находится

,
для
того
,
чтобы
вынуть
карту

ползунок
,
удобное
и
простое

.
Меню
В
С350
главное
меню
представлено
кругом
из
иконок
с
подписями
,
это
типичная
организация
для
всех
последних
моделей
.
Быстрой
навигации
по

с

цифровых
последовательностей
не
предусмотрено
.
В
качестве
замены
или
даже
аналога

присутствует
функция
MyMenu
.
Вы
можете
создать
ярлык
для
любого
пункта
меню
и
затем
в
два
нажатия

в
него
.
Функционально
телефон
аналогичен
модели
С33х
,
так
что

некоторые

соответствующего

.
Телефонная
книга
.
В
памяти
телефона
можно
сохранить
до
100
номеров
,
причем
для
одного

записывается
несколько
телефонов
,
но
общее

не
может
превышать
100
.
В
общем

из
памяти
аппарата
,
так
и
с
SIM-карты
.
Все
имена
,
для
которых

несколько
номеров
,
показываются
несколько
раз
,
одно
за
другим
,
справа

иконка

телефона
.
Помимо
номеров
можно
записывать
и
адреса

,
а
также
группировать
отдельные

категориям
.
Для
каждого

и
иконку
,
которая

вызове
.
Для
25
номеров
можно
записать
голосовые
метки
,
распознавание

хорошо
.
Сообщения
.
В
памяти
аппарата
можно
сохранить
до
20
SMS-сообщений
.
Поддерживается
функция
SMS-чат
,
можно
создавать
шаблоны
.
Телефон
поддерживает
стандарт
EMS
,
соответственно
,
в
нем
есть
картинки
и
мелодии
для

их
при
составлении
сообщений
.
В
нашей
версии
телефона
на
открытие
отдельного

уходило
порядка
4

,
эксперименты
с
различными
SIM-картами
показали
,
что

примерно
одинаково
.
Сложилось
ощущение
,
что

даже
не
время
считывания
информации
с

,
а

на

.
Другая

,

,

.
Так
,

,

,

иконки
.
При

,

.

.
В
представительстве

этот

,
заверив
,
что

его
не
будет
.
Медиа-центр
.

Centre
.
Из

,

.
Тут

,

.
Их

35
,

до
32

.
Новая
функция
,

в

,
это

,

их

,
обои

.
Чат
.
Chat
.
В

располагается
SMS-чат
,
создание
отдельного

меню
для
этой
функции

,

.
Голосовой
набор
.
Отдельным

набор
(
20
меток
)
.
Удобно
,

в
одном

всех
имен

,

есть
метка
,

удобно
,
что

,
а
не
в

самой

.
Параметры
.
Традиционно

,
wap-браузера
,
выбор

,
времени

.
В

Можно

При

,

,

.
Офисные

.
Office
tools
.
В

до
500

,

.
Есть

.

В

.

.
Пользователь

.
Ничего

.

Именно

.
Любопытная

,

Для

Минусом

,

о

,

В

(

,

)
.

.

Dial

В

.

.

.

.

Web

.
Продублирован

.

.
В

MotoGP
)
,
за
счет
худшего
экрана
в
нее
не
так
приятно
играть
,

все

.
Astrosmash
-

Впечатления
По
качеству
связи
это
типичный
аппарат
от
Motorola
,
никаких
проблем
с
ним
нет
.
Уровень
полифонического
звонка
по
сравнению
с
той
же
С33х
увеличен
,
теперь
звонок
хорошо
слышно
.
За
счет
другой
компоновки
корпуса
виброзвонок
стал
практически
незаметен
,
рассчитывать
приходится
исключительно
на
то
,
что
вы
услышите
сигнал
вызова
.
Учитывая
,
что
в
аппарате
16-тональная
полифония
,
то
даже
увеличенная
громкость
звонка
не
всегда
спасает
на
шумной
улице
.
Громкость
звука
в
динамике
вполне
достаточна
,
по
этому
параметру
телефон
неплох
.
Главное
меню
производит
впечатление
перегруженности
,
так
,
навигация
по
нему
затруднена
,
всего
в
этом
меню
16
пунктов
.
Некоторые
из
них
,
например
,
Browser
,
Web
Access
,
Web
Sessions
было
бы
логичным
объединить
в
один
пункт
.
То
же
относится
и
к
некоторым
другим
пунктам
,
зачастую
неиспользуемым
большинством
пользователей
никогда
(
когда
вы
пользовались
фиксированным
набором
?
)
.
Наличие
функции
MyMenu
спасает
положение
лишь
частично
.
В
целом
эргономика
меню
оставляет
впечатление
сумбурности
,
виден
отказ
от
старых
принципов
формирования
пунктов
,
но
новые
пока
,
судя
по
всему
,
не
оформились
в
ясную
идею
.
Использование
той
же
платформы
,
что
и
в
С33х
помимо
плюсов
имеет
и
явные
минусы
.
Так
,
цветной
экран
накладывает
большую
нагрузку
на
процессор
телефона
,
передвижение
по
меню
не
всегда
происходит
быстро
,
есть
притормаживания
.
В
целом
телефон
можно
охарактеризовать
,
как
задумчивый
,
многие
операции
выполняются
достаточно
долго
.
Сказывается
недостаток
вычислительных
мощностей
.
По
времени
работы
С350
-
средний
аппарат
,
два
дня
на
сегодняшний
день
,
это
минимальное
время
работы
для
телефонов
такого
класса
.
На
этом
позвольте
остановиться
и
сказать
несколько
слов
о
рыночных
перспективах
аппарата
.
Motorola
С350
открывает
новую
линейку
компании
Motorola
в
2003
году
(
с33х
,
С300
это
все-таки
прошлогодние
модели
)
.
Причем
первые
продажи
приходятся
именно
на
Россию
(
официальный
запуск
с
20
февраля
)
,
в
Европе
телефон
появится
только
двумя
,
тремя
неделями
позже
.
Цена
за
этот
аппарат
в
самом
начале
продаж
составит
примерно
170
долларов
,
что
делает
его
самой
дешевой
моделью
среднего
класса
.
Именно
в
цене
следует
искать
причину
будущих
продаж
этой
модели
,
ведь
уступая
конкурентам
по
тем
или
иным
параметрам
,
Motorola
C350
берет
свое
за
счет
самой
низкой
цены
.
В
будущем
видимо
подобная
ситуация
сохранится
,
и
до
сентября
этого
года
Motorola
C350
будет
оставаться
наиболее
дешевым
GSM-аппаратом
с
цветным
экраном
.
Возможно
,
что
уже
к
концу
лета
цена
за
этот
аппарат
пройдет
отметку
в
120
долларов
.
Благодарим
представительство
компании
Motorola
за
предоставленный
телефон
