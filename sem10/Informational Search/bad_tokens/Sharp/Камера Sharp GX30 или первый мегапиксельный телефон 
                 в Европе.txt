Камера
Sharp
GX30
или
первый
мегапиксельный
телефон
в
Европе
Компания
Sharp
первой
смогла
начать
поставки
на
европейский
рынок
телефона
с
мегапиксельной
камерой
.
Такие
продукты
интересны
тем
,
что
качество
фотографий
уже
можно
с
натяжкой
назвать
приемлемым
,
а
в
сравнении
с
VGA-камерами
ощутимо
лучшим
.
По-прежнему
не
стоит
ожидать
от
телефона
,
что
он
заменит
даже
простенькую
цифровую
мыльницу
с
нормальной
оптикой
.
В
то
же
время
такая
камера
бывает
полезной
и
избавляет
от
необходимости
носить
с
собой
нормальный
фотоаппарат
.
В
этом
вы
убедитесь
сами
.
Обзор
телефона
Sharp
GX30
,
вы
прочитаете
здесь
,
сейчас
же
только
рассказ
о
его
камере
.
Фотографии
,
видео
,
достаточно
объемны
,
поэтому
большинство
приводится
по
дополнительной
ссылке
,
зачастую
в
архиве
.
В
тексте
вы
увидите
буквально
несколько
фотографий
.
Камера
располагается
на
внешней
части
раскладушки
.
Прямо
под
камерой
располагается
световой
индикатор
,
он
же
играет
роль
вспышки
в
телефоне
.
Всего
индикатор
может
светиться
семью
различными
цветами
(
белый
,
красный
,
желтый
,
зеленый
,
синий
,
фиолетовый
,
голубой
)
,
плюс
доступен
режим
Диско
(
смена
различных
цветов
)
.
В
режиме
вспышки
индикатор
срабатывает
кратковременно
,
также
вы
можете
активировать
его
на
постоянное
свечение
.
Естественно
,
что
в
этом
режиме
можно
выбрать
любой
из
доступных
цветов
,
за
исключением
Диско
.
Принудительно
можно
включить
менее
яркое
свечение
индикатора
,
реальной
потребности
в
этом
нет
.
Нажав
клавишу
камеры
,
мы
запускаем
режим
фотографирования
,
режим
видео
доступен
только
через
меню
.
Внутренний
экран
превращается
в
видоискатель
,
картинка
на
нем
выглядит
практически
идеально
(
сказывается
и
размер
экрана
,
и
его
качество
)
.
При
быстром
перемещении
телефона
вы
увидите
дискретность
отображения
картинки
,
она
показывается
рывками
.
Более
того
,
если
попытаться
сфотографировать
во
время
перемещения
телефона
,
то
результат
вам
с
большей
долей
вероятности
не
понравиться
,
фото
будет
немного
смазано
.
Рекомендация
простая
,
фиксировать
телефон
и
затем
уже
фотографировать
,
особенно
это
критично
для
съемки
в
условиях
искусственного
освещения
.
В
настройках
фотосъемки
доступен
пункт
Скорость
затвора
(
нормальная
установлена
по
умолчания
,
есть
медленная
и
быстрая
)
.
Выбирать
медленную
скорость
стоит
при
съемке
в
плохих
условиях
освещенности
,
быструю
на
ярком
солнце
,
либо
фотографировании
подвижных
объектов
,
к
примеру
,
автомобилей
.
Большого
влияния
на
качество
снимков
скорость
затвора
не
оказывает
,
но
наличие
подобной
функции
интересно
.
Ведь
во
многом
стоит
рассматривать
данный
аппарат
как
прообраз
будущих
моделей
.
Фотографии
сохраняются
в
формате
JPEG
,
доступно
два
качества
сжатия
(
Normal
,
Fine
)
.
Из
меню
вы
может
выбрать
место
,
куда
будут
сохраняться
фотографии
,
память
телефона
(
около
6
Мб
)
,
либо
память
карты
памяти
.
Рекомендую
сохранять
все
снимки
на
карту
памяти
.
Конечно
,
если
вы
не
планируете
использовать
их
для
соотнесения
именам
телефонной
книги
или
в
качестве
обоев
экрана
.
По
скорости
сохранения
снимков
в
эти
два
типа
памяти
нет
никакой
разницы
.
В

можно
передать
любой
снимок
из
одного

в
другой
,

ограничений
.

Звук
затвора
отключить
нельзя
,
можно
выбрать

из
трех
предустановленных
.
Из
интересных
возможностей
есть
съемка
по
таймеру
,
а
также
фотографирование
себя
.
В
последнем
режиме
картинка
выводится
на
внешний
экран
,
но
ее
качество
оставляет
желать

,
возврат

по

со
звездочкой
.
Галерея
картинок
.
Просмотреть
можно
одновременно
только
память
телефона
,
либо
карту
.
В
принципе
это
логично
и
не
создает
трудностей
.
В
списке
отображается
название
фотографии
,
дата
,
размер
файла
.
Слева
вы
можете
видеть
миниатюрное
изображение
.
Если
у
вас
хранится
около
70
фотографий
и
все
в
большом
разрешении
,
то

можете
и
не
дождаться
создания
маленьких
картинок
,
они
прорисовываются
крайне
медленно
(
передвижение
по
меню
при
этом
без
тормозов
)
.
Галерея
предлагает
самые
необходимые
операции
,
тут
и
копирование
файлов
из
одного
типа
памяти
в
другой
,
просмотр
данных
о
изображении
.
Но
есть
и
ряд
любопытных
возможностей
,
первое
это
масштабирование
картинки
под
свойства
внешнего
или
внутреннего
экрана
,
для
заставки
включения/выключения
,
иконки
входящего
звонка
,
будильника
.
Мелочь
казалось
бы
,
но
мелочь
приятная
.
В
аппарате
есть
встроенный
редактор
изображений
,
в
нем
отказались
от
функций
,
которые
на
телефоне
смотрелись
бы
не
к
месту
и
оставили
наиболее
интересные
.
Стандартно
можно
вращать
картинку
,
добавлять
рамку
к
ней
(
можно
и
в
момент
фотографирования
,
но
смысла
в
этом
нет
особого
,
лучше
потом
,
чтобы
не
портить

исходное
изображение
)
.
Есть

к

из

(
их
всего
10
)
,
это

,
усы
и

.
Наибольший
восторг
у

вызвала
возможность
применить
к
фотографии
различные

,
а
их
в

(
Sepia
,

,

,

,

,

,

,

)
.
К

,

Painting

вашу

в

,
для
некоторых

такое
преобразование
выглядит
крайне
интересно
.
Видео
.
Телефон

Вы

95

.
Данное

,

.
Как

,

,

,

.
Если

в

,
то

у

,
не
знаю
уж
что
