,

и

То

Еще

.

,

,

Дизайн
,

,

Знаете
,

.
Ни

,

.
Посмотрите
,

,

.
Фотографии

Изначально

Надеюсь

Задняя

И

Справа

Кнопка

На

.
Несмотря

,

,

Конечно

Фронтальная

.
И

В

Кроме

,

.
Как

,

,

.
Последний

,

.
За

у

,

,

.

Выглядит
как
баловство
,
но
,
может
быть
,
кому-то
понравится
.
Также
у
камеры
есть
режим

значения

,
делается
это

,

,

,

.
Видно
,

и

.
Это
же

и
портретного
режима
,
на
мой
взгляд
,
софт
слишком
агрессивно

фон
,
в
итоге
снимок
выглядит
неестественным
.
К
дневной
съемке
у
меня
нет
никаких
претензий
,
снимки
получаются
хорошими
,
а
вот
в
сложном
искусственном

камера
показывает
себя
не
так
хорошо
,
это
касается
не
только
четкости
картинки
,
но
и
баланса
белого
,
обратите
внимание
на
фото
с

,

на

ББ
.
На

,
в

,
если

яркость
вручную
,
то

,

всё

не

.
На

ниже
слева

,
а

—
с

ISO
.
авто
ручной
авто
ручной
авто
ручной
Для

,

.
С

,

.
Слева

,

—

.
Максимальное

FHD
,

,

.
Мультимедийные

Напомню
,

,

(

)
,

звука
,
вы

,

видеороликов
можно
вполне
комфортно
.
Что
касается
звука
в
наушниках
,
то
его
иначе
как
«
обычным
»
не

,
аппарат

же
,

и

его

,
то

с

Huawei

,

.

Программные

И

ZTE

,

MiFavor
,

,

.
Собственная

,

же

,

,

.

,

.

Android
8
Впечатления
К

,

.

.

.

,

,

,

.

.

.
