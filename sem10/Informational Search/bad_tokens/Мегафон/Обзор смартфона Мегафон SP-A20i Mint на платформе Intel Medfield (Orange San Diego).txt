.

,

,

,

(

)
.
Например

,

Смотрим

.

.
Причина

,

,

,

не

.
Они

,

.
И

то
,
что

.
С

Intel

,

у

.
Это

в
том
,
что

на

о
том
,

Atom

.
Например
,

(

)
,

,

.
Во

,

,

,

.
До

,

.
Надо

,

Ананд
Чандрасекер

,

,

.
Напомню
,

,

,
и

.
В

Atom
(

)

.
В

300

,

.
Несмотря

то
,

Intel

,

.
За

,

.
Фактически
,

,
и

Lenovo
не

об
ее

,
этот

был

,
а

.
В

)
.
К

.
Остается

,

.

Вначале

.
Не

,

.
Исторически

,

.
И

,

,

.
Стратегически

,

что

не

того
,

.
Поэтому

.
С

,

(

-
Intel
,

,
