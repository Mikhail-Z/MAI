.
Причем

.

.
Продажи

и

.
Так
,

,

,

.
С

эта

в

,
так

,

и

.
Необходимо
создавать
собственные

,

нельзя

в

.
О
том
,
что
в

будет

,
стало

2014

,

,

,
от
5

и
выше
.
Задачу
тогда

–

фаблет
,

,

,

и

,

,

.
Кризис

смешал

,
так

изменились

,
они
в
России

новых

Тем

,

,
выставлять

моделей
,
отталкиваясь
от
его

.
Забавно

,

Login

,
так

были

с

,

,

.
Вернемся
к

.
Этот

для

,

Android-смартфон

,

.
Во

,

2.5-4

,

у

.
В

и

,

,

.
Это

,

.
Даже

Китае

,

получите

на
полке
,
что
у
Login+
,
забывая
о
том
,
что
ее
надо
привезти
в
Россию
,
что-то
заработать
рознице
и
так
далее
.
Шансов
на
то
,
что
у
Login+
появится

серьезный
конкурент
в
ближайшие
пару
месяцев
,
нет
.
Это
и

о

,

.
Внутри

не

,

,

оно

,

.
Но

–
антикризисный

,

и

,

-

.
Что

,

,

.
Ни

,

-
это

в

.
Но
это
продукт
,
который
имеет
самое

цена/качество
в

.
Давайте
посмотрим
,

.
К

>
>
>
Дизайн
,
размеры
,
управляющие
элементы
В
МегаФон
долго
искали
свой
стиль
,
пытались
нащупать
то
,
что
нравится
потребителям
в
сегменте
Login
.
Первые
устройства
были
обычными
–
пластик
,
глянцевый
пластик
,
одни
из
тысяч
,
не
имеющие
своего
лица
.
Например
,
посмотрите
на

двух

–
Login
2
и
Login+
.
Уверен
,
что
первый
вам
будет
сложно
вспомнить
уже
через
десять
минут
,
второй

запоминается
за
счет

из

.
В
сравнении
с
Login
2
Впервые
такую
спинку
применили
в

Login
3
,
это
решение
пришлось
по

–

его

.
И

не

о
том
,
что

3

истираться
,
обнажается

.
Несколько

на
отзывы
,
что
появившийся

-
это
,

,

,
что
тут
не
обманули
и
это
не
пластик
.
Поэтому
немудрено
,
что
в
последующих

придерживаться
такого
решения
–
так
,
в
Login+
спинка
имеет
вставку
ровно
из
такого
же
металла
.
Но
тут
нет
изгиба
,
поэтому
механического
истирания
удалось
избежать
,
панель

царапаться
о
твердые

,

,

.
Но
и
это
происходит
не
моментально
.
Матовый
пластик

,
он
не

,

в
руках
.
Качество

,

оценить
по
тому
,

задняя

,
нужно
приложить
усилие
.
И
она
совсем
не
тонкая
,
хотя
вес
аппарата
в
153

для
таких
устройств
.
Размеры
телефона
-
155х77х9
мм
,
хорошо
ложится
в
руку
,
не
толстый
.
Это

,

.
С

,

.
В

с
Samsung
Galaxy
Note

На

боковой

,

-
кнопка
включения/выключения
.
Они
немного
утоплены
,
требуется
нащупать
их
,
чтобы
нажать
,
но
проблем
это
не
доставляет
.
На
верхнем
торце
находится
привычный
разъем
3.5
мм
,
тут
же
microUSB-разъем
.
А
вот
микрофон
в
аппарате
единственный
и
он
вынесен
на
нижний
торец
.
Под
задней
крышкой
можно
видеть
аккумулятор
,
а
также
слот
для
microSIM-карты
,
тут
же
разъем
для
карты
памяти
.
У
аппарата
есть
фронтальная
камера
,
а

датчик
освещенности
.
Оговорюсь
,

3G

,

только

,

,
Skype
.
Качество

.
Осталось

.
У

,

.
На

.
Само

,

им
–

,

,

.
При

,
чем
на
миллиметр
,
утоплен

,
можно

,
ему

будет
.
В
сухом

сказать
,

нравится

,
так
и

.
Для

исполнение

.

,

,

.
Разрешение

-
540х960
точек
,

,

,

того
,

.
Уверен
,

,

.
Но

.
Матрица

.
Есть

,

.
Я
использовал
аппарат
на

подсветки
около

,
это
вполне
комфортно
.
К
содержанию
>
>
>
Аккумулятор
Емкость
Li-Ion
аккумулятора
2000
мАч
,
что
по

ничто
и
навевает
грустные
мысли
.
В
МегаФон
не
стали
указывать

в

,
что
же
,
попытаемся

этот

.
Вначале
посмотрел
на
то
,
как
долго
аппарат
сможет
проигрывать
видео
(
100
%

,

,
HD-ролик
в
MX
Player
)
.
Результат
оказался
обескураживающим
–
чуть
менее
3
часов
.
Поэтому
с
большим
опасением
использовал
аппарат
в
обычной
жизни
,
но
все
оказалось
не
так
страшно
.
Для
звонков
и
SMS

вполне
подходит
–

часе

SMS

.
Тут
нет
никаких
проблем
.
При

социальных
сетей
,
веб-страничек

и
его

.
В
среднем

около
2

сохранении
количества
звонков
.
Для
тех
,

на

,
эта

точно
не
подойдет
,
хотя
можно
приобрести
вторую
батарейку
,
и
тогда

веселее
,
но

-
еще
то
удовольствие
.
Для

,
что

в

сетях
и

от
времени
в

дня
,
не
проводят
там
весь
свой
день
,
этот
аппарат
будет
вполне
комфортен
.
Размышляя
о
том
,
относятся
ли
потребители
линейки
Login
к
таким
людям
,
уверенно
могу
сказать
,
что
да
.
И
надо
держать
в
голове
этот
факт
,
прежде
чем
подходить
с
общей
меркой
к
оценке
времени
работы
аккумулятора
.
Для
меня
его
точно
не
хватает
,
но
и
мои
потребности
совсем
другие
.
Интересно
,
что
создатели
телефона
пытались
сделать
так
,
чтобы
время
его
работы
было
максимальным
в
режиме
ожидания
,
аппарат
спокойно
работает
порядка
3-3.5
дней
в
таком
режиме
и
с
минимальной
пользовательской
активностью
.
Полное
время
зарядки
аккумулятора
-
чуть
более
2
часов
.

содержанию
>

Производительность
,
память
,
чипсет
Для
такого
рода
устройств
стандартом
стало
использование
чипсета
MediaTek
MT6582
,
у
него
4
ядра
до
1.3
ГГц
,
хорошая
производительность
интерфейса
.
И
в
этом
аппарате
мы
видим
ровно
то
же
самое
.
Объем
оперативной
памяти
-
1
ГБ
,
встроенной
-
4
ГБ
,
и
есть
поддержка
карт
памяти
до
32
ГБ
.
В
синтетических
тестах
аппарат
показывает
предсказуемый
результат
,
который
для
этого
класса
устройств
выглядит
типичным
.
К
содержанию
>
>
>
Коммуникационные
возможности
Смартфон
работает
в
сотовых
сетях
2G
(
850/900/1800/1900
МГц
)
и
3G
(
900/2100
МГц
)
.
Скорость
HSDPA
–
до
21.6
Мбит/с
,
HSUPA
–
до
5.76
Мбит/с
.
В

Bluetooth
версии
3.0
HS
(
со
стереопрофилем
A2DP
)
.Подключение
внешних
беспроводных
колонок
не
вызывает
никаких
проблем
–все
работает
отлично
.
Присутствует
беспроводная
связь
Wi-Fi
802.11
b/g/n
(
2,4
ГГц
)
.
Аппарат
может
быть
задействован
в
качестве
точки
доступа
(
Wi-Fi
Hotspot
)
и
передачи
файлов
(
Wi-Fi
Direct
)
.
В
настройках
есть
пункт
«
Модем
и
точка
доступа
»
.
Для
передачи
файлов
и

USB

.
При

(

)

(

)
.

Камера
Основная
камера
имеет
честные
5

,

неплохую
картинку
в
ясный
,
солнечный

.
Камера
не
самая
быстрая
,
фокусировка
происходит
иногда
с
небольшой
задержкой
,
особенно
если
день
пасмурный
.
Качество
снимков
типично
для

продуктов
,
на
экране
они
выглядят
неплохо
,
на
ПК
уже
заметно
хуже
.
Но
в
целом
,
оценить
камеру
можно

.
Пример
видео
(
3GP
,
15
МБ
)

Программные
особенности
–
предустановленные
программы
Аппарат
из
коробки
поставляется
с
Android
4.4.2
–
никаких
надстроек
нет
.
Предусмотрено
обновление
по
воздуху
,
поэтому
проблем
с
получением
новых
версий
не
будет
.
Ждать
,
что
этот
продукт
получит
Android
5
,
не
стоит
,
внутри
стоит
чипсет
,
который
не
поддерживает
эту
версию
Android
.
Обзор
операционной
системы
Android
4.2
Начну
с
приятной
новости
,
в
МегаФон
отказались
от
насыщения
софта
смартфона
одинаковыми
программами
,
которые
дублируют
друг
друга
и
путают
пользователей
.
Сотрудничество
с
Яндекс
осталось
как
альтернатива
встроенному
браузеру
Chrome
,
предлагается
аналог
от
Яндекс
,
но
его
можно
и
удалить
.
Других
приложений
от
Яндекс
нет
,
что
создает
ощущение
чистого
Android
.
Впрочем
,
оператор
хорошо
подобрал
собственные
приложения
,
они
довольно
функциональны
.
Так
,
есть
личный
кабинет
МегаФон
,
где
можно
видеть
остатки
услуг
по
пакетам
,
менять
тариф
и
так
далее
.
Одним
словом
,
управлять
своим
телефоном
.
Сервис
МегаДиск
предлагает
автоматически
сохранять
все
файлы
в
облако
от
МегаФон
,
довольно
удобно
и
просто
.
Сервис
Трава.ру
-
это
доступ
к
различной
музыке
от
МегаФон
,
Смотри+
-
просмотр
спортивных
мероприятий
в
телефоне
,
а
функция
Радар
позволяет
искать
других
абонентов
на
карте
–
отличный
сервис
для
родителей
.
Также
в
телефоне
предустановлен
антивирус
Kaspersky
Internet
Security
,
для
кого-то
это
станет
преимуществом
.
Повторю
,
что
аппарат
и
его
меню
не
производят
впечатления
захламлённости
,
как
предыдущие
модели
,
и
это
отлично
.
Видно
,
что
в
МегаФон
сделали
выводы
из
отзывов
как
пользователей
,
так
и
журналистов
.
Встроенное
FM-радио
поддерживает
запись
из
эфира
.
В
остальном
функциональность
привычная
.
С
точки
зрения
мультимедиа
,
это
стандартный
и
всеядный
Android
,
можно
смотреть
любые
видео
,
слушать
любую
музыку
.
Впечатления
По
качеству
связи
никаких
нареканий
к
этому
аппарату
нет
.
Придирчиво
изучал
его
возможности
в
разных
условиях
,
везде
он
справляется
хорошо
.
Из
минусов
отмечу
,
что
комплектная
гарнитура
недорогая
,
она
может
шуметь
на
ветру
.
В
остальном
все
хорошо
–
звонок
громкий
,
виброзвонок
по
силе
выше
среднего
и
хорошо
ощущается
.
Стоимость
этого
аппарата
зависит
от
региона
,
продажи
начинаются
с
1
февраля
по
всей
стране
.
Цена
указывается
как
4
990
рублей
при
условии
покупки
интернет-пакета
от
одного
до
четырех
месяцев
.
Например
,
в
Москве
при
подключении
тарифа
«
Интернет
S
»
на
2
месяца
стоимость
фаблета
составит
5
690
рублей
.
В
новых
реалиях
рынка
эта
цена
выглядит
не
просто
адекватной
,
а
очень
низкой
,
что
сделает
продажи
устройства
крайне
высокими
.
Например
,
вышедший
летом
Alcatel
POP
C9
стоит
порядка
7-7.5
тысяч
рублей
(
выходил
по
7
990
рублей
)
,
это
остатки
,
которые
не
представлены
широко
.
Функционально
похож
на
Login+
,
но
имеет
две
SIM-карты
,
чуть
больше
аккумулятор
.
В
сегменте
фаблетов
можно
найти
и
6-дюймовые
устройства
(
480х854
точки
в
этом
случае
)
,
например
,
Fly
Era
Style
2
.
Модель
стоит
пока
еще
около
7
000
рублей
,
но
цена
будет
пересмотрена
,
или
она
совсем
исчезнет
с
рынка
.
Перечислять
модели
от
разных
китайских
компаний
можно
долго
,
их
в
общей
сложности
больше
десятка
.
Со
сходными
характеристиками
они
стоят
от
7
000
рублей
(
и
это
сегодня
,
дальше
цены
,
видимо
,
будут
меняться
)
.
Чуть
лучшие
характеристики
,
например
,
HD-экран
уже
начинается
с
8
000
рублей
.
Но
вопрос
заключается
в
том
,
что
все
эти
модели
не
имеют
никакой
рекламной
поддержки
,
о
них
не
знает
массовый
потребитель
,
и
они
не
представлены
по
всей
стране
,
как
это
сделано
для
Login+
.
И
отсюда
следует
еще
один
вывод
,
именно
Login+
в
продажах
станет
номером
один
в
сегменте
фаблетов
с
5.5-дюймовым
экраном
.
Это
будут
вовсе
не
модели
от
Samsung
,
Apple
или
кого-то
еще
,
а
именно
этот
аппарат
.
И
на
пару
месяцев
он
точно
вырвется
в
лидеры
продаж
,
а
потом
просто
закончится
тот
объем
,
что
запланирован
в
МегаФон
.
Возможно
,
что
модель
снова
появится
,
а
может
,
и
нет
.
В
любом
случае
,
у
вас
только
пара
месяцев
,
чтобы
гарантированно
купить
этот
аппарат
,
дальше
это
может
быть
затруднительно
.
Для
своей
цены
аппарат
получился
интересным
и
хорошим
,
самый
большой
недостаток
-
исключительно
во
времени
работы
.
Но
он
нивелируется
тем
,
что
для
целевой
аудитории
сценарии
использования
несколько
отличаются
от
постоянных
игр
и
нахождения
в
сети
.
Возможно
,
поэтому
телефон
не
очень
подойдет
детям
,
что
любят
играть
,
а
вот
для
более
взрослой
аудитории
это
интересное
предложение
.
И
могу
сказать
точно
,
это
бестселлер
,
который
повторит
успех
предыдущих
моделей
линейки
Login
.
