.

надо

.

Я

в
стратегии
,
выбранной
Sony
,
ведь
в
линейку
Z5
пока

,
может

,

планшет
.
Фаблет
,
смартфон
и

компактных

-

,

4K
,
другим

хорошего

810
,

)

Sony

Compact
,

,

.

:

,

Z5
,

«
премиум
»
-

,

«

»

.

,

.

.

-

.

.

,

.

,
