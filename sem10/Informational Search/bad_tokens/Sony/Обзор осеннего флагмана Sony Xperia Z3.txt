Позиционирование
В

.

.

,

.
Но

,

.
В

С

,

,

.
В

,

,

.
Отдельно

,

с

S5

)

(

)

.
В

,

,

,

.

,

,

,

.

(

)
.
Теперь

вполне

.
Это

.
В

.
В

,

–

,

,

.

Аккумулятор
В
телефоне
стоит
литий-полимерный
аккумулятор
емкостью
3100
мАч
(
в
Z2
3200
мАч
)
.
Для
флагмана
такая
емкость
батареи
отлична
,
она
вполне
сравнима
с
Note
3/4
,
впрочем
,
и
по
времени
работы
эти
аппараты
схожи
.
В
описании
этого
аппарата
Sony
говорит
о
том
,
что
он

два

,
при

до
5

в

активной

.
Методика
,
к
сожалению
,
не
приводится
,
но

с
сайта
компании
.
Мое
первое
впечатление
от
аккумулятора
было
фантастическим
–
он
работал
очень
долго
в
режиме
ожидания
,
я
вставил
карту
и
почти
не
гонял
телефон
,
а
он
почти
не
разряжался
.
Когда
стал
активно
использовать
аппарат
,
он
показал

лучше
,
чем
Z2
,

не

.
Причина
заключается
в
том
,
что
в

,
и
в

ожидания
аппарат
почти
ничего
не
потребляет
.
То
есть
,
фактически
,
работает
одно
ядро

и
на
очень
низкой
частоте
.
Поколдовали
на
славу
,
в
режиме
ожидания
аппарат
работает
просто
на

.
А
вот

обычных
режимах
чуда
не
произошло
,
тут
сказывается
не
столько
процессор
,
сколько
яркость
подсветки
экрана
и
ваши
настройки
.
Учитывая
,
что
яркость
в
экране
возросла
,
то
и
потребление
стало
чуть
больше
,
что
нивелирует
все
настройки

.
Так
,

–

)
.
Заявленное

–

16

,

–

920

(

–
740
часов
)
.
Время

–

130

.
На

телефон

,

Note

,

.
Вы

.
Если

,

.
Время

)

.
До

2

.
Из

,

,

,

.
Но

,

,

,

.
В

,

со
своими
одноклассниками
(
Galaxy
S5
,

)
,
выигрывает
у

и

с

3
.
На

,
этот

можно

,
кто

в

,

.
Из

,

,

,

,

данных
в

,

и

.
Это

,

.
В

,

,

.

,

,

.
Есть

.

Если

Это

.
Посмотрите

,

В

.

.
В

,

.

.

Мультимедийные

,

Если

,

.
Чтобы

,

,

.
Технически

DSEE

,

и

«

»

,

.
Работает

,

,

,

,

.
К

,

,

.

В

,

.
