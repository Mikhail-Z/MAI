Интересно

В

,

.
Но

.

>
>
>
Дизайн
,

,

Разработчики

Xperia
pro
,

,

.
Несмотря

,

Немного

(

)
.
По

хромированная

,

.
Устройство

,

.
Габариты

:
120х57х13.5

146

.
Например

165

180

.
В

.
При

,

.
В

.
На

.
Рядом

–

.
Громкость

,

.
Собеседника

,

.
Под

–

:
«

»
,
«

»
и
«

»
.
Нажатие

,

.
Слева
и

.
Они

.
На

.
На

,

,

.
Чуть

–

(

,

,

,

.
На

–
microHDMI-разъем
,

–
индикатор

и

.
Ниже
-

,

-

фотокамеры
.
Глазок

,

,
второй

(

)
и

.
Как

,

Клавиатура
Пожалуй
,
одна
из

и

на

.
На

Для

Он

,

,

и

.
Кнопки
(

42
)

(

)

,

,

с

.
Букву
«

«

»
,
«

»

«

,

»
,
«

»

«

.
Для

Переключать

.

»

.
Подсветка

,

,

.
При

,

.
