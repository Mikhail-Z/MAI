.

,

.

,

.

.

,

.

,

,

,

,

.
Во-первых
,

SE
,

,

.

,

,

.

,

,

.

>
Материалы
корпуса
,
качество
сборки
Аппарат

из
глянцевого
пластика
(

идет
о

белого
,

)
.
На

.
За

SE

он

,
я
не

,

или

,
царапины
.
Хотя

я
носил
как
в

,
так
и
просто
в
рюкзаке
,
иногда

его

с

телефонами
.
Покрытие

,

и

,
то
совсем
незаметны
.
Аппарат

выполнен
из
матового
пластика
с
soft-touch
покрытием
.
К
качеству
сборки
нареканий
почти
нет
,
кроме
одного
:
в
моем
образце
был
небольшой
зазор
между
корпусом
и
левой
нижней
частью
крышки
аккумулятора
.
Касательно
материалов
корпуса
,
конечно
,
можно
поворчать
,
что
SE
почти
не
используют
в
своих
моделях
металл
или
просто
более
интересные
,
чем
глянцевый
пластик
,
материалы
,
как
HTC
,
например
.
Но
если
сравнить
SE
Ray
по

с

цене
предложений
от
Samsung
,
LG
и
HTC
,
то
можно
принять
использование
простого
пластика
как
закономерность
,
у
смартфонов
SE
есть
свой
интересный

,
и

.

.

Элементы
управления
В
SE
постепенно
начинают
отказываться
от
полностью
аппаратного
блока
клавиш
под
экраном
,
на
мой
взгляд
,
это

,

.
В
SE
Ray
две
клавиши
под
экраном
сенсорные
(
«
Назад
»
и
«
Меню
»
)
и

-
«
Домик
»
.
Плюс
в
том
,
что

на
«
Домик
»
можно

,

-
не

раз
тянуться
к

питания
,
которая

на

,
что

смартфонов
HTC
,
например
.
Вокруг
клавиши
«
Домик
»

(

ее

)
,
она

ПК

,

.
Если

и

с

-
кольцо

белым
.
Если
есть

-

.
Расположение
светового
индикатора
(

)
,

,

,

,

на
SE
Arc
.
Ощущение
,

SE
использовали

и

,

о
его

назначении
.
Нахождение
индикатора
в

удобно
.
На
правом
ребре

располагается
кнопка
регулировки
громкости
,
для
меня
эта
клавиша
оказалась
неудобной
:
маленькая
,
тонкая
,
короткая
и
с
едва
ощутимым
ходом
нажатия
.
Пользоваться
ей
в
процессе
разговора
сложно
,
особенно
если
учесть
,
что
у
правши
на
эту
маленькую
кнопку
ложится
как

палец
.
На
левом
ребре

находится
microUSB
разъем
для
зарядки
и
подключения
к
ПК
.
На
верхнем
торце
-
уже
упомянутая
клавиша
питания
и
3.5
мм
разъем
для
наушников
или
гарнитуры
.
Снизу
-
отверстие
микрофона
и
ушко
для
крепления
шнурка
или
ремешка
на
руку
,
в
новых
смартфонах
оно
встречается
все
реже
,
хотя
своей

для

с
годами
вовсе
не
потеряло
.
На
лицевой
стороне

,

экраном
,
расположены
разговорный
динамик
,
объектив
фронтальной
VGA
камеры
,
а

сенсор

и
датчик

.
Последний
в
аппарате
не
задействован
,

,

4.0.A.2.368
(

,

)
нельзя

,

вручную
.
Крепление

,

образом
,
что
ни
«
симку
»
,

памяти
нельзя

,
не
сняв
перед

.
Лично

отсутствие

–

,
я

.
Для

,

.
К

>
>
>
Экран
В
Sony
Ericsson

емкостного

.
Диагональ

3,3
”
,

(

)
,

16

.
Физические

экрана
всего
74x41
мм
,
экран

в

и
очень
узкий
.
За
счет

и

SE
Ray

ppi
(

точек
на
дюйм
)
-
297
.
Для
сравнения
-
ppi
в
HTC
Desire
S
составляет
252
,
HTC
Sensation
-
256
,
Apple
iPhone
4
-
330
.
Экран
имеет
средний
запас
яркости
,
однако

солнце
все
равно

себя

,
картинка
остается
читаемой
.
По

хорош
,
цвета
на
нем

к
натуральным
,
белый
выглядит
как
белый
,
а
черный
-
очень
черный
(
простите
за
тавтологию
)
,
а
не
темно-серый
или
какой-то
еще
.
В
этом
плюс
экрана
перед
некоторыми
конкурентами
.
Минус
–
как
у
SE
Arc
,
здесь
не
максимальные
углы
обзора
,
при
отклонении
экрана
картинка
на
нем
начинает
слегка
выцветать
.
Как
и
во

,
здесь
используется
фирменная
технология
Mobile
BRAVIA
Engine
.
Как
и
в
Arc
,
я
не
заметил
отличий
в
картинке
на
экране
при
включенной
опции
и
выключенной
,
ни
при
простой
работе
со
смартфоном
,
ни

фотографий
или
видео
.
Экран
Sony
Ericsson
Ray
покрыт
защитной
пленкой
,
как
и
в
Arc
.
На
мой
взгляд
,
смысла
носить
аппарат
с
пленкой
нет
никакого
-
экран
хорошо
защищает
и
закаленное
стекло
,
которое

используется
,
а
пленка
просто
мешает
:
на
ней
остается

,
мелких

и

в

смартфона
.
К
содержанию
>
>
>
Удобство
работы
со
смартфоном
Дисплей
SE
Ray
маленький
,
и
это
накладывает
свои
ограничения
на
удобство
работы
с
устройством
.
Если
же

иначе
-
то
как
смартфон

SE
Ray
выглядит
слабо
именно
из-за
небольшого
физического
размера
экрана
.
Выражается
это
в
следующих
вещах
.
Во-первых
,
набирать
текст
в
портретной

с

экранной
клавиатуры
очень

.
Постоянные

,

по
тонким
клавишам
.
В

ориентации

,

лично
мне
и
в
таком

было
сложно
,
я
мог
набрать
буквально
одно
сообщение
и
потом
«
остывал
»
и
успокаивал
свои
нервные
окончания
,
потому
что
после
набора
эсэмэски
хотелось
бросить
Ray
об
стену
.
Ситуация
улучшается
,
если
поставить
альтернативную
клавиатуру
,
например
HTC
IME
MOD
.
Тогда
с
набором
текста
становится
полегче
,
особенно
в
ландшафтной
ориентации
.
Но
все
равно
по
удобству
набора
смартфон
не
дотягивает
даже
до
бюджетников
с
HVGA
экранами
,
но
с
более
комфортными
размерами
экранов
и
привычным
соотношением
сторон
.

полностью
)
повторяет
тот
интерфейс
,
что
реализован
в
Arc
.
Оболочка
далека
от
идеала
или
просто
удобства
,
так
,
например
,
окно
набора
номера
не
имеет
функции
SmartDial
,
а
при
открытии
профиля
контакта
вы
видите
пережатое
и
растянутое
по
ширине
изображение
человека
на
заднем
плане
.
Также
в
аппарате
есть
виджет
TimeScape
с
лентой
сообщений
от
друзей
и
знакомых
в
twitter
и
facebook
и
с
текстовыми
сообщениями
.
Этот
виджет
как
был
странным
,
так
и
остался
,
сообщения
на
небольших
графических
панельках
режутся
,
а
фон
панелек
–
пережатые
фотографии
абонентов
очень
низкого
качества
и
всегда
в
неожиданных
ракурсах
.
Потому
что
вся
фотография
на
панельку
не
лезет
,
и
система
сама
берет
ее
центральный
кусок
.
От
MediaScape
в
компании
отказались
,
его
не
было
и
в
SE
Arc
.
Для
проигрывания
музыки
здесь
используется
фирменный
плеер
,
в
меру
простой
и
удобный
,
для
просмотра
картинок
-
стандартная
галерея
Android
.
0
До
того
,
чтобы
смартфоны
от
SE
стали
хороши
по
программной
части
,
им
осталось
всего
немного
-
полностью
отказаться
от
своих
странных
надстроек
для
системы
и
оставить
классический
Android
,
или
же
взять
пример
с
Huawei
и
устанавливать
в
свои
смартфоны
стороннюю
оболочку
.
И
хотя
SPB
Shell
3D
уже
занята
теми
же
Huawei
,
есть
много
других
качественных
интерфейсов
,
которые
на
голову
удобнее
и
функциональнее
стандартного
Android
,
и
уж
тем
более
надстроек
от
Sony
Ericsson
.
ADWLauncher
EX
,
GO
Launcher
EX
,
LauncherPro
Plus
.
Битва
титанов

Конкуренты
В
конкуренты
для
SE
Ray
,
с
одной
стороны
,
можно
записать
почти
все
смартфоны
среднего
класса
на
базе
Android
.
C
другой
,
если
учитывать
габариты
модели
,
используемую
платформу
,
как
у
флагмана
Arc
,
то
я
бы
остановился
на
сравнении
Ray
с
HTC
Desire
S.
Второй
немного
дороже
и
при
том
уступает
смартфону
от
SE
по
части
камеры
и
дизайна
.
Но
в
HTC
Desire
S
интереснее
материалы
корпуса
,
есть
удобный
интерфейс
Sense
,
и
,
что
важнее
всего
,
им
удобнее
пользоваться
именно
как
смартфоном
:
набирать
на
нем
текст
,
смотреть
информацию
в
интернете
,
аппарат
из-за
оболочки
и
больших
размеров
,
в
том
числе
экрана
,
просто
удобнее
в
работе
.
К
содержанию
>

>
Заключение
К
качеству
связи
примерно
за
месяц
эксплуатации
у
меня
не
возникло
никаких
претензий
.
Уровень
громкости
вызывного
динамика
высокий
,
к
тому
же
,
есть
настройка
,
которая
позволяет
поднять
уровень
еще
выше
(
xLoud
)
.
Разговорный
динамик
средний
по
уровню
громкости
.
Виброзвонок
слабый
,
я
чаще
всего
не
замечал
его
во
время
входящих
звонков
.
В
России
смартфон
появился
в
продаже
в
августе
,
официальная
стоимость
модели
-
15
990
рублей
.
Впрочем
,
найти
«
серый
»
аппарат
можно
и
за
13
000
рублей
,
и
именно
по
этой
цене
SE
Ray
выглядит
наиболее
привлекательно
(
а
как
же
иначе
?
)
.
Как
человек
,
который
чаще
пользуется
смартфонами
с
большими
экранами
,
я
не
могу
смотреть
на
Ray
максимально
объективно
,
потому
что
небольшие
физические
размеры
дисплея
для
меня
–
недостаток
,
а
не
особенность
,
продиктованная
компактными
габаритами
.
С
другой
стороны
,
наверняка
есть
те
,
кто
даже
на
таком
небольшом
экране
сможет
без
труда
набирать
длинные
текстовые
сообщения
или
часами
сидеть
в
интернете
.
В
конце
концов
,
это
дело
привычки
.
И
все-таки
,
я
не
стану
рекомендовать
SE
Ray
как
Android-смартфон
,
слишком
велика
цена
потери
удобства
,
которую
заплатит
его
обладатель
за
классный
дизайн
и
компактность
устройства
.
Зато
если
вы
ищете
в
первую
очередь
красивый
и
небольшой
аппарат
с
хорошим
функционалом
про
запас
и
скоростью
работы
-
то
SE
Ray
в
средней
ценовой
категории
Android-смартфонов
оказывается
почти
что
вне
конкуренции
.
Для
своей
стоимости
он
оснащен
шикарным
по
качеству
картинки
экраном
и
отличной
камерой
,
держится
от
одного
заряда
почти
сутки
и
имеет
все
функции
Android-смартфона
.
