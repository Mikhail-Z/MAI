Позиционирование
Компания
Fly
представила
очередной
недорогой
смартфон
на
«
андроиде
»
с
двумя

,
довольно
большим
3.5-дюймовым

емкостного
типа
,

,
наличием
беспроводных
технологий

,
Wi-Fi
и
GPS
.
Назвали
новинку
«
Whizz
»
,
что
в
переводе
с
английского
-
«
свист
»
.
Вроде
бы
,
название
как

,
но
если
вспомнить
,
кого
называют
«
свистунами
»
на
Руси
(
тех
,
кто
,
мягко
говоря
,
привирает
)
,
то
как-то
подозрительно
:
)
Однако
в
британском
английском
слово
означает
что-то
быстрое
и
ловкое
.
Вот
такие
получаются
иногда
лингвистические
казусы
.
Стоимость
Fly
IQ240

4

.

>

,

,

945

.
Размеры
IQ240
–

,

–
126

.
Сделан

.
Например
,

и
окантовка

(

)
,
а

,
черного

(

)
.
Материалы
корпуса
достаточно
крепкие
.
За
время

не
появилось
ни
царапин
,
ни
потертостей
.
Впрочем
,
на
экран

бы
купить
защитную
пленку
.
Пользоваться
устройством
действительно
очень
удобно
благодаря
небольшим
габаритам
,
плавным
изгибам
корпуса
и
добротным
материалам
,
особенно
после
громадных
сенсорных
«
лопат
»
.
В
руке
Fly
Whizz
лежит
отлично
.
На
лицевой
панели
находятся
два
датчика
:
освещения
и
приближения
.
Оба
работают
хорошо
,
и
за
период
тестирования
не
было
ни
одного
случая
,
когда
бы
датчик
приближения
непроизвольно
разблокировал
экран
.
Более
того
,
могу
сказать
,
что
он
реагировал
молниеносно
:
только
отводишь
трубку
от
уха
–
сразу
активируется
дисплей
,
и
можно
нажимать
,
например
,
на
«
Отбой
»
.
Частенько
в
других
аппаратах
приходится
пару
секунд
ждать
,
а
потом
несколько
раз
кликать
на
завершение
разговора
.
Динамик
прикрыт
металлической
сеточкой
.
Его
громкость
очень
высокая
,
в
большинстве
ситуаций
приходится

потише
.
Собеседника
слышно
четко
и
разборчиво
.
Не
сочтите
за
рекламу
,
но
на
момент
тестирования

IQ240
я

из
лучших
речевых

в
Explay
SL240
.
Теперь

на

место
Whizz
.
Рядом
с

–
фронтальная

.
Под

три

«
Назад
»
,
«
Домой
»
и
«
Меню
»
.
Их

твердое
,
ход
небольшой
.
Подсветка
тусклая
,
белого
цвета
.
Краска
нанесена
в
канавки
.
Под

«
Назад
»
-
отверстие
микрофона
.
На
нижней

находятся
дужка

и
стандартный
3.5
мм
аудиовыход
на

или
гарнитуру
.
Слева
нет
никаких
элементов
,
а
справа
–
широкая
тонкая
клавиша-качелька
регулировки
громкости
и
microUSB-разъем
.
На
верхнем
торце
–
утопленная
в
корпус
кнопка
включения/выключения
телефона
,
также
выполняющая
роль
блокировки/разблокировки
дисплея
.
Основная
камера
,
светодиодная

и
прорезь

на
тыльной

аппарата
.
Чтобы
раскрыть

,
ее

.
Слоты

(

)
и

(

)

,
рядом

(
«

»

)
.

Дисплей
Несмотря
на
не
очень
большую
стоимость

,

установил

3.5

,

,

.

.
Физический

–
49х73

,

–

Экран

,

.
Тип

–

,

5

.
Чувствительность

,

,

Если

,

.
Углы

,

.
Максимального

.
В

В

Яркость

.
Дисплей

Если

.
Было

,

.

От

Коммуникационные
возможности
Смартфон
работает
в

2G
(
900/1800

,

)

.
Вместо

,

«

»

.
В

,

Присутствует

Интересно

Обычно

Для

За

Память
и

По

меркам

,

.
Впрочем
,

(

)
,

В

Для

,

120

.
К

,

,
придется

Максимальный

быть
32
ГБ
.
К

>
>
>
Камера
В
аппарате
используется
3-мегаписельный

без
автофокуса
.
Присутствует
односекционная

.
Светит

20-30

(

,

)
.
Качество

:

,

Лучшие

Сфотографированный

Видео

Запускается

.
Интерфейс

:

–

,

,

;

–

,

–

.
