.
Думаю

,

»

,

,

и

далее
,

хотя
бы
до

.
К

,
пока

,

.
Поэтому

.
Однако

–

размером
с
кулак
и
весом
как

же

.

вспоминается
сюжет
«
Сделка
»

«

»

:

!
Куда
сорвался
?

для

(

)

.
В

,
такими
темпами
будет

не
встраивать
аккумуляторы
в
смартфоны
,
а
наоборот
,
выпускать

типа
портативной

с

.
К
содержанию
>
>
>
Дизайн
,
размеры
,
управляющие
элементы
Внешний
вид
устройства
,
по
крайней
мере
,
с
лицевой
стороны
,
мне
показался
довольно
привлекательным
:
прямоугольная
форма
,
заостренные
края
,
черного
цвета
панель
,
закрытая
стеклом
,
и
тонкая
матовая
пластиковая
вставка
снизу
,
слегка
выгнутые
верхняя
и
нижняя
грани
.
Окантовка
выполнена
из
глянцевого
пластика
черного
цвета
,
остальная
часть
корпуса
,
включая
заднюю
панель
,
–
матовый
пластик
.
Собственно
,
на
этом
весь
шарм
(
если
он
был
)
и
заканчивается
:
Boost
II
с
обратной
стороны
–
типичный
«
нонейм
»
-девайс
.
Поскольку
производителем
предусмотрена
замена
батарей
разной
емкости
,
а
соответственно
,
и
размеров
,
то
в
комплекте
вы
найдете
две
тыльные
крышки
.
Одна
тонкая
,
другая
довольно
крупная
.
Габариты
устройства
с
аккумулятором
на
3
000
мАч
–
140х68х9,8
мм
,
с
батарейкой
на
6
000
мАч
толщина
увеличивается
до
14,8
мм
.
В
первом
случае
аппарат
весит
151
грамм
,
во
втором
–
203
грамма
.
В
принципе
,
не
такой
и
громоздкий
Highscreen
получается
даже
с
усиленным
аккумулятором
.
Другой
вопрос
,
что
таким
девайсом
не
очень
удобно
каждодневно
пользоваться
:
брюки
и
джинсы
оттопыриваются
,
в
карман
рубашки
даже
смешно
умещать
,
остается
только
носить
в
карманах
пиджака
или
в
борсетке
.
Со
сборкой
тоже
не
все
гладко
.
После
установки
тонкой
крышки
она
слегка
прогибается
до
аккумулятора
,
немного
люфтит
.
С
большой
панелькой
появляются
и
люфт
,
и
скрип
,
и
хруст
(
возможно
,
для
кого-то
это
незначительный
фактор
–
главное
,
что
автономность
хорошая
)
.
В
остальном
все
нормально
.
Корпус
немаркий
,
практически
не
подвержен
внешним
воздействиям
,
экран
защищен
стеклом
,
за
несколько
недель
на
нем
не
появилось
ни
одной
царапины
.
Сверху
на
лицевой
панели
элегантно
вписан
речевой
динамик
,
прикрытый
блестящей
металлической
сеточкой
темного
цвета
.
Слева
от
него
–
датчики
освещения
и
приближения
,
а
справа
–
фронтальная
камера
.
Громкость
динамика
высокая
,
запас
небольшой
,
но
есть
.
Собеседника
слышно
разборчиво
и
четко
,
звук
немного
глухой
.
Под
экраном
расположены
сенсорные
кнопки
.
Центральная
имеет
круглую
форму
(
нечто
подобное
вы
могли
видеть
в
Meizu
MX2/3
)
и
отвечает
за
выход
из
всех
приложений
на
главный
экран
.
Подсвечивается
белым
цветом
и
мигает
при
пропуске
различных
событий
,
а
также
при
заряде
батареи
.
Слева
от
круглой
кнопки
–
светящаяся
точка
вызова
меню
,
справа
–
«
Назад
»
.
На
левом
торце
находится
тонкая
и
длинная
клавиша-качелька
регулировки
громкости
.
Она
сделана
из
глянцевого
пластика
черного
цвета
.
Справа
вверху
–
такого
же
типа
кнопка
питания
.
Достаточно
удобная
в
обращении
,
но
хорошо
бы
разместить
ее
чуть
ниже
.
Под
кнопкой
блокировки/разблокировки
–
активация
фотокамеры
при
длительном
удержании
.
К
сожалению
,
она
запускает
камеру
,
только
если
экран
активен
.
В
настройках
можно
назначить
на
нее
запуск
музыки
или
создание
снимка
экрана
.
На
нижнем
торце
–
основной
микрофон
,
на
верхнем
–
micro-USB
и
стандартный
3.5
мм
аудиовыход
.
Камера
,
вспышка
,
второй
микрофон
и
динамик
расположены
на
тыльной
части
Boost
II
.
Если
на
корпус
надета
крышка
для
аккумулятора
3
000
мАч
,
то
камера
совпадает
по
высоте
с
крышкой
,
а
если
используется
усиленная
батарея
,
то
камера
находится
в
углублении
.
Чтобы
снять
заднюю
панельку
,
ее
необходимо
поддеть
за
маленькую
выемку
на
нижнем
левом
ребре
.
Сверху
слева
находится
слот
для
карты
памяти
microSD
,
справа
–
два
разъема
под
microSIM
(
не
путать
с
обычной
большой
miniSIM
)
.
Сравнительные
размеры
:
Highscreen
и
Meizu
MX3
Highscreen
и
Nokia
1020
Highscreen
и
Samsung

.

,

.
