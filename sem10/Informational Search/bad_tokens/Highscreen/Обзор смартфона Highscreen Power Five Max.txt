)
.

–

,

и

.
В

Highscreen

(
так

)
можно

21
000

.
К
содержанию
>
>
>
Дизайн
,
размеры
,
управляющие
элементы
Уверен
,
что
Highscreen
до
настоящего
времени
ничего
лучше
с
точки
зрения
материалов
и
дизайна
не
делали
.
Аппарат
Power
Five
Max
действительно
и
выглядит
отлично
,
и
собран
качественно
.
Конечно
,
немного
напоминает
OnePlus
3
,
но
сейчас
практически
все
устройства
похожи
.
Внешний
вид
PFM
лаконичный
,
без
лишних
рюшечек
и
элементов
украшения
–
строгий
мужской
смартфон
бронзового
цвета
.
Лицевая
панель
имеет
стекло
2.5D
,
рамки
визуально
очень
тонкие
,
по
периметру
проходят
два
шлифованных
канта
(
кстати
,
чуть
ли
не
до
последнего
момента
перед
обзором
думал
,
что
они
синие
,
а
оказалось
,
на
них
была
наклеена
защитная
пленка
)
.
Задняя
крышка
немного
покатая
,
за
счет
чего
телефон
удобно
держать
в
руках
.
Сверху
и
снизу
присутствуют
две
тонкие
пластиковые
горизонтальные
перегородки
(
видимо
,
там
расположены
антенны
)
.
Заметьте
–
не
пластиковые
вставки
,
а
перегородки
–
всё
как
у
взрослых
:
)
Еще
раз
повторю
,
что
корпус
сделан
из
6000-й
серии
алюминия
(
аналогичный
материал
используется
в
Apple
iPhone
6
и
6S
)
.
За
время
использования
устройства
на
экране
Highscreen
не
осталось
ни
одной
царапины
,
на
тыльной
крышке
,
а
также
на
боковых
рамках
–
отличный
результат
,
теоретически
можно
поверить
в
силу
прочного
корпуса
из
алюминия
6000-й
серии
.
Все
элементы
находятся
симметрично
относительно
друг
друга
,
что
неплохо
.
Олеофобное
покрытие
присутствует
,
отличное
качество
:
отпечатки
практически
не
видны
,
палец
скользит
без
проблем
.
Сверху
по
центру
–
разговорный
динамик
,
забранный
металлической
сеткой
.
Спикер
громкий
,
запас
по
громкости
есть
,
тембр
приятный
.
Слева
от
него
–
камера
и
индикатор
пропущенных
событий
,
справа
–
датчики
освещения
и
приближения
.
Под
экраном
расположена
механическая
кнопка
со
встроенным
в
нее
сканером
отпечатков
пальцев
.
Кнопка
чуть
утоплена
в
корпус
,
нажатие
твердое
,
ход
небольшой
.
Справа
и
слева
–
«
Назад
»
,
«
Меню
»
без
подсветки
.
Снизу
–
отверстия
под
динамик
громкой
связи
и
под
микрофон
,
microUSB
.
Странно
,
что
именно
этот
тип
USB
использовали
в
данном
аппарате
,
ведь
гаджет
новый
,
поддерживает
быструю
зарядку
–
можно
было
бы
и
Type-C
использовать
.
Слева
на
бортике
имеется
металлический
слот
для
карты
памяти
microSD
.
Справа
–
слоты
под
две
симкарты
формата
microSIM
.
Обратите
внимание
,
что
есть
две
симкарты
и
отдельно
карта
памяти
,
с
такой
схемой
практически
нет
хороших
гаджетов
.
Однако
Samsung
A5
и
A7
будут
работать
именно
так
,
впрочем
,
их
стоимость
выше
Power
Five
Max
.
Кнопка
питания
и
клавиша
регулирования

также
справа
.
Они
металлические
,
удобные
в
обращении
.
Сверху
вы
найдете
3.5
мм
аудиовыход
на
наушники
и
ИК-передатчик
.
Насколько
нужен
последний
,
вероятно
,
решать
не
мне
,
но
то
,
что
он
есть
,
–
дополнительный
плюс
к

оценке
смартфона
в
заключении
.
С
обратной

находятся
:
модуль

,
выступающий

над

,
второй

и

.
В

,

,

и

мне

:
возникает
ощущение
,
что

,

о

Сравнительные

Дисплей
Устройство

,

.
Физический

–
68х121

,

–

.
Разрешение

–

,

,

–

.
Матрица

.
Какое

,

.
Значительным

)
,

всегда
ядреные
,
и
даже
я
(

)
не
могу
долго

на
дисплей
:
глаза
достаточно

устают
от
яркости
,

и
сверхсочных

.
Яркость

,

.

>
Аккумулятор

.
Емкость

,

:

,

–

,

.
В

.
Гаджет

,
сетевой

–

,
7В
–

,
9В
–

,

–

.
Если

то
,
что

выйдет
из
строя
,
то
в

заряд
.
Однозначно
плюс
!

Если

,

,

.

.

.

.
Хотелось

>

.
Громкость

.
В

.

.
Чаще

оно

,

.
Иногда

.
Аппарат

,

.
Такие
