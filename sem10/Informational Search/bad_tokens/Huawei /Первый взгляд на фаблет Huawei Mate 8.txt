цвет
пластика
и
небольшие
закругления
по
краям
-
это
примета
последнего
года
.
На
задней
поверхности
находится
датчик
отпечатка
пальца
,
такое
расположение
необычно
,
но
к
нему
быстро
привыкаешь
.
Удобнее
,
если
бы
датчик
был
на
центральной
кнопке
(
но
ее
тут
нет
)
либо
на
боковой
,
как
в
телефонах
Sony
.
Такое
расположение
,
пожалуй
,
самое
неудачное
,
но
многие
люди
привыкают
к
нему
.
Тут
проблема
только
в
массивном
корпусе
и
в
том
,
что
надо
тянуться
к
кнопке
.
По
качеству
сборки
,
кроме
упомянутой
шероховатости
,
у
меня
нет
никаких
нареканий
.
Тут
нет
ничего
особенного
,
многие
компании
освоили
этот
уровень
сегодня
.
А
вот
упаковка
очень
приятна
,
черная
коробочка
,
конверт
внутри
для
инструкции
и
бампера
.
К
сожалению
,
тут
немного
недосмотрели
,
при
транспортировке
зарядка
и
наушники
трутся
о
черную
коробку
,
на
них
появляется
краска
.
Ее
можно
стереть
,
но
когда
вы
открываете
коробочки
впервые
,
то
не
понимаете
,
почему
ваши
новые
наушники
испачканы
.
К

Дисплей
Вопрос
того
,
какая
матрица
используется
в
Mate
8
,
во
время
презентации
старательно
обходили
стороной
,
да
и
потом
в
кулуарах
как-то
проглатывали
ответ
,
стараясь
не
говорить
об
этом
.
Оказалось
,
что
тут
нет
AMOLED
,
стоит
обычный
IPS-экран
от
Japan
Displays
,
обновленная
версия
того
,
что
мы
видели
почти
два
года
назад
в
Mate
7
.
Разрешение
-
всего
лишь
FullHD
на
6
дюймах
(
368
ppi
)
,
хотя
тут
QHD-экран
был
бы
уместен
.
Напомню
,
что
матрица
в
Mate
7
вызывала
в
то
время
исключительно
положительные
отзывы
,
но
годы
идут
,
и
новое
поколение
AMOLED
заметно
выигрывает
как
по
цветопередаче
,
так
и
по
разрешению
,
минимальной
и
максимальной
яркости
экрана
.
Получается
,
что
дисплей
в
Mate
8
на
среднем
уровне
.
Он
хороший
,
но
не
может
похвастаться
чем-то
необычным
.
Мне
максимальной
яркости
на
улице
зачастую
не
хватало
.
В
настройках
есть
возможность
отрегулировать
цветовую
температуру
,
поменять
размер
шрифта
.
Все
стандартно
,
и
ничего
необычного
нет
.
Автоматическая
регулировка
яркости
немного
приглушенная
,
мне
ее
не
хватало
.
Есть
режим
работы
в
перчатках
.

Аккумулятор
Встроенная
Li-Pol
батарея
имеет
емкость
4000
мАч
.
В
Huawei
во
время
презентации
рассказали
о
каких-то
космических
результатах
работы
аппарата
,
по
их
мнению
,
он
может
крутить
видео
до
17
часов
,
но
при
этом
работает
2
с
небольшим
дня
.
Не
находите
тут
никакого
противоречия
?
Телефон
умеет
крутить
видео
17
часов
,
а
это
одна
из
самых
энергоемких
задач
,
какая
только
может
быть
,
но
при
этом
работает
только
2.3
дня
.
Меня
еще
во
время
презентации
это
удивило
,
и
поэтому
я
попробовал
посмотреть
,
сколько
же
в
реальности
может
работать
полностью
заряженный
телефон
при
проигрывании
HD-видео
в
MX
Player
.
Результат
меня
разочаровал
,
немногим
меньше
7
часов
.
Это
очень
немного
для
батареи
такой
емкости
,
видимо
,
сказывается
потребление
экрана
,
другого
объяснения
не
нахожу
.
Удивительно
,
но
при
умеренном
использовании
аппарат
живет
около
двух
дней
(
3.5
часа
работы
экрана
при
этом
,
передача
данных
Wi-Fi
)
.
К
сожалению
,
я
не
увидел
каких-то
огромных
достижений
в
этом
аспекте
и
,
более
того
,
хочу
отметить
,
что
в
настройках
лучше
понижать
частоту
процессора
,
так
как
на
быстродействии
системы
это
не
скажется
,
визуально
отличий
нет
,
а
вот
время
работы
вырастет
на
10-15
процентов
.
В
комплект
входит
“
быстрая
зарядка
”
,
про
нее
на
презентации
много
говорили
,
но
на
поверку
она
несколько
глупее
,
чем
у
конкурентов
,
так
как
она
умеет
заряжать
только
в
двух
режимах
-
начальном
на
9В/2А
и
последующем
5В/2А
.
У
всех
других

и

,

и

.
В
Mate
8
полная

занимает
150
минут
,
а
вот
зарядить
ее
на
50

можно
примерно
за
45-50
минут
.

.

,

.

,

.

Программные
особенности
Аппарат
построен
на
Android
6.0
,
но
используется
собственная
оболочка
EMUI
4.0
,
версия
подросла
по
сравнению
с
предыдущими
аппаратами
.
Тут
не
так
много
визуальных
отличий
,
но
есть
жесты
,
например
,
можно
постучать
по
экрану
костяшками
пальцев
,
чтобы
вызвать
определенные
функции
.
На
отпечаток
пальца
назначили
не
только
разблокировку
устройства
,
но
и
доступ
к
дополнительным
возможностям
.
С
обновлением
софта
по
воздуху
появится
возможность
делать
скриншоты
на
всю
высоту
страницы
,
в
том
числе
ее
невидимой
части
(
сюрприз
,
и
где
это
мы
видели
?
)
.
Но
в
целом
,
это
типичный
Android
,
в
котором
не
так
много
изменений
.

В
аппарате
система
из
трех
микрофонов
,
шумоподавление
работает
хорошо
,
плюс
у

режим
интервью
,
когда
определяется
местоположение
говорящего
.
Громкость
динамика
хороша
,
его
слышно
в
разных
условиях
.
В
Huawei
считают
основными
покупателями
линейки
Mate
бизнесменов
.
Возможно
,
что
они
есть
среди
покупателей
,
но
я

группой
все-таки
энтузиастов
технологий
.
Цена
на
эту
модель
высокая
-
за
версию
на
32
ГБ
600
евро
,
на
64
ГБ
-
700
евро
.
За
аналогичные
или
даже
меньшие
деньги
можно
купить
Note
5
или
EDGE+
,
которые
выигрывают
практически
во
всем
.
Высокая
стоимость
не
позволит
увидеть
этот
продукт
в
официальных
поставках
в
России
,
он
бы
стоил
на
полке
55-60
тысяч
рублей
,
а
то
и
больше
.
Приятный
аппарат
,
во
многом
копирующий
то
,
что
мы
видели
в
Mate
7
,
но
неоднозначный
.
Нельзя
сказать
,
что
он
выигрышный
и
можно
его

лучше
,
чем

других
компаний
.
Однозначно
Huawei
прогрессирует
,
но
пока

Samsung
явно
не

,
и

.
Вроде
бы

есть
,
но

и

,

,

.
А

.

,

.
