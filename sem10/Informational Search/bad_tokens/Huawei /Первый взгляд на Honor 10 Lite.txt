телефон
на
коротком
проводе
привинчен
к
стойке
,
поэтому
поделать
интересных

не

.
Но

.
У

.
Изначально
думал
,

смажется
,

очень
даже
неплохо
.
В
Honor
отдельно

качественный
режим

ночью
.
И
в

можно
сказать
,
что
не
соврали
.
Но
есть
загвоздка
.
В

надо

аппарат
неподвижно
3

.
Из-за
этого
не
получится
фотографировать
объекты
в
движении
.
Например
,
вот
так
камера
снимает
по
умолчанию
в
тёмном
зале
:
А
вот
что
будет
,
если
включить
ночной
режим
.
Качество
фото
значительно
лучше
,
но
появились
и
элементы
sci-fi
в
кадре
:
А
вот
пример
съёмки
на
фронтальную
камеру
с
эффектом
боке
.
Слегка
подмылило
ухо
,
и
,

,

растерялся
из-за
прожектора
,
который

бил

левую
(
на
фото
)

.
Однако
,

сложный
свет
,
то
можно
говорить
,
что

очень

.
А
вот

пример
видео
.
Тоже
,

,
очень
хороший

:
Про

России

8
февраля
.
Смартфоны

,

Honor
,

–

.

3

,

,

2

3
+
32
ГБ
за
14
990

и
3
+
64
ГБ
за
16
990
рублей
.
Не
забывайте
,

оба

поддерживают
microSD
до
512
Гб
.
Впечатления
За
14
990
рублей
–
это
отличный
аппарат
на
каждый
день
,
у

видимых
проколов
.
Мощности

,

.
У

же

,

,

)

(
14
нм

,

.
У

10

.
Разумеется
,

,
