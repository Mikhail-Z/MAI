,

о

В

о

,

.

.
Дизайн
,

,

,

.
Например
,

:

,

,

(

,

)
.
Я

,

,

,

в

.
Габариты

«

»

x
78
x

,

–
190

(

)

.
В

не

,

.
Несмотря

,

.
Гаджет

выполнен

.
Боковые

,

и

–

,

.
В

две

:
в

и
в

.
Думаю
,
что
второй

будет

более
солидно
.
Со
сборкой
все
отлично
.
Олеофобное
покрытие
есть
,
но
слабое
.
Изначально
на
экран
наклеена
защитная
пленка
,
а
экран
защищен

(
точно
не
пластик
)
.
За

недель

на

не
появилось
глубоких
царапин
.
Сверху
на
лицевой

и

,
фронтальная

и

.
Спикер

и

,

,

идет

.
Индикатора

нет
.
Под

–
сенсорные

Android

.
Снизу
–

(

»
)
,

и

.
Рядом

,

,

не

.
Странно
,
но

«

»
.
Сверху
–
3.5
мм
аудиовыход
.
Справа
нет

,

и

.
С

,

«

.
Под

.
