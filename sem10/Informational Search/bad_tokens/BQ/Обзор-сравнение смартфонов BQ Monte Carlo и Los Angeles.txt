В

Оба

Например

(

)
присутствует

.
Поэтому
я

не

,
а

обзор-сравнение
.
Стоимость

Monte
Carlo
–
6
690

,
а
Los
Angeles
–
6
590

.
Дизайн
,
размеры
,
управляющие
элементы
Необходимо
сразу
оговориться
,
что
в

BQ
,
как
правило
,
делают
устройства
,
которые
,
так

иначе
,
копируют
внешний

или
просто

аппаратов
.
Две

новинки
не

:
BQS-5011
тыльной

и

,
а

практически

(

,

)
.
Лицевые

–

«
нонейм
»
.
Форма

,

–

.
Тыльная

,

у

Note
5
;
задняя

Los
Angeles
более
плоская
.
Более

имеет
гаджет
Los
Angeles
(
143х72х8,8
мм
)
,

в

(
146х72х9

)

.
Оба

,

,

,

,
не

,

,

(

,

.
Несмотря

,

Гаджет

,

.
Расположение

.

Они

.
Под

На

,

,

)
.
Сверху
–

.
С

–

(
у

она

,
у

Angeles
–

)

.
В

,

«
Меню
»
;
спикер

–

,

.
Крышки

.
Под

,

лучше
,

,
есть

и

.
Los

шустрый
и

,

,

,
