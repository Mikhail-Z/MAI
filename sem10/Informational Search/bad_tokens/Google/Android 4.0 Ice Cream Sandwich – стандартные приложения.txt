Android

ОС
,

.
Часть

Описывая

Android
,

.
Вы

,

.
Этот

.

Браузер
изменился
как

с

2.x
,

.
В

больше
схожести
с
тем
браузером
,
что

в
Honeycomb
,

полностью
переработано
ядро
(
он
построен
на
Webkit2
)
.
В
Google
попытались
создать
браузер
,
который
максимально
будет
схож
с
настольной
версией
,
но
при
этом
учитывая
то
,
что
использоваться
он
будет
на
мобильных
устройствах
.
Интерфейс
браузера
привычен
по
мобильным
устройствам
:
строка
поиска
,
а
также
иконка
закладок
страниц
(
Tab
–
всего
может
быть
открыто
до
16
страниц
,
закладки
видны
на
планшетах
,
в
телефонах
их
нет
ради
экономии
места
)
.
При
открытии
каждой
новой
ссылки
вы
можете
открыть
ее
в
фоновом
режиме
,
оставаясь
на
той
странице
,
что
просматриваете
.
Более
того
,
при
загрузке
файлов
(
в
том
числе
в
форматах
,
не
поддерживаемых
аппаратом
,
их
можно
только
сохранить
)
вы
можете
сохранять
их
в
фоновом
режиме
,
что
довольно
удобно
.
С
другой
стороны
,
например
,
загружая
мр3
файл
,
можно
использовать
моментальное
проигрывание
,
когда
он
начнет
воспроизводиться
сразу
же
.
На
мой
вкус
,
по
удобству
этого
параметра
данная
версия
браузера
максимальна
.
При
просмотре
html-страниц
вы
можете
использовать
двойной
тап
для
центрирования
текста
по
ширине
экрана
,
повторение
двойного
тапа
,
и
снова
перед
вами
полная
страница
.
В
настройках
для
каждой
открытой
страницы
можно
выбрать
отдельно
настройки
шрифтов
,
а
также
зума
(
системные
шрифты
и
их
настройки
могут
быть
заменены
,
устанавливается
минимальный
и
максимальный
размеры
шрифта
,
пример
показан
тут
же
в
настройках
)
.
Есть
интересная
опция
инвертирования
цвета
шрифта
и
фона
страницы
(
белый
в
черный
и
обратно
)
,
в
некоторых
условиях
это
будет
интересно
.
На
странице
не
видны
управляющие
элементы
,
в
частности
строка
.
Достаточно
только
сдвинуть
страницу
,
как
строчка
появляется
.
При
наборе
адреса
появляется
подсказка
(
как
последние
набранные
,
так
и
закладки
,
плюс
поиск
по
вводимым
словам
в
сети
)
.
Закладки
могут
синхронизироваться
с
вашим
аккаунтом
Gmail
и
Google
Chrome
,
этот
процесс
происходит
автоматически
и
не
требует
вашего
участия
.
Любая
загруженная
страница
может
быть
сохранена
для
последующего
чтения
без
подключения
сети
(
offline
reading
)
.
Это
удобно
,
когда
вы
не
успеваете
прочитать
страницу
,
а
текст
вам
интересен
.
Напомню
,
что
в
браузере
поддерживается
копирование
текста
и
его
вставка
в
другие
приложения
.
В
рамках
одной
страницы
есть
поиск
текста
,
он
полноценный
.
Также
страница
может
быть
просмотрена
не
в
мобильном
варианте
,
адаптированном
для
телефона
или
планшета
,
а
в
режиме
настольного
ПК
(
desktop
mode
)
.
В
этом
случае
загружаются
все
элементы
страницы
,
она
не
адаптируется
под
устройство
.
На
мой
взгляд
,
большой
разницы
,
вы
не
заметите
,
но
скорость
загрузки
будет
чуть
меньше
.
Наконец
,
надо
отметить
,
что
браузер
умеет
запоминать
все
формы
и
поля
.
Например
,
вы
можете
запомнить
поле
для
автозаполнения
ваших
данных
(
почта
,
ФИО
и
так
далее
)
,
поля
с
логинами
и
паролями
для
определенных
страниц
.
В
настройках
есть
стандартные
поля
,
которые
вы
можете
заполнить
,
это
повышает
удобство
работы
с
браузером
.
Надо
отметить
,
что
в
настройках
также
можно
запретить
автозаполнение
форм
,
запоминание
паролей
и
так
далее
,
все
настройки
отдельные
,
что
позволяет
гибко
настроить
браузер
под
себя
.
В
браузере
можно
отключить
загрузку
картинок
,
ограничить
быстрый
поиск
в
зависимости
от
типа
соединения
(
например
,
при
Wi-Fi
он
будет
работать
,
при
мобильном
нет
)
.
Для
отдельных
веб-сайтов
можно
установить
свои
настройки
,
сейчас
это
геометки
(
включены
или
нет
)
.
Безусловно
,
поддерживается
выбор
кодировки
текста
для
любой
страницы
и
также
выбор
кодировки
по
умолчанию
.
В
разделе
Labs
можно
включить
полноэкранный
режим
и
всплывающее
меню
по
долгому
нажатию
у
края
экрана
.
Учитывая
,
что
во
всех
случаях
браузер
и
так
максимально
задействует
всю
площадь
экрана
,
в
этом
нет
никакой
надобности
.
Браузер
запоминает
cookies
,
но
можно
их
отключить
.
Для
тех
,
кто
хочет
остаться
невидимым
,
создан
режим
инкогнито
,
браузер
делает
одну
из
вкладок
анонимной
,
и

не
определяет
вас
.
Как

,

максимум

настроек
,
что
делает

одним
из

в
мобильном
пространстве
.
В

,

(

-

,

.
В

,

,

.

,

то

иное

.

Календарь
.
Внешний
вид
календаря
переработан
сильно
,
хотя
идеологически
ничего
не
изменилось
.
По
умолчанию
вам
показывается
неделя
со
всеми
вашими
делами
,
причем
можно
использовать
зумирование
двумя
пальцами
,
тогда
часовая
сетка
разъезжается
или
сужается
.
Всего
представлений
четыре
–
день
,
неделя
,
месяц
и
список
дел
.
Поддерживается
произвольное
число
календарей
из
разных
источников
,
в
этом
случае
события
каждого
календаря
выделяются
своим
цветом
.
Из
приятных
нововведений
стоит
отметить
возможность

отображение

недель
,
также
можно
использовать
ваше
домашнее
время
при
путешествиях
(
все

остаются
в
вашей

,
не
перенастраиваются
автоматически
)
.
В
календаре
выставляются
свои
звуковые
сигналы
для
событий
.
При
создании
событий
в
календаре
можно
добавлять
участников
,
в
этом
случае
им
на
почту
может
быть
отправлено
напоминание
.
Ровно
так
же
из
почты
события
могут
добавлять
в
само
приложение
календарь
.
Не
буду
описывать
стандартные
функции
–
это
повторяемые
,
перекрывающиеся
события
и
так
далее
.
Все
на
уровне
современных
телефонов
.

>

,

.

Будильники

.

,

.
Почта
(
Email
)
.
Приложение

работы

переработке
,

не
только
ответить
на
письмо
,

переслать
его
.
В
письмах
отображается
html
,
поддерживаются
вложения
(
картинки

в
тексте
)
,
вложения
можно
сохранить
удобным
для
вас
способом
.
В
почте
можно
установить
свои

в
сообщениях
,

размеров
кегля
(

,

есть

письмо
и

максимальных

,

не
заданы
жестко
,
а
пять
размеров
шрифта
-
это
скорее
настройки
для
удобства
)
.
В
почте
поддерживаются
папки
для
IMAP/Exchange

,

,

на

,
так

на
сервере
.
Для
каждой

свои
правила
.
Появились
шаблоны

ответа
,

можете
создать
их
самостоятельно
.
При

(

)
.
Для

EAS
v14
(
как
аутентификация
,

и
ABQ
strings
,
ограничение

вложений
,
их

,

в

)
.
На

.
В

,

,

есть
Метки
(
Labels
)

,

.
Но

.

.
В

В

.
Фактически

,

.

Используются

.

В

.

Сообщения

.
При

)
,

.
В

,

,

,

)

,
оно

SMS
в

или

.
В

поиск
.
Movie
Studio
.
В

можно
редактировать

,

,

,

.

>
Музыка
.
Отображаются

,

,

,

,

,

Недавнее
(
Recent
)
.
В
нем
,

перевернуть
аппарат
в
горизонтальную

,

показаны
обложки
всех
альбомов
.
Не
думаю
,
что

,

(
видимо
,

с
iOS
)
.
В
режиме
просмотра

все

,
можно
отметить
понравившуюся
(
это
не
рейтинг
из

,
а
просто
голосование
,

или

)
.
Все

,

,

(
Thumb
Up
)
.
В

,

ОС
используются
новые
алгоритмы
,

музыка
звучит
несколько
иначе
(
субъективно
лучше
)
.
Добавили
эквалайзеры
–
8
предустановленных
и
один
пользовательский
.
Можно
изменять
каждый
из
эквалайзеров
.
Также
есть
Bass
Boost
и

при
прослушивании
музыки
.
При
переходе
от
музыкального

элементы
управления
плеером
.
Также
при
блокировке
телефона
плеером
можно
управлять
с
экрана
.
Качества
плеера
хватит
для
большинства
людей
,
его
сервисные
возможности
более
чем
достаточны
.

содержанию

Люди
(
People
)
.
В
этом
разделе
собраны
группы
людей
и
их
контакты
.
Три
закладки
,
первая
-
это
классические
группы
.
Можно
создавать
свои
группы
,
они
привязываются
к
аккаунту
Gmail
.
Импортируются
группы
из
Google+
.
На
закладке
контактов
отображаются
все
ваши
контакты
из
разных
сервисов
(
тут
вы
вольны
выбирать
,
какие
сервисы
показываются
,
а
какие
нет
,
например
,
я
не
включил
отображение
контактов
из
Facebook
)
.
Любой
из
контактов
можно
добавить
в
избранное
,
тогда
он
появится
в
третьей
закладке
.
Причем
он
будет
показываться
как
большая
фотография
.
В
той
же
закладке
списком
даны
те
контакты
,
с
которыми
вы
общаетесь
больше
всего
.
При
заполнении
контакта
есть
дополнительные
поля
,
например
,
отношения
(
отец
,
партнер
,
сожитель
и
так
далее
,
довольно
забавное
добавление
)
.
Количество
доступных
полей
максимально
и
учитывает
запросы
самого
взыскательного
пользователя
,
проблем
с
синхронизацией
с
MS
Outlook
нет
(
некоторые
поля
автоматически
добавляются
,
хотя
в
меню
они
отсутствуют
изначально
)
.
Для
каждого
имени
можно
добавить
соединение
с
социальной
сетью
,
тогда
в
общем
списке
будет
показываться
последнее
сообщение
или
статус
.
При
этом
при
просмотре
контакта
в
соответствующей
вкладке
вы
сможете
увидеть
информацию
из
социальной
сети
.
На
мой
взгляд
,
приложение
People
особо
не
нужно
,
оно
не
слишком
функционально
и
копирует
обычный
экран
телефона
,
где
есть
доступ
ко
всем
контактам
и
их
поиску
.
Как
результат
,
можно
считать
его
скорее
ответом
на
WP7
,
где
есть
подобный
хаб
.
За
месяц
использования
телефона
с
ICS
у
меня
не
возникло
ни
разу
потребности
в
этом
приложении
.
К
содержанию
>
>
>
Набор
номера
.
В

номеронабиратель
,

,

,

)
.
В

,

.
Посмотреть

нельзя
(

,

)
.
На

экранах
есть

,
можно
искать
имя
человека
в
записной
книжке
.
Это

,

залезать
в
People
просто
нет
.
Во

страну
и
город
звонящего
.
Поиск
.
Строка
поиска
представлена
на

рабочих
столах
,

есть
иконка
в
главном
меню
.
Поиск
работает
как
в
телефоне
(
контакты
,
сообщения
,
почта
,
пункты
меню
и
все
остальное
)
,
так
и
в
интернете
.
Любопытно
,
что
вы
можете
выбрать
поисковую
систему
по
умолчанию
(
присутствует
и
Bing
)
.
Показывается
история
поиска
в
Google
,
что
также
несомненный
плюс
.
Безопасность
.
Из
интересных
особенностей
стоит
отметить

памяти

его

.
Другое

-

разблокировки
с

камеры
,

достаточно
посмотреть
в

,
и

будет
разблокирован
.
Если
вдруг
ваше
лицо
не
будет
распознано
,
то
на
выбор

или
цифровой
код
,
или
графический
ключ
блокировки
.
Достаточно
удобная
функция
,
которой
приятно
пользоваться
в
повседневной
жизни
.
К
содержанию
>

>

4.x

,

По

,

.
