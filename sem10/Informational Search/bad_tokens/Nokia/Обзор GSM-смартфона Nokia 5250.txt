Это

Основная

Между

.

Переход

.

,

.

,

.

.
Ведь

,

.

Маленькая

Недорогое

,

Альтернатив

Дизайн
,
размеры
,
управляющие

Меньший

диктует
и

,

105х50х14

111х51.7х15.5

в

5228
.
Вес
106

(
в
5228
-
115

)
.
В

,

,

.
Причина

:

,

моделей
,
она
толстая
.
В

,
не
знаю
,
моя
догадка
,
что

означает

стоимость
,
других

нет
.
Аппарат
сбит
настолько
плотно
,
что
даже
сильно
сжимая
его
в
руке
,
не
слышишь
ничего
.
В
этом
огромное

телефонов
Nokia
,

,
есть

.
Тут

.
Качество

,
он

,

,
никаких

не
остается
,
да

.
Специально

,

,

,
оставили

,
его
встроили
в
корпус
.
Крышка

,

ее

.
Компоновка

,

,

,

.
Это

,

,

.
На

,

же

.
На

,

.
Ползунок

,

.
Всего

5250

,

.
Почему-то

,

,

,

.

Дисплей
Диагональ
-
2.8

,

16:9
,

)

Дисплей

Моделям

.
У

,

(

)
.
В

,
углы

,

45

,

.
В

,

и

,
можно

его

.
Экран

.
На

одна

.
Дисплей
не
утоплен
в
корпус
,
он
одного
уровня
с
ним
.
Экран
резистивный
,
что
означает
,

он
реагирует
на
стилус
,
входящий
в
комплект
.
Мне
не
понравилось
,
что

от
Nokia
5228
,

,

5800
.
Возможно
,

,

,

.
На

14

,

.
Дисплей

,

,

.

Клавиатура
,
ввод
информации
Аппаратных
клавиш
на
лицевой
поверхности
три
-
это
кнопка
посыла
вызова
,
отбоя
,
а
также
клавиша
меню
.
Во
всех
меню
нажатие
клавиши
Отбой
выводит
вас
на
один
уровень
вверх
.
Ввод
текста
можно
осуществлять
двумя
разными
способами
,
точнее
,
с
двумя
разными
видами
клавиатур
.
Во-первых
,
это
обычная
клавиатура
с
последовательным
нажатием
на
кнопки
,
которая
имитирует
на
сенсорном
экране
привычные
телефоны
.
Работает
она
только
в
вертикальной
ориентации
телефона
,
с
ней
легко
работать
одной
рукой
.
Второй
тип
клавиатуры
–
обычная
QWERTY-клавиатура
.
Она
используется
только
в
горизонтальном
режиме
просмотра
.
Клавиши
удобные
,
набирать
можно
двумя
руками
.
Сказать
что-то
особенное
о
том
и
другом
типе
ввода
нельзя
,
устройство
предлагает
наибольшее
число
возможностей
из
всех
существующих
на
рынке
.
Можно
набирать
как
рукой
,
так
и
стилусом
.
Выбирать
вам
.
При
предиктивном
вводе
слово
подчеркивается
,
щелчком
по
слову
можно
подобрать
его
аналог
.
Рукописное
распознавание
текста
,
на
мой
взгляд
,
в
большей
мере
баловство
,
хотя
со
стилусом
вполне
возможная
вещь
,
работа
с
телефоном
не
составит
тогда
затруднений
.
Не
уверен
,
что
многие
используют
данную
функцию
,
хотя
она
явно
востребована
в
Азии
.
Тактильная
отдача
в
аппарате
реализована
аналогично
технологии
VibeZ
от
Samsung
.
Аппарат
вибрирует
при
нажатии
,
эта
вибрация
неплохо
ощущается
,
дает
обратную
связь
.
Ощущения
,
что
нажата
определенная
клавиша
,
нет
,
это
скорее
реакция
,
что
нажатие
произошло
.

Прослушивание

25.5

Меньшая

Но

проблем
,
в
то
же
время
название
текущей
композиции
на
экране
не
отображается
.
К
содержанию
>
>
>
Камера
Здесь
применяется
CMOS-матрица
,
разрешение
-
2
мегапикселя
,
автофокус
отсутствует
.
Типичная
бюджетная
матрица
,
без
каких-либо
особенностей
.
Настройки
камеры
следующие
,
вы
можете
выбрать
одно
из
3
разрешений
:
Print
2M
–
Large
Print
0.8M
–
Medium
Multimedia
message
0.3M
Есть
следующие
режимы
(
Scene
mode
)
:
Automatic
,
User
Defined
,
Portrait
,
Landscape
,
Sports
,
Night
.
Можно
выставить
сетку
на
экране
,
установить
таймер
для
съемки
себя
.
Цветовые
эффекты
следующие
:
Sepia
,
B
&
W
;
,
Vivid
,
Negative
.
Баланс
белого
–
автоматический
,
Sunny
,
Cloudy
,
Incandescent
,
Fluorescent
.
Контрастность
от
-2
до
+2
.
Резкость
по
трем
параметрам
,
выставляется
ISO
(
low
,
medium
,
high
)
.
Одним
словом
,
вполне
обычные
настройки
камеры
.
Давайте
взглянем
на
фотографии
,
вы
сами
оцените
,
что
камера
звезд
с
неба
не
хватает
,
но
вполне
комфортна
для
этого
аппарата
.
Запись
видео
.
При
записи
видео
число
настроек
значительно
меньше
,
чем
в
режиме
фотографирования
.
Присутствует
программный
стабилизатор
картинки
.
Баланс
белого
–
автоматический
,
солнце
,
облачно
,
Incandescent
,
Fluorescent
.
В
качестве
эффектов
можно
использовать
Sepia
,
Black
&
White
;
,
Negative
,
Vivid
.
Режим
съемки
–
только
автомат
и
ночной
.
Максимальное
разрешение
–
640х480
точек
,
при
этом
формат
–
mpeg4
,
возможно
отключить
запись
звука
(
регулировки
качества
кодирования
нет
,
всегда
30
кадров
в
секунду
)
.
Общее
время
записи
ограничено
только
свободным
местом
.
Плюс
,
поддерживается
разрешение
640х350
точек
,
оно
адаптировано
под
экран
устройства
,
но
записывать
в
нем
не
имеет
смысла
,
если
вы
будете
смотреть
на
ПК
или
другом
телефоне
свое
творчество
.
Пример
видео
(
mp4
,
6,7
мб
)
>
>
>
К
содержанию
>
>
>
Впечатления
По
качеству
связи
аппарат
не
имеет
никаких
проблем
,
все
хорошо
.
Громкость
звонка
высока
,
его
слышно
в
разных
условиях
.
Динамик
на
максимальной
громкости
не
так
чист
.
Виброзвонок
слабенький
,
он
ощущается
не
всегда
.
По
версии
Nokia
,
бюджетная
модель
должна
стоить
в
Европе
115
евро
без
учета
налогов
,
что
делает
ее
автоматически
самым
дешевым
аппаратом
компании
с
сенсорным
экраном
(
S40
Touch
в
рассмотрение
не
берем
)
.
Меньшая
диагональ
экрана
накладывает
определенные
ограничения
,
модель
получилась
не
столь
удобной
,
как
старшие
собратья
.
Комплект
поставки
таков
,
что
в
аппарате
нет
ни
карты
памяти
,
ни
USB-кабеля
,
максимальная
экономия
,
которая
оправдана
в
этом
случае
.
В
России
модель
на
один
месяц
поставляется
эксклюзивно
в
Связной
,
стоимость
аппарата
составляет
6
000
рублей
,
что
выше
цены
на
Nokia
5228
(
около
5
500
рублей
там
же
в
Связном
)
.
При
такой
цене
понятно
,
что
выбор
Nokia
5228
оправдан
,
тут
больше
экран
,
и
это
играет
роль
.
Удивительность
ситуации
связана
с
ценовой
политикой
Nokia
относительно
российского
рынка
:
оптовая
стоимость
без
учета
НДС
на
Nokia
5250
составляет
4
300
рублей
,
а
цена
на
Nokia
5228
-
всего
4
000
рублей
.
Видимо
,
это
связано
с
тем
,
что
в
компании
опасаются
,
что
дешевая
модель
способна
снизить
продажи
других
решений
.
В
Европе
стоимость
Nokia
5228
и
Nokia
5250
находится
на
одном
уровне
,
или
5250
чуть
дешевле
.
При
такой
ситуации
стоит
покупать
Nokia
5228
,
в
то
время
как
продажи
Nokia
5250
взорвутся
в
момент
,
когда
компания
сделает
ее
по
стоимости
ниже
на
15-20
евро
.
В
таком
случае
этот
аппарат
будет
продаваться
в
больших
количествах
.
Но
,
к
сожалению
,
при
такой
ценовой
планке
он
не
может
составить
конкуренцию
корейским
сенсорным
телефонам
в
сегменте
до
100
евро
.
И
это
самый
большой
минус
этой
модели
.
Она
хороша
для
тех
,
кто
выбирает
Nokia
.
Для
всех
остальных
это
-
выстрел
мимо
цели
.

форуме
