без
этого
.
Многие
смартфоны
на
Android
также
быстрее
.
Модели
на
Bada
выглядят
,
как
минимум
,
такими
же
быстрыми
,
а
зачастую
чуточку
быстрее
во
многих
операциях
.
К
сожалению
,
это
не
самое
быстрое
устройство
.
И
ведь
речь
идет
о
моделях
,
которые
нам
известны
,
которые
представлены
на
рынке
.
Я
не
забегаю
вперед
и
не
говорю
об
Android
3.0
,
так
как
для
этого
рано
,
а
там
производительность
выросла
,
и
намного
.
Я
также
не
беру
в
расчет
самые
топовые
устройства
,
которые
стоят
в
разы
дороже
.
Нет
,
примерно
тот
же
класс
.
В
Nokia
,
рассказывая
о
N8
,
все
время
делают
акцент
на
мультимедийных
возможностях
модели
,
они
хороши
,
но
не
впечатляющи
настолько
.
Обычный
плеер
для
видео
,
поддержка
DivX
,
что
также
в
плюс
аппарату
)
.
Камеру
стоит
отнести
к
плюсам
модели
,
но
только
в
области
фотографии
и
разрешения
снимков
.
Критически
качество
снимков
не
выросло
,
а
запись
видео
-
на
уровне
других
моделей
(
Vivaz
пишет
аналогично
,
Motorola
XT720
также
,
да
и
другие
модели
,
это
стандартная
функциональность
для
рынка
)
.
В
повседневной
жизни
к
минусам
относится
браузер
,
экран
для
просмотра
видео
далек
от
идеала
(
смотреть
можно
,
но
это
не
наилучшее
качество
)
.
Можно
считать
это
придирками
,
но
совокупно
аппарат
просто
не
поражает
воображения
.
Знаете
,
у
меня
как
у
владельца
Nokia
N97/X6
ощущение
от
Nokia
N8
двоякое
.
С
одной
стороны
,
это
наскучивший
интерфейс
,
который
быстрее
,
но
имеет
те
же
недостатки
(
экран
не
всегда
реагирует
,
виджеты
рабочего
стола
не
слишком
информативны
–
зачем
мне
одна
строка
из
Facebook
или
Twitter
?
)
.
С
другой
стороны
,
появились
приятные
добавления
–
редактор
фото
и
видео
(
мне
они
не
нужны
вовсе
)
,
USB
OTG
,
подключение
по
HDMI
(
также
не
нужно
)
плюс
интеграция
программ
в
саму
платформу
(
например
,
Search
и
карты
)
.
Почтовый
клиент
стал
получше
,
но
не
ошеломляюще
лучше
.
Интересно
,
что
в
Nokia
теперь
многие
проекты
будут
развиваться
отдельно
.
Мы
это
видим
на
примере
карт
,
их
можно
загрузить
отдельно
на
устройства
по
мере
обновления
.
В
ноябре
также
будет
представлена
бета-версия
браузера
под
восьмой
версией
,
полная
версия
,
видимо
,
ожидается
для
Nokia
N8
вместе
с
первым
основным
обновлением
в
феврале-марте
.
Ровно
тот
же
подход
действует
для
почтового
клиента
.
Мне
этот
подход
нравится
больше
,
чем
ожидание
отдельной
прошивки
для
устройства
.
Это
эволюционное
развитие
,
которое
не
поражает
воображения
.
Лучше
,
чем
раньше
,
но
никакого
прорыва
нет
.
На
фоне
конкурентов
,
если
убрать
за
скобки
камеру
и
фотосоставляющую
,
есть
множество
более
интересных
решений
.
Ведь
нас
не
интересует
тут
скорость
развития
платформы
,
а
надо
понять
,
стоит
ли
этот
продукт
своих
денег
.
Думаю
,
что
стоит
,
так
как
Nokia
устанавливает
такую
стоимость
,
исходя
из
необходимости
продать
множество
Nokia
N8
.
И
ориентиром
тут
служит
Samsung
Wave
,
который
уже
начал
продаваться
ровно
по
той
же
цене
,
но
раньше
на
2
месяца
.
Эти
продукты
в
чем-то
сравнимы
,
в
чем-то
-
нет
.
Они
разные
.
Но
ориентируются
примерно
на
одного
и
того
же
потребителя
.
А
ведь
есть
еще
продукты
от
HTC
,
Motorola
,
Sony
Ericsson
.
И
тут
возникает
вопрос
,
что
Nokia
N8
хороша
для
тех
,
кто
не
пользовался
ничем
,
кроме
Nokia
,
например
,
никогда
не
пробовал
Android
.
Эта
ОС
сегодня
на
взлете
и
привлекает
все
больше
внимания
,
с
конца
октября
,
уверен
,
состоится
очередной
взлет
интереса
и
,
как
следствие
,
продаж
.
Ведь
Android
уже
превратился
во
взрослую
ОС
и
имеет
либо
паритет
по
функциям
с
Symbian
,
либо
обходит
его
по
многим
параметрам
.
Итого
,
мы
получаем
продукт
,
который
подойдет
тем
,
кто
привык
к
Symbian
.
Они
не
разочаруются
,
работать
телефон
будет
лучше
,
чем
предыдущие
модели
,
но
не
идеально
.
Те
,
кто
ищет
наилучшее
,
самое
технологически
интересное
решение
,
эта
модель
таковой
не
станет
.
Запас
прочности
в
технологиях
у
нее
на
пару
месяцев
.
Если
сравнивать
Nokia
N8
с
N9
,
то
последний
аппарат
в
плане
интерфейса
просто
стоит
на
другой
эволюционной
ступеньке
.
Заимствовав
многое
из
Android
(
статусная
строка
,
ряд
анимационных
элементов
)
,
что-то
из
Bada
,
в
MeeGo
добились
внешнего
сходства
с
современными
ОС
.
В
S^3
такой
полировки
интерфейса
просто
не
было
.
Это
промежуточный
продукт
,
как
и
Nokia
N8
.
Задача
этой
модели
-
поддержать
продажи
Nokia
на
фоне
конкурентов
,
которые
выигрывают
технологиями
и
загоняют
компанию
в
нижний
ценовой
сегмент
.
Задача
-
пустить
пыль
в
глаза
,
отвлечь
внимание
от
самой
ОС
,
от
того
,
что
она
промежуточная
.
Отвлечь
на
пару-тройку
месяцев
,
пока
не
будет
представлена
следующая
версия
Symbian
.
Добротный
,
но
проходной
продукт
,
не
поражающий
воображения
.
Вот
,
что
такое
Nokia
N8
.
Шум
вокруг
этой
модели
поутихнет
через
пару
месяцев
,
впереди
нас
ждут
другие
герои
от
Nokia
,
которые
в
плане
технологий
выглядят
интереснее
,
чем
этот
продукт
.
И
поэтому
только
цена
способна
сохранить
его
,
особенно
на
фоне
конкурентов
.
Впрочем
,
в
Nokia
это
прекрасно
осознают
.
И
даже
жертвуют
своими
иными
продуктами
,
например
,
стоимость
Nokia
C7-00
выставляется
на
сравнимом
с
N8
уровне
,
что
моментально
делает
этот
аппарат
непривлекательным
(
разница
в
цене
около
50
евро
)
.
Получается
,
что
задача
Nokia
-
продать
как
можно
больше
Nokia
N8
,
и
для
этого
прикладываются
все
усилия
.
Вердикт
–
отличное
обновление
для
текущих
моделей
от
Nokia
,
но
не
прорыв
в
рамках
рынка
.
Праздник
для
поклонников
продуктов
Nokia
,
для
остальных
-
интересная
модель
только
в
области
фоточасти
.
Если
она
не
интересна
так
сильно
(
именно
фотографии
)
,
то
можно
смотреть
на
другие
решения
.
К
сожалению
,
это
не
решение
«
все
в
одном
»
,
каким
был
Nokia
N95
.
На
меня
аппарат
не
производит
ошеломляющего
впечатления
,
хорошая
рабочая
лошадка
,
например
,
тот
же
Nokia
C7
мне
нравится
больше
,
но
не
нравится
его
ценовое
позиционирование
.
Но
Nokia
N8
вполне
способна
привлечь
внимание
и
за
счет
низкой
стартовой
цены
будет
востребована
.
Уверен
,
что
этот
аппарат
станет
заметным
продуктом
для
Nokia
.
