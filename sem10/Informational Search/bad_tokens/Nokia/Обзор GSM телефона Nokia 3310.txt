Действительно

.
``
Мы

,

Хотите

В

.

.

-

Внешне

Однако

Из

.

.

,

,

,

,

Безусловно

.

.
Честно

,

.
Казалось

,

,

поддавалась…
Пришлось

``
грубую
мужскую
силу
''
,

странно
тоже
справилась

.
С

еще

.
Может

?
А

,

,

,

.
Короче

``

''

,

.
Слава

,

3210

.
со

.
С

.

.
В

.
Дополнительные

В

,

.
Руководство

русском
и

.
Несмотря

,

,

,

,

,

.
О

.
Хочется

.
Аккумулятор
,

,

,

-
1250мАч

Без

)
.
По

2ч30мин.-
4ч
.
30
мин.
,

55ч-260ч
.
Недостатком

.
С

(
850
мАч
)

,

-
2ч.20мин
.
-
4ч.15мин.
,

-
55ч.-245ч
.
Зато

.
Дисплей
После
включения

появляется

,
которая
легко
редактируется
.
Причем
,
что
самое
приятное
,
написать
ее
можно
русскими
буквами
.
Далее

в
сети

,

-
часы
,

-
индикатор

(
5

)
,

-

(
5
делений
)
.
То

,

,

у

.
Не

,

Вам
,

.

Обычный
''

''

Еще

Всего

Дисплей

.

В

На

Поэтому

Меню
Система

же
,

у

аппаратов
Nokia
-

.
Управляется

с

одной
кнопки
,

иерархия

и

,

.
путем

``
Меню
''
,

.
Основные

,

13
,

,

.
Также

,

С
''

.
В

,

-
сигнал

и

.
Язык
,

,

семи

(
английский
,
французский
,

,

и

)
.
Я
естественно

русский
.
Руссификация

.
Сделана
она

.
Единственное
,

,

``
Звонки
''

``
Сигналы
''
.
Очень

Звонки
''

,

,

.
А

Сигналы
''

Все

,

.
Мелодии
К

,

.
Nokia
3310

Вам

Здесь

35

Мне

The

''

Но
,

,

,
Вас

Но

Мне
,

Большим

.
Выбор
режима
осуществляется
двумя
способами
,
либо
с
помощью
клавиши
на
верхней
панели
телефона
,
либо
в
пункте
меню
``
Режимы
''
.
Интересно
,
что
смену
настроек
профилей
,
можно
выполнить
только
в
разделе
меню
``
Режимы
''
.
Всего
можно
задать
6
профилей
,
причем
4
из
них
заводские
(
``
Обычный
''
,
``
Без
звука
''
,
``
Короткий
''
,
``
Громко
''
)
.
Телефонная
книга
Память
телефонной
книги
Nokia
3310
в
отличие
от
многих
других
моделей
ограничивается
только
памятью
SIM-карты
,
собственной
памяти
телефон
не
имеет
.
Однако
в
большинстве
случаев
такого
количества
номеров
,
хватает
и
больше
не
требуется
.
Попасть
в
телефонную
книгу
можно
либо
через
меню
,
либо
нажав
нижнюю
стрелочку
клавиши
прокрутки
.
Однако
во
втором
случае
редактировать
записи
нельзя
.
Любопытно
,
что
этот
аппарат
позволяет
сохранять
имена
абонентов
,
написанные
русскими
буквами
.
Вопрос
плюс
это
данной
модели
или
минус
-
очень
спорный
.
С
одной
стороны
,
безусловно
,
удобно
,
когда
имена
имеют
привычный
вид
.
Но
с
другой
-
очень
неудобная
система
набора
русских
символов
,
так
как
их
приходится
набирать
наобум
.
Ведь
на
кнопках
есть
надписи
только
латинских
букв
.
Автоматически
становится
неудобной
и
система
поиска
того
или
иного
номера
.
Если
отказаться
от
кириллицы
,
то
тоже
неудобно
.
Дело
в
том
,
что
при
наборе
первыми
идут
именно
русские
символы
,
а
потом
уже
латиница
.
Поэтому
приходится
делать
массу
ненужных
действий
,
как
при
наборе
,
так
и
при
поиске
.
Еще
минусом
является
и
то
,
что
при
смене
аппарата
на
другой
,
который
не
поддерживает
русский
язык
в
таком
виде
,
эти
надписи
превратятся
не
понятно
во
что
.
Открывается
очень
богатое
поле
для
фантазий
!
В
этой
модели
есть
опция
задания
персональных
звонков
абонентам
из
телефонной
книги
.
При
включении
``
Определителя
номера
''
Вы
сможете
идентифицировать
звонящего
,
еще
не
доставая
телефона
.
Главное
,
не
перепутать
мелодии
!
Еще
одним
серьезным
``
наворотом
''
телефона
является
голосовой
набор
.
Таким
способом
можно
набрать
до
8
номеров
.
Работа
этой
функции
не
всегда
идеальна
(
в
шумных
местах
может
и
не
происходить
распознавание
)
,
но
вполне
приличная
.
Очень
удобно
пользоваться
ей
зимой
,
когда
с
помощью
нажатия
всего
одной
клавиши
(
долгое
удержание
кнопки
``
Меню
''
)
можно
вызвать
нужный
номер
.
До
8
номеров
телефонов
доступны
для
быстрого
набора
.
Причем
при
нажатии
комбинации
``
цифра
''
+
``
Меню
''
телефон
начинает
звонить
по
номеру
,
который
вы
присвоили
данной
цифре
,
а
при
нажатии
комбинации
``
цифра
''
+
``
#
''
вызов
будет
производиться
по
номеру
,
который
находится
в
ячейке
памяти
,
соответствующей
этой
цифре
.
Списки
входящих
и
исходящих
звонков
находятся
в
меню
``
Звонки
''
.
Причем
входящие
звонки
в
Nokia
3310
поделены
на
две
разных
группы
.
Отдельно
идет
список
(
до
10
номеров
)
с
принятыми
вызовами
,
отдельно
-
с
не
принятыми
(
до
10
номеров
)
.
В
списке
исходящих
звонков
может
находится
до
20
последних
набранных
номеров
.
Работа
с
SMS-сообщениями
В
данной
модели
телефона
работа
с
SMS-сообщениями
очень
разнообразна
.
Во-первых
,
сообщения
можно
отправлять
и
на
русском
языке
.
Однако
с
набором
текста
опять
же
встает
та
же
проблема
,
что
и
при
наборе
имен
в
телефонной
книге
.
Также
надо
отметить
,
что
у
многих
пользователей
при
передаче
русских
сообщений
возникает
масса
проблем
такие
,
как
нечитаемый
формат
текста
на
телефоне
получателя
,
невозможность
потом
отправить
сообщения
,
написанные
латиницей
с
собственного
телефона
и
др
.
Меня
правда
,
сия
учесть
миновала
,
но
,
возможно
,
мне
,
пока
просто
везло
.
При
возникновении
таких
проблем
надо
обращать
большое
внимание
на
кодировку
символов
:
``
Уникод
''
или
``
GSM-алфавит
''
,
автоматически
это
иногда
делается
не
совсем
верно
.
И
от
этого
возникают
проблемы
.
Во-вторых
,
есть
возможность
режима
SMS-чата
(
функция
``
Диалог
''
)
.
Вы
указываете
свой
псевдоним
,
и
на
экране
отображается
диалог
,
как
,
например
,
при
использовании
ICQ
на
компьютере
.
На
другой
стороне
,
если
телефон
не
поддерживает
чат
,
высвечивается
обычное
SMS-сообщение
с
псевдонимом
вначале
.
Очень
удобная
функция
,
так
как
приходящие
сообщения
не
остаются
в
памяти
,
и
стираются
сами
при
каждом
следующем
.
В-третьих
,
есть
система
интеллектуального
ввода
текста
T9
.
У
меня
она
работала
на
трех
языках
-
английском
,
немецком
и
французском
.
В-четвертых
,
можно
посылать
очень
длинные
сообщения
(
до
480
символов
)
.
Автоматически
они
разбивается
на
части
,
и
в
зависимости
от
длины
текста
,
сообщение
пойдет
в
виде
одной
,
двух
или
трех
SMS
.
Существует
индикатор
,
который
показывает
,
сколько
символов
уже
осталось
до
конца
,
и
на
сколько
сообщений
уже
разбито
ваше
.
В-пятых
,
много
приятных
мелочей
.
В
сообщения
можно
вставлять
смайлики
или
``
улыбочки
''
,
которые
позволяют
передать
различные
эмоции
.
Когда
встречаешь
в
тексте
такие
значки
,
он
всегда
становится
намного
живее
.
По
умолчанию
установлено
5
улыбочек
,
но
их
можно
менять
,
еще
для
трех
новых
есть
место
.
Также
можно
вставлять
и
графику
(
до
13
картинок
)
.
Наконец
,
есть
шаблоны
сообщений
(
5
по
умолчанию
+
3
своих
)
.
Некоторые
заготовки
достаточно
часто
используемых
фраз
,
типа
``
я
задерживаюсь
''
,
``
перезвони
''
и
т.д
.
избавляют
Вас
от
необходимости
каждый
раз
набирать
новое
сообщение
.
Шаблоны
сделаны
на
английском
языке
.
Если
говорить
о
работе
редактора
SMS
в
целом
,
то
можно
отметить
следующие
недостатки
:
некоторая
задержка
при
открытии
сообщений
,
расположение
сообщений
в
списке
по
именам
,
а
не
по
времени
прихода
,
только
ручное
удаление
сообщений
при
недостатке
памяти
,
необходимость
в
ручную
снимать
блокировку
клавиатуры
при
приходе
нового
сообщения
.
Дополнительный
набор
инструментов
Помимо
всего
прочего
,
в
телефоне
имеется
будильник
,
который
может
работать
даже
при
выключенном
телефоне
,
причем
в
двух
режимах
.
Кроме
основного
режима
``
Подъем
''
можно
еще
включить
режим
``
Дремать
''
.
В
этом
случае
раздастся
еще
один
звонок
через
5
минут
.
Так
же
есть
таймер
обратного
отсчета
и
секундомер
.
И
тот
и
другой
можно
включить
и
вернуться
в
режим
ожидания
.
Тогда
на
экране
появляется
небольшая
иконка
.
Цена
деления
у
таймера
обратного
отсчета
и
секундомера
разные
.
У
первого
-
1
минута
,
а
у
второго
0.01
секунды
.
Есть
таймеры
разговоров
.
В
меню
``
Звонки
''
можно
посмотреть
общую
длительность
разговоров
,
длительность
входящих
звонков
,
исходящих
звонков
и
последнего
звонка
,
а
также
стоимость
разговоров
.
Быстро
и
сразу
после
беседы
время
последнего
разговора
определить
нельзя
,
для
этого
надо
обязательно
лезть
в
меню
.
Из
полезных
функций
-
еще
органайзер
,
который
может
содержать
до
10
напоминаний
,
калькулятор
и
конвертор
валют
.
Если
изначально
ввести
курс
обмена
,
то
,
потом
можно
быстро
осуществлять
пересчеты
,
нажав
дважды
только
одну
клавишу
прокрутки
меню
.
Также
имеется
переадресация
(
меню
``
Перевод
''
)
.
Есть
возможность
переадресовывать
все
звонки
,
можно
,
когда
телефон
занят
или
нет
ответа
,
а
можно
,
когда
аппарат
выключен
.
При
переадресации
``
Нет
ответа
''
можно
задать
интервал
времени
(
от
5
до
30
секунд
)
,
после
которого
идет
перенаправление
.
В
телефоне
есть
еще
4
игры
:
``
Snake
''
,
``
Space
Impact
''
,
``
Bantumi
''
и
``
Pairs
''
.
В
играх
есть
собственные
установки
:
включение/выключение
звука
,
вибросигнала
,
подсветки
.
В
обычных
настройках
регулировки
подсветки
не
имеется
.
Чувствительность
телефона
и
качество
речи
Несмотря
на
то
,
что
этот
телефон
,
как
уже
было
замечено
выше
,
имеет
встроенную
в
корпус
антенну
,
чувствительность
его
достаточно
высока
и
ничуть
не
страдает
.
Во
многих
местах
со
слабым
приемом
телефон
работал
абсолютно
нормально
,
без
сбоев
.
Таким
образом
,
внутренняя
антенна
дает
только
одни
преимущества
:
удобно
носить
,
нет
никакой
случайности
того
,
что
антенна
обломится
,
более
интересный
дизайн
.
В
редких
случаях
появлялось
эхо
,
но
в
целом
качество
звучания
тоже
очень
хорошее
.
Звук
мягкий
и
естественный
.
По-моему
,
это
вообще
типично
для
телефонов
Nokia
.
Выводы
По-моему
,
телефон
получился
очень
удачным
.
На
фоне
Nokia
3210
,
Nokia
3310
,
который
также
является
аппаратом
среднего
класса
,
явно
выделяется
на
шаг
вперед
.
Из
глобальных
новшеств
можно
отметить
виброзвонок
,
голосовой
набор
,
режимы
профилей
,
увеличенный
размер
SMS
-
сообщений
,
улучшенный
дизайн
,
возможность
уменьшения
веса
с
помощью
Li-Ion
аккумулятора
,
из
мелких
-
смайлики
,
чат
,
большее
количество
игрушек
и
другие
.
Возможно
,
не
всегда
эти
мелочи
и
жизненно
необходимо
,
но
пользоваться
ими
-
одно
удовольствие
!
Еще
большим
плюсом
Nokia
3310
является
хорошее
качество
звука
и
чувствительность
аппарата
,
а
также
длительная
работа
телефона
без
подзарядки
.
К
минусам
можно
отнести
слабый
вибровызов
,
не
очень
громкий
звонок
,
Ni-MH
аккумулятор
и
не
очень
продуманную
конструкцию
смены
панелей
корпуса
.
Но
достоинства
этой
модели
сильно
затмевают
все
ее
недостатки
.
И
на
этот
раз
финны
оказались
на
высоте
и
подтвердили
свое
мастерство
в
области
мобильной
связи
!
Цена
аппарата
колеблется
в
среднем
от180
до
190
долларов
,
что
вполне
оправдано
.
И
я
думаю
,
что
вы
не
пожалеете
,
если
приобретете
такой
телефон
.
