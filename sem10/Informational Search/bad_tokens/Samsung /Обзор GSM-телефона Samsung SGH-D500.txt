Впервые

и

о

,
наблюдениями
.
Продукт
,

,

,
он

В

.

.
В

,

и

,

написанное

.
Оговорюсь
,

,

моему

,

аппаратов
,

каждом

то
или
иное

,
D500

.
Но
эта
модель
наиболее
близка

,

.
Давайте
попытаемся
понять
,
что

сегодня
,
а
не
годом

.
Долгие

,

,

.
Компания

,

,

.
На

Samsung
выросли
и

,

.
Посмотрите
сами
,
сколько

на

,

и
LG
,
Pantech
,
Innostream
,
Sky
(

,

и

)
.
В
других

,

,

,

,

.
Малоизвестный

,

,

,

,

Samsun
,

и
не

,
их
масса
.
Компания
в

,

,

,
формирует
моду
.
Этот

не
осознается
многими

,
они
слишком
косны
в
своих
оценках
.
В
феврале
руководитель
одной
из
компаний
производителей
на
фразу
о
том
,
что
продукты
от
Samsung
скоро
будут
доминировать
на
рынке
,
совершенно
открыто
и
честно
изложил
свою
точку
зрения
.
Вкратце
ее
изложение
сводится
к
тому
,
что
корейские
компании
могут
предложить
только
внешний
блеск
,
но
функциональность
телефонов
еще
многие
годы
останется
на
низком
уровне
.
Позволю
себе
выдержку
из
прямой
речи
:
«
Они
не
имеют
того
количества
патентов
,
что
и
мы
.
Но
дело
даже
не
в
патентах
,
это
мертвое
право
,
а
в
том
потенциале
,
который
заложен
в
наших
инженерах
,
команде
,
работающей
по
созданию
продуктов
,
продвижению
их
на
рынок
.
Создать
на
пустом
месте
производство
технологичных
моделей
,
которые
будут
на
вершине
дня
сегодняшнего
нельзя
за
несколько
лет
,
на
это
уходят
десятилетия
.
Требуется
понимание
желаний
(
запросов
)
обычных
потребителей
,
сегодня
именно
мы
формируем
моду
на
мобильные
телефоны
,
хотя
Samsung
и
пытается
играть
в
эту
игру
.
Это
позиция
слабого
,
продажа
красивой
обертки
и
пустого
содержания
»
.
В
тот
раз
я
не
смог
согласиться
с
доводами
оппонента
,
не
могу
сделать
этого
и
сейчас
.
Для
понимания
пути
развития
компании
необходимо
знать
,
что
именно
сейчас
компанией
выпускается
и
для
чего
.
Обыватель
с
удовольствием
проявит
свои
знания
в
данной
области
и
сходу
сможет
назвать
такие
продукты
,

мр3
плееры
,
стиральные
машины
,
телевизоры
.
Человек
хоть
как-то
связанный
с
областью
IT

производство
памяти
,
здесь
компания
является
одним
из
лидеров
мирового
рынка
.
Но
список
не
ограничивается
только
этими
областями
,
компания
также
производит
матрицы
для
ПК
,
мобильных
устройств
,
процессоры
,
различную
микроэлектронику
.
Сегодня
на
рынке
мобильных
телефонов
нет
ни
одного
из
производителей
,
который
настолько
полно
может
обеспечить
себя
своими
же
комплектующими
.
Конкуренты
компании
вынуждены

ее

,
иногда
у
них
нет
выбора
.
Не
удивляйтесь
,
случайно
узнав
,
что
в
вашем

от
европейского
производителя
стоит
дисплей
от
Samsung
,
их

у

.
В

от
экранов
,
используемых
в
родных

,
продажные

всегда

худшими

.
Нет
,
безусловно
,
заказчик
может
приобрести
и
новейшие

(

)
,
но

высока
.
Внутри
же
компании
стоимость
своих

,

резко

себестоимость

.
Маржа
компании
на
ряде
продуктов
высокого

превышает
сотню

(

,
расходов
на
R
&

)
.
Ни
один
из
производителей
не
может
похвастаться
такими
показателями
,

,
их

не
демонстрирует
публично
и

Samsung
.
Логика
продвижения
своих
продуктов

,

те
функции
,

или

.
При

на
собственную

и
попутно

марки
,
сделать
продукты
желанными
.
Выбор
проходных
решений
был
очевиден
:
Непривычный
форм-фактор
раскладушки
Упор
на
предложение
цветных
экранов
Полифония
Создание
большого
числа
имиджевых
продуктов
,
формирования
моды
на
них
Постоянное
увеличение
рыночной
доли
компании
в
Европе
показывает
,
что
выбранное
направление
развития
было
верным
.
Задача
первого
этапа
была
решена
,
в
имиджевом
сегменте
компания
безусловный
лидер
,
в
различных
ценовых
сегментах
компания
контролирует
от
15
до
55
процентов
рынка
.
Расширить
свою
рыночную
долю
,
предлагая
еще
большее
число
таких
же
продуктов
нельзя
,
она
уже
близка
к
максимуму
.
Второй
этап
выглядит
также
просто
,
предложить
рынку
модель
с

уровне

моделей
от
конкурентов
,
но
с

имиджевых
аппаратов
.
Естественно
,
что
для
данной
модели
и
последующих
аппаратов
,
также
используется
своя
элементная
база
.
Это
позволяет
на
волне
популярности
марки
добиться
высоких
продаж
и
привлечь
новых
потребителей
,
тех
,
кого
до
сегодняшнего

не
устраивала
функциональность
продуктов
от
Samsung
.
Для
компании
это
новый
сегмент
рынка
,
практически
не
освоенный
,
ведь

продуктов
в
линейке
от
Samsung
никогда
не
было
.
Бизнес
телефоны
от
Samsung
–
робкие
попытки
и
закрытие
проектов
Отсутствие
бизнес
моделей
в
линейке
в
какой-то
мере
было

,
обусловленным

от

.
Компания
предпринимала

,

не

предложить
достойного

функций
.
Одной
из
наиболее
запомнившихся

Q
,

.
Наличие

.
Проект

,

Косвенной

.
До

,

Бизнес
,

.
Именно

,

,
что

Samsung
,

.
Они

цветами
,
звуком
,

.
Компания
не
пыталась
бороться
с

,

,

их

.
В

,

,

.
Стало
необходимым
расширение
продуктового

,
иначе

доля
не
могла

увеличена
.
В

аппаратов

.
В
2003
году

С100
,

.
В

.
В

продажи

с

продуктов
от
Nokia
,
это

чистая
победа
.
Далее
все
было
делом
техники
,
расширение

моделей

,
Х600
,

с

.
Можно
сравнить
ситуацию
с
тем
,
что
,
приоткрыв
себе
дверь
в
этот
сегмент
с

С100
,
компания
затем
влезла
в
него

.
Эти

мы
вспомним

,

говорить
о
развитии
бизнес
линейки
.
В

2004
года
компания
Samsung
не
имела
своих
продуктов
в
двух
больших
рыночных
сегментах
–
бюджетные

,
бизнес
аппараты
.
Здесь
продукты
компании

.
С

,

90
к
10
)
.
Ранее
компания
объявляла

(

70
Евро

)
,
но

.
Это

,

,

.
Первый

или
Samsung
D500
Разработка
бизнес
модели
процесс
в
большинстве
случаев
долгий
,
данная
судьба
была
характерна
и
для
D500
.
В
компании
около
года
работали
над
этим
аппаратом
,
причем
именно
он
был
призван
стать
флагманом
линейки
и
удерживать
эту
позицию
в
течение
полугода
.
Основной
задачей
для
разработчиков
стало
обеспечение
полной
функциональности
всех
встроенных
приложений
,
плюс
полная
функциональность
средств
коммуникации
(
ИК-порт
,
bluetooth
)
.
В
качестве
ориентира
негласно
использовались
продукты
от
Sony
Ericsson
.
Разработчики
сравнивали
реализацию
тех
или
иных
функций
,
удобство
их
использования
и
предлагали
свои
варианты
решения
.
Забегая
вперед
,
отмечу
,
что
опыт
оказался
крайне
удачным
,
модель
имеет
свое
яркое
лицо
.
В
качестве
форм-фактора
был
выбран
слайдер
,
он
становится
наиболее
популярным
на
рынке
.
Компания
не
отказывает
бизнес
пользователям
в
возможности
выбрать
продукт
отличный
от
классических
форм
.
Тут
мы
видим
подход
,
отличный
от
европейских
производителей
.
Давайте
вспомним
хотя
бы
Siemens
S65
и
SL65
.
Имиджевый
слайдер
SL65
отличается
от
своего
прототипа
S65
меньшим
экраном
,
худшей
камерой
,
отсутствием
bluetooth
.
То
есть
основная
составляющая
стоимости
-
имидж
,
а
не
функционал
.
В
случае
D500
подход
ровно
обратный
,
предложить
самые
последние
возможности
в
наиболее
популярном
корпусе
.
Именно
поэтому
компания
не
стала
придумывать
новый
сегмент
для
этого
продукта
,
а
позиционирует
его
как
Trend
Leader
.
Скептики
или
те
,
кто
консервативен
в
своих
воззрениях
могут
смело
возразить
,
что
D500
нельзя
отнести
к
бизнес
сегменту
исходя
из
его
форм-фактора
.
На
это
сложно
что-то
возразить
,
каждый
имеет
право
на
свою
точку
зрения
.
На
мой
взгляд
,
в
современном
мире
бизнес
пользователи
используют
уже
давно
и
раскладушки
и
аппараты
других
форм-факторов
.
Основные
факторы
для
выбора
функциональность
,
торговая
марка
(
статус
)
и
дизайн
.
Всего
разработчики
предусмотрели
два
цвета
корпуса
:
черный
и
серебристый
.
Обе
расцветки
выглядят
строго
,
светлый
цвет
более
воздушный
.
Признаюсь
,
что
аппарат
черного
цвета
выглядит
исключительно
стильно
,
подходит
под
любую
одежду
и
случай
.
Модель
D500
перекликается
по
размерам
корпуса
с
Samsung
E800/E820
,
но
аппарат
немного
больше
(
93.5
x
45.7
x
23.5
мм
)
.
Эта
разница
позволяет
говорить
о
том
,
что
телефон
в
какой-то
мере
ориентирован
на
мужскую
аудиторию
,
в
меньшей
степени
на
женщин
.
С
другой
стороны
,
разница
видна
только
при
прямом
сравнении
,
в
руках
у
девушки
аппарат
вполне
уместен
.
Размеры
телефона
стали
комфортнее
,
вес
типичен
для
своего
класса
(
99
грамм
)
.
В
отличие
от
Е800
изменившийся
размер
позволяет
более
комфортно
открывать
аппарат
,
механизм
автодоводки
работает
отлично
.
На
лицевой
панели
есть
резиновая
накладка
,
палец
упирается
в
нее
при
раскрытии
аппарата
.
Говорить
по
телефону
можно
как
в
раскрытом
,
так
и
закрытом
состоянии
.
Более
того
,
в
настройках
присутствует
возможность
сделать
слайдер
активным
при
раскрытии
,
на
закрытие
назначается
отмена
всех
операций
,
либо
их
продолжение
.
Второй
вариант
не
совсем
обычен
,
к
примеру
,
вы
говорите
по
телефону
,
а
затем
закрываете
его
,
а
звонок
не
прерывается
.
Надобность
в
таком
режиме
есть
для
тех
,
кто
часто
пользуется
во
время
разговора
различными
функциями
(
короткие
сообщения
,
записная
книжка
,
органайзер
)
,
но
не
любит
говорить
по
телефону
в
раскрытом
состоянии
.
Дополнительным
плюсом
стоит
назвать
сохранение
текущего
меню
при
закрытии
аппарата
в
режиме
ожидания
.
В
телефоне
есть
автоматическая
блокировка
клавиатуры
,
она
срабатывает
при
закрытии
слайдера
(
можно
и
отключить
)
.
Большинство
функций
D500
доступно
с
помощью
функциональных
клавиш
,
телефон
не
обязательно
открывать
.
Если
вы
набираете
номера
из
телефонной
книги
,
то
и
тогда
раскрывать
аппарат
будет
необходимо
только
при
вводе
информации
.
Экран
в
данной
модели
отображает
262000
цветов
и
имеет
разрешение
176х220
точек
.
По
качеству
экранов
модели
от
Samsung
всегда
отличались
в
лучшую
сторону
от

аппаратов
,
та
же
ситуация
и
здесь
.
Модель
имеет
весьма
яркие
,
живые
цвета
,
дисплей
на
порядок
лучше
экранов
других
производителей
(
исключение
составляет
лишь
компания
Sharp
,
а
также
модели

Motorola
,

экраны
)
.
Качество
экрана
для
своего
класса
не
имеет
аналогов
,
тот
же
Siemens
SL65
отличается
в
разы
по
качеству
картинки
.
Сравнение
с
Motorola
v620
не
показало
значительной
разницы
,
цветопередача
,
яркость
картинки
примерно
сравнимы
.
На
солнце
TFT-экран
ведет

неплохо
,
картинка
видна
.
В
режиме
ожидания

,
они
выступают
в
качестве
хранителя
экрана
.
Навигационная
клавиша
четырехпозиционная
,
в
нее

ОК
,
она
же
служит
для
доступа
к
wap
-браузеру
из
режима
ожидания
.
В
ряде
моделей
кнопка
ОК
присутствовала
,
но
явного
ее

не

,
что
вводило
некоторых
пользователей
в

.
Функциональные
клавиши
удобны
,
хорошо
расположены
.
Цифровая
клавиатура
комфортна
в

,
подсветка
белого
цвета
,
обозначения
видны
в
различных
условиях
.
В

от
других
моделей
Samsung
в
этом
форм-факторе
нет
проблем
при
работе
с
верхним
рядом
цифровой
клавиатуры
.
Скос
верхней
половинки
сделан
удобно
,
палец
не
задевает
за
корпус
.
На
левой
боковой
поверхности
находится
спаренная
клавиша

,
чуть
ниже
можно
видеть
окошко
ИК-порта
.
Забегая
вперед
,
хочу
отметить
,
что
впервые
для
компании
Samsung
ИК-порт
полностью
функционален
.
Не
требуется
никакого
программного
обеспечения
на
ПК
,
передавать
файлы
с

и
на

можно
напрямую
.
Поддерживаются
все
форматы
,
которые
понимает
телефон
.
Если
переименовать
в
JPEG
файл
,
то
и
его
телефон
примет
,
затем
можно
будет
передать
его
далее
(
использование
телефона
для
транспортировки
файлов
)
.
На
правой
боковой
поверхности
вы

видеть
клавишу
доступа
к
цифровой

,
чуть

-
разъем
для
подключения

гарнитуры
.
Разъем
выполнен
в
формате
miniUSB
,
причем

входит
в
комплект
.
Недостаток
гарнитуры
в
том
,
что
она

цвета
и
контрастирует
с
аппаратом
черного
цвета
.
В

телефоном
входит
аксессуар

(
в

не

)
.
Внешне

небольшую
пирамиду
в
цвет
корпуса
телефона
.
Ее
с
помощью
шнурка
можно
прикрепить
к
телефону
.
Сняв
верхнюю
часть
пирамидки
,
вы
увидите
miniUSB
разъем
,
с
его
помощью
устройство
подключается
к
телефону
.
Никто
не
догадался
,
какую
роль
выполняет
SoundMate
?
Функция
этого
устройства
проста
–
это
внешний
динамик
.
В
телефоне
уже
реализована
функция
громкой
связи
,
SoundMate
лишь
расширяет
ее
возможности
.
Субъективно
громкость
возрастает
примерно
в
два
раза
,
не
требуется
отдельно
подключать
устройство
в
меню
,
достаточно
вставить
его
и
оно
начинает
работать
.
Любопытно
,
что
прослушивание
мр3
мелодий
в
режиме
громкой
связи
с
SoundMate
превращает
аппарат
в
небольшой
музыкальный
центр
.
Минус
использования
данного
аксессуара
только
один
,
батарея
садится
значительно
быстрее
.
Так
,
примерно
два
с
половиной
часа

в

батарею

целиком
.
В

примерно
на

.
Аксессуар
подходит
для
всех
моделей
Samsung
,
имеющих
miniUSB
разъем
(
к

,
Samsung
E630
)
,
на

он

продаваться
отдельно
для
других
моделей
.
Ориентировочная
стоимость
составит

10
долларов
.
Аккумулятор
в
данной
модели
является
задней

,

.
Вы

,

и

.
Емкость

-
850
мАч
.
По

производителя
,
он

до
220

.
Во

,

,

.
В

МТС

3

50

20

использования

(

,
камера
,
органайзер
)
.
В

,
если
вы
используете
постоянно
bluetooth
,
то
время
работы

один
день
(
30

разговоров
,
до
25

других
функций
)
.
Признаюсь
честно
,
что
в

батарею
целиком

,
даже
те
,
кто
много
говорит
и
использует
большинство
функций
аппарата
,
останутся
довольны
.
В

о

,
что

время

от
2

3

.
Время

.
На

видеть

,
он

с

.
Вторая

-
Меню
,

,

,
МР3

,

>
