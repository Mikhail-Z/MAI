Обзор
GSM-телефона
Siemens
C35i
Siemens

в
России
.
Ее

.
Мобильные

.
За

Ответ

Сименса
,

2000г
-
Siemens
C35i
,

,

,
на

.

Одна

Однако

Но

,

На

Телефон

,

.
Этому

(
118х46х21
мм
)
.
Когда

,

,

,

нее

.
Уж

!
Самое

,

,

,
если

,

,
не

.
А
,
учитывая
то
,
что

не
так

(
около
110г
)
,
то

полов
.
Возможно
,

.
Но

,
что

и

,

,
синей
,
желтой
.
Всего

6

.
Это

Аккумулятор

,

,

.
В

)
.
Однако

При

.
Это
,

(

15

)
,

,

,

Интернет
(

)

Полная

Зарядное

Скорей

Когда

По

3-5

аккумулятора

Первый

Вставив

Несмотря

Конструкция

Выбор

Улучив

др
.
)
,
то
можно
включить
профиль
``
Тишина
''
.
Тогда
автоматически
громкость
всех
звонков
становится
минимальной
,
включается
виброзвонок
,
а
при
пролистывании
меню
даже
исчезают
легкие
щелчки
,
указывающие
на
то
,
что
вы
перешли
к
следующему
пункту
меню
.
Так
что
вы
никого
не
побеспокоите
.
Все
для
комфорта
Разработчики
Siemens
С35i
позаботились
о
том
,
чтобы
телефон
можно
было
использовать
не
только
как
прибор
для
связи
,
но
чтобы
он
выполнял
и
ряд
других
функций
.
Итак
,
приобретая
одну
трубку
,
вы
вдобавок
получаете
будильник
,
секундомер
,
таймер
с
обратным
отсчетом
,
калькулятор
(
отсутствует
в
Siemens
C35
)
,
конвертор
валют
,
календарь
с
напоминаниями
(
30
записей
)
,
электронные
игры
и
др
.
В
общем
массу
всего
интересного
и
занимательного…
Будильник
и
календарь
с
напоминаниями
(
пункт
``
Встречи
''
)
-
почти
одно
и
то
же
.
Хотя
для
них
предназначены
и
два
разных
пункта
меню
,
в
итоге
вы
все
равно
окажетесь
в
общем
подменю
.
Настроек
для
будильника
много
.
Он
может
напоминать
вам
что-то
ежедневно
,
еженедельно
и
даже
ежегодно
(
например
,
дни
рождения
)
.
Игрушки
хорошо
помогают
убить
время
и
посадить
батарейки
.
Здесь
их
четыре
:
``
Лабиринт
''
,
``
Реверси
''
,
``
Кваттрополи
''
и
``
Сапер
''
.
Возможно
,
конечно
мне
не
хватило
пространственного
воображения
,
но
``
Лабиринт
''
на
меня
не
произвел
впечатления
.
``
Реверси
''
и
``
Кваттрополи
''
-
логические
игрушки
по
заполнению
клеточек
.
Когда
нечего
делать
,
вполне…
``
Сапер
''
-
обычные
минки
от
Microsoft
.
Играть
в
них
можно
,
но
только
человеку
с
очень
хорошим
зрением
,
который
не
боится
его
испортить
.
На
дисплее
телефона
клеточки
выглядят
микроскопическими
,
а
лупа
почему-то
в
комплект
не
входит
.
Может
,
если
бы
экранчик
был
побольше
,
все
было
бы
не
так
трагично
,
но
,
увы…
Таймер
,
считающий
общее
время
разговоров
,
я
нашла
,
но
после
долгих
поисков
.
Он
почему-то
находился
в
меню
``
Время/Плата
''
-
``
Настроить
тариф
''
-
``
Стоимость
''
.
Работа
с
SMS-сообщениями
облегчена
до
минимума
.
Кроме
системы
T9
,
упрощающей
ввод
текста
,
если
,
вы
пишите
,
на
английском
или
немецком
языках
,
и
усложняющей
,
если
вы
набираете
русские
слова
английскими
буквами
,
в
телефоне
имеется
еще
3
шаблона
сообщений
.
При
передаче
сообщений
можно
вставлять
и
графику
.
Всего
в
базе
существует
25
симпатичных
изображений
.
Также
аппарат
оснащен
встроенным
модемом
и
WAP
браузером
(
версии
1.1
)
.
С
помощью
WAP
вы
легко
сможете
путешествовать
по
сети
Интернет
и
получать
самую
свежую
информацию
.
В
целом
,
телефон
очень
удобный
.
С
меню
работать
приятно
,
управляется
оно
всего
двумя
клавишами
,
построено
достаточно
логично
.
Есть
пункт
с
подсказками
,
где
даны
краткие
пояснения
к
наиболее
важным
функциям
.
Объем
меню
также
регулируется
.
Можно
использовать
сокращенный
вариант
с
основными
функциями
,
а
можно
полный
.
Все
сделано
для
увеличения
комфорта
пользователя
.
А
чтобы
было
не
скучно
,
каждый
главный
пункт
меню
снабжен
какой-нибудь
анимашкой
.
Так
,
например
,
рядом
с
надписью
``
Подсказки
''
можно
увидеть
раскрывающуюся
книгу
,
а
рядом
с
пунктом
``
Настройки
''
-
молоток
,
забивающий
гвозди…
Записная
книжка
Попасть
в
телефонную
книжку
очень
просто
и
легко
.
Это
можно
сделать
путем
нажатия
всего
одной
кнопки
с
рисунком
книги
.
Повторное
ее
нажатие
открывает
VIP-раздел
записей
.
Сюда
можно
отнести
,
например
,
людей
,
которым
вы
чаще
всего
звоните
и
к
телефонам
которых
вам
нужен
быстрый
доступ
.
Пользоваться
телефонной
книжкой
очень
удобно
.
Нужный
номер
можно
найти
,
либо
листая
книжку
,
либо
воспользовавшись
системой
поиска
по
имени
(
по
первой
букве
)
.
Всего
в
телефонной
книге
может
быть
сохранено
до
100
номеров
телефонов
(
без
учета
памяти
SIM-карты
)
.
Для
быстрого
набора
может
быть
назначено
до
10
номеров
.
Телефон
также
автоматически
запоминает
10
последних
входящих
звонков
,
10
исходящих
и
10
оставшихся
последних
без
ответа
звонков
с
фиксацией
времени
звонка
.
Посмотреть
весь
этот
список
можно
в
меню
``
Протоколы
''
.
Новую
запись
в
телефонную
книгу
можно
добавить
не
только
в
режиме
ожидания
,
но
и
во
время
разговора
,
что
опять
же
очень
удобно
,
так
как
не
нужно
тратить
время
на
поиски
ручки
и
бумаги
.
Свои
записи
в
телефонной
книге
можно
снабдить
еще
маленькими
иконками
с
рисунками
.
Мелкий
,
но
приятный
пустячок
,
который
отсутствует
даже
в
более
дорогой
модели
Siemens
S35
.
Оригинально
то
,
что
к
записи
можно
добавить
не
только
картинку
из
базы
(
10
штук
)
,
но
и
нарисовать
``
схематический
портрет
''
абонента
.
Для
составления
такого
своеобразного
фоторобота
можно
использовать
9
элементов
.
Когда
раздается
звонок
,
то
при
наличии
услуги
``
Определитель
номера
''
,
на
дисплее
высвечивается
имя
абонента
(
запись
вашего
телефонного
справочника
)
и
симпатичная
картинка
,
закрепленная
за
ним
.
Все
записи
с
графикой
сохраняются
только
в
памяти
телефона
.
Заключение
Если
бы
меня
кто-нибудь
попросил
выразить
свое
мнение
об
аппарате
Siemens
C35i
в
двух
словах
,
я
бы
,
скорей
всего
,
первым
номером
поставила
слово
``
удобный
''
.
Его
,
действительно
,
удобно
держать
;
по
нему
удобно
звонить
;
удобно
разговаривать
,
без
помех
.
Дальше
,
пожалуй
,
перечислять
не
буду
.
Долго…
А
второе
слово
было
бы
-
``
приятный
''
.
Приятный
дизайн
,
приятное
меню
,
много
приятных
картинок
и
анимашек
.
Также
можно
добавить
слово
``
функциональный
''
,
для
полного
представления
.
Безусловно
,
недостатки
имеются
,
и
некоторым
они
,
может
,
даже
покажутся
очень
весомыми
.
Это
,
прежде
всего
,
наличие
в
стандартной
комплектации
только
никель
-
металлгидридного
аккумулятора
,
не
очень
хорошая
связь
в
автомобиле
,
хилый
виброзвонок
,
недостаточно
большой
экран
и
ряд
других
более
мелких
.
Но
в
целом
,
впечатление
об
этом
аппарате
очень
благоприятное
.
При
наличии
массы
достоинств
все
недостатки
плавно
отходят
на
второй
план
.
К
тому
же
нельзя
забывать
,
что
эта
модель
относится
к
аппаратам
среднего
класса
.
Средняя
стоимость
данного
телефона
-
170
долларов
.
Учитывая
соотношение
цены
и
качества
,
мне
кажется
,
что
многие
захотят
приобрести
данную
модель
.
Средняя
цена
при
хорошем
качестве
.
Вот
эта
фраза
и
стала
ответом
на
мой
вопрос
,
который
я
задавали
в
начале
.
Наверное
,
именно
за
это
и
полюбили
российские
покупатели
фирму
``
Siemens
''
.
