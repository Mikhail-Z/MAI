Toshiba
e400
:
сверхтонкий
бюджетный
КПК
Комплект
поставки
:
КПК
Зарядное
устройство
Диск
с
программным
обеспечением
Документация
В
последнее
время
усиливается
конкуренция
в
сегменте
бюджетных
КПК
,
особенно
на
рынке
Pocket
PC
.
Средняя
цена
на
них
опустилась
с
$
300
(
в
начале
2003
г
)
до
$
220-250
в
конце
.
Очередное
бюджетное
решение
представила
компания
Toshiba
,
модель
Toshiba
e400
.
Она
обладает
схожими
с
конкурирующими
моделями
характеристиками
и
на
фоне
других
бюджетных
решений
практически
не
выделяется
.
Для
начала
попробуем
определить
,
что
же
являет
собой
«
бюджетный
»
КПК
,
какие
основные
характеристики
ему
присущи
?
Это
карманный
компьютер
маленького
размера
(
меньше
,
чем
у
старших/дорогих
моделей
)
,
обладающий
одним
слотом
расширения
(
наиболее
часто
–
это
слот
расширения
SD/MMС
)
,
неплохим
цветным
экраном
с
диагональю
не
более
3,5
”
,
бедной
комплектацией
(
обычно
без
док-станции
)
.
В
«
бюджетниках
»
практически
всегда
отсутствуют
беспроводные
адаптеры
(
bluetooth
,
Wi-Fi
)
,
это
прерогатива
дорогих
моделей
.
Таков
«
портрет
»
среднестатистического
Pocket
PC
.
Это
же
описание
можно
легко
перенести
с
незначительными
оговорками
и
на
бюджетные
КПК
Palm
,
но
речь
сейчас
не
об
этом
.
Итак
,
после
небольшого
вступления
приступаем
к
обзору
модели
.
Первое
,
что
бросается
,
если
можно
так
сказать
,
в
глаза
–
это
необычайная
тонкость
наладонника
(
чуть
больше
сантиметра
)
,
буквально
как
обычная
бумажная
записная
книжка
.
Это
же
отмечали
и
остальные
,
кто
знакомился
с
моделью
.
Эльдар
Муртазин
сразу
прозвал
модель
«
Тонкий
»
,
довольно
точно
на
мой
взгляд
.
Однако
по
ширине
и
высоте
(
линейные
размеры
КПК
–
124,5х76,2х10,2
мм
)
он
больше
чем
конкурирующий
с
ним
наладонник
HP
iPaq
1930
(
113,3х69,8х12,8
мм
)
,
из-за
этого
создается
впечатление
,
что
КПК
немного
«
приплюснут
»
.
Корпус
,
можно
сказать
,
«
натянут
»
на
внутреннюю
плату
:
на
задней
части
КПК
проступают
внутренние
детали
,
такие
как
слот
SD/MMC
,
гнезда
для
стилуса
и
наушников
.
Они
рельефно
выделяются
,
это
выглядит
неожиданно
.
В
руке
КПК
лежит
отлично
,
работать
с
ним
удобно
(
не
последняя
роль
в
этом
колесика
прокрутки
,
но
об
этом
ниже
)
.
Вес
в
130
г
позволяет
носить
устройство
практически
где
угодно
(
в
кармане
рубашки
,
джинсов
)
,
дискомфорта
это
не
создает
.
На
этот
раз
вместо
металлического
корпуса
Toshiba
использовала
в
КПК
пластик
синего
цвета
с
черными
боковыми
вставками
.
Последние
на
ощупь
напоминают
резину
и
не
дают
устройству
скользить
в
руке
.
Это
отличает
ее
от
старшей
модели
(
Toshiba
e800
)
,
где
боковые
вставки
выполнены
под
металл
,
из-за
чего
корпус
скользит
в
руке
.
На
лицевой
части
никаких
особенностей
(
все
повторяет
элементы
на
модели
Toshiba
e800
)
.
Под
экраном
–
прямоугольная
клавиша
навигации
с
маленькой
кнопкой
ввода
посередине
,
работает
в
восьми
или
четырех
направлениях
,
это
настраивается
в
меню
(
что
для
аркадных
игр
просто
необходимо
)
.
Около
кнопок
приложений
нанесены
значки
управления
Windows
Media
Player
(
«
стоп
»
,
«
пауза
»
,
«
следующий
клип
»
)
,
впрочем
,
их
раньше
можно
было
назначить
самостоятельно
,
но
теперь
об
этом
позаботился
производитель
.
Справа
от
навигационных
кнопок
находится
динамик
среднего
качества
,
на
максимальном
уровне
громкости
слышен
хрип
.
Минус
,
но
не
слишком
серьезный
минус
.
Слева
–
индикатор
заряда
и
напоминаний
.
Заметим
интересную
особенность
,
если
температура
слишком
низкая
для
функционирования
КПК
,
то
индикатор
перемигивается
тремя
цветами
(
желтым
,
зеленым
и
оранжевым
)
.
По
крайней
мера
так
записано
в
инструкции
,
на
практике
этого
добиться
не
удалось
,
хотя
на
улице
и
стоял
лютый
холод
(
Москва
,
Декабрь
2003
)
.
На
левом
торце
находится
кнопка
диктофона
,
колесико
прокрутки
,
кнопка
«
hold
»
,
инфракрасный
порт
.
Колесико
прокрутки
удобно
для
работы
с
меню
и
чтения
электронных
книг
(
то
есть
,
выполнения
своих
основных
функций
)
,
Toshiba
в
отличие
от
других
производителей
не
отказывается
его
применения
в
КПК
.
Правый
торец
чист
от
управляющих
элементов
.
На
верхнем
торце
находится
кнопка
питания
(
на
наш
взгляд
,
слишком
маленькая
)
,
микрофон
,
слот
SD/MMC
с
поддержкой
SDIO
стандартный
вход
для
наушников
и
гнездо
стилуса
.
На
задней
стенке
КПК
расположена
кнопка
мягкого
сброса
устройства
и
ушко
для
ношения
на
веревке
.
Вес
устройства
вполне
позволяет
носить
его
на
шее
или
запястье
.
Это
хороший
способ
выделится
из
толпы
девушке
или
юноше
–
КПК
не
так
сильно
распространены
как
сотовые
телефоны
.
Стилус
раскладной
(
телескопический
)
,
обладает
средним
размером
и
удобен
в
работе
,
что-то
дополнительно
про
него
сказать
сложно
.
На
нижнем
торце
мы
обнаруживаем
следующие
элементы
:
гнездо
для
подключения
зарядного
устройства
,
коммуникационный
порт
для
подключения
кабеля
или
док-станции
(
последний
не
идет
в
комплекте
,
т.к
.
это
бюджетное
решение
)
,
переключатель
для
жесткой
перезагрузки
устройства
,
он
закрыт
резиновой
заглушкой
.
Не
слишком
удачное
решение
–
резиновая
заглушка
не
выглядит
надежно
,
со
временем
может
отвалиться
(
или
вернее
–
оторваться
)
,
хотя
,
внешнего
вида
устройства
это
не
испортит
.
Буквально
пару
слов
об
экране
,
он
типичен
для
бюджетных
КПК
и
ничем
особенным
не
выделяется
(
характеристики
его
приведены
в
описании
ниже
,
обсуждать
их
тут
,
думаю
,
не
стоит
)
.
Изображение
хорошо
видно
в
помещении
и
на
улице
,
цветопередача
на
высоте
,
в
том
числе
и
белого
.
Единственный
минус
экрана
–
минимальная
подсветка
слишком
яркая
для
чтения
книг
в
темноте
,
устают
глаза
(
но
она
ниже
,
чем
на
модели
HP
iPaq
1930
,
что
радует
)
.
Таким
образом
,
по
экрану
–
все
«
на
уровне
»
.
Автономная
работа
На
КПК
установлен
Li-Ion
аккумулятор
емкостью
980
мАч
.
Он
несъемный
,
поэтому
заменить
его
,
например
на
расширенный
(
большей
емкости
)
или
новый
,
не
получится
.
Это
является
серьезным
недостатком
,
если
КПК
покупается
на
год
и
больше
,
за
это
время
аккумулятор
может
элементарно
сесть
.
По
нашим
тестам
в
режиме
воспроизведения
MP3
(
выключенный
экран
,
звук
через
наушники
)
КПК
проработал
от
одного
заряда
12
часов
,
такой
же
показатель
при
чтении
книги
программой
Haali
Reader
на
минимальном
уровне
подсветки
экрана
.
Это
неплохой
показатель
,
выше
среднего
.
К
примеру
,
модель
HP
iPaq
1930
не
дотягивает
и
до
10
часов
в
этих
режимах
.
При
максимальной
нагрузке
КПК
продержался
3
часа
40
минут
,
тоже
весьма
неплохо
.
Производительность
Наладонник
обладает
стандартным
объемом
памяти
в
64
Мб
(
62
доступно
пользователю
)
.
Этого
с
лихвой
хватит
для
всех
задач
,
на
которые
КПК
рассчитан
.
Тридцать
один
мегабайт
встроенной
постоянной
памяти
логичнее
всего
использовать
для
резервного
копирования
данных
или
для
установки
программ
(
на
первое
время
,
пока
не
куплена
карта
памяти
большего
объема
)
.
При
наличии
большой
карты
памяти
SD
или
MMC
(
на
256
Мб
,
к
примеру
)
преимущества
встроенной
энергонезависимой
памяти
теряются
,
ее
остается
использовать
разве
что
для
резервного
копирования
данных
и
для
тех
случаев
,
когда
у
вас
несколько
карт
памяти
(
например
,
одна
–
для
КПК
,
вторая
–
для
цифровой
камеры
)
и
логичнее
часть
программ
записать
во
встроенную
постоянную
память
.
Это
что
касается
памяти
,
переходим
к
процессору
.
Процессор
установленный
на
КПК
относительно
новый
,
а
именно
–
Intel
XScale
PXA261
300
МГц
(
подробнее
об
этой
серии
процессоров
мы
расскажем
в
обзоре
Toshiba
e800
,
сейчас
на
этом
не
будем
заострять
внимание
,
скажем
лишь
,
что
существенные
отличия
от
PXA255
–
только
на
бумаге
)
.
Как
обычно
,
для
теста
производительности
мы
использовали
пакет
SPB
Benchmark
.
Для
сравнения
были
отобраны
бюджетные
модели
с
одним
слотом
расширения
.
Индекс
описывает
производительность
центрального
процессора
.
Пусть
читателя
не
смущает
отставание
модели
Toshiba
e400
от
конкурентов
по
тесту
центрального
процессора
,
сторонними
утилитами
процессор
можно
легко
(
и
безболезненно
)
разогнать
,
при
этом
результаты
будут
сравнимы
.
Значительное
отставание
«
старенького
»
Rover
объясняется
тремя
причинами
:
неоптимизированная
под
XScale
операционная
система
,
частотой
процессора
в
200
МГц
(
что
логично
)
,
внутренней
шиной
на
100
МГц
у
PXA250
(
вместо
200
МГц
шины
у
PXA255
и
PXA26х
)
.
Этот
индекс
описывает
скорость
работы
с
оперативной
памятью
(
чтение/запись/перемещение
)
.
Средние
показатели
демонстрирует
«
бюджетный
»
КПК
от
Toshiba
по
скорости
работы
с
оперативной
памятью
.
Индекс
описывает
скорость
работы
встроенных
программ
.
Как
видим
,
результаты
e400
–
весьма
средние
.
Во
многом
это
происходит
из-за
того
,
что
КПК
по
умолчанию
работает
на
частоте
300
МГц
.
При
разгоне
картина
меняется
.
Индекс
описывает
общую
производительность
системы
,
Toshiba
e400
–
крепкий
середнячек
по
итоговому
показателю
.
Стоит
помнить
,
что
у
программы
от
SPB
итоговый
индекс
искажается
тестом
графической
подсистемы
.
Для
графического
теста
мы
воспользовались
игрой
PocketQuake
с
отключенным
звуком
(
звук
отключался
консольной
командой
nosound
1
)
,
но
с
сохранением
остальных
настроек
.
Результаты
этого
теста
не
впечатлили
:
При
воспроизведении
видео
или
работе
с
другими
мультимедийными
приложениями
проблем
со
скоростью
работы
не
возникало
.
Все
работало
достаточно
быстро
.
Мощности
и
потенциала
(
разгон
)
процессора
с
лихвой
хватит
на
выполнение
практически
любых
задач
,
хотя
,
казалось
бы
,
этого
и
не
требуется
от
бюджетного
решения
.
Программное
обеспечение
Рассматриваемый
КПК
работает
под
управлением
операционной
системы
Windows
Mobile
2003
for
Pocket
PC
Professional
Edition
.
О
входящих
в
ее
состав
стандартных
программах
подробно
рассказано
в
соответствующем
обзоре
.
Дополнительно
на
КПК
уже
установлено
несколько
программ
,
о
них
и
пойдет
далее
речь
.
Естественно
,
не
обошлось
без
программы
для
резервного
копирования
данных
,
в
данном
случае
она
называется
DataBackup
.
Утилита
позволяет
сохранить
либо
личные
данные
PIM
(
календарь
,
контакты
,
заметки
)
,
либо
все
данные
сразу
.
Избирательной
настройки
того
,
что
именно
нужно
сохранить
–
нет
.
Можно
задать
работу
программы
по
расписанию
–
ежедневное
копирование
или
в
определенный
день
недели
.
Аппаратная
кнопка
«
Home
»
автоматически
запускает
менеджер
задач
,
который
является
альтернативой
стандартному
экрану
Today
.
Что
удобно
–
выводится
список
запущенных
программ
,
отсюда
же
можно
легко
закрыть
ненужные
в
данный
момент
приложения
.
Программа
показывает
уровень
заряда
батареи
и
предоставляет
быстрый
доступ
к
настройкам
.
Внешний
вид
программы
можно
настраивать
.
Альтернативная
программа
для
записи
с
микрофона
называется
Voice
Recorder
.
Ничего
интересного
в
ней
нет
,
особенно
с
учетом
того
,
что
в
отличие
от
«
старшей
»
модели
(
Toshiba
e800
)
,
к
«
младшей
»
не
подключается
внешний
микрофон
.
Качество
записи
с
любыми
настройками
весьма
среднее
–
звук
тихий
и
голос
можно
отчетливо
разобрать
только
при
записи
на
расстоянии
10-50
см.¦
Программа
World
Clock
интересна
прежде
всего
тем
пользователям
,
которые
много
перемещаются
между
часовыми
поясами
.
Она
показывает
местное
время
в
различных
точках
мира
.
Утилита
может
служить
и
будильником
.
Это
все
дополнительные
программы
,
но
наше
внимание
привлекли
звуковые
настройки
.
Уровень
громкости
выставляется
отдельно
для
встроенного
динамика
и
наушников
,
есть
намеки
на
эквалайзер
.
По
неофициальным
данным
,
русификацией
устройства
займется
компания
Paragon
.
На
российский
рынок
КПК
будет
поставляться
полностью
русифицированным
.
Итоги
Новинка
официально
поступит
в
продажу
в
России
в
конце
января
по
цене
$
420
,
это
неоправданно
много
за
бюджетное
решение
(
в
приватных
разговорах
это
признают
и
сотрудники
российского
представительства
Toshiba
,
не
рассчитывая
на
масштабные
продажи
)
.
Модель
будет
актуальна
для
покупки
при
снижении
цены
на
нее
до
$
250
и
ниже
.
