На

.
Большая

,

,

.
Придумать

.
Преимущество

,

,

,

.
В

,

,

–

,

.
Это

,

,

.
Продукт

,

,

,

,

.
Этакое

.
Давайте

,

.

Дизайн
,

,
управляющие

Защищенные

,

решения
–

черный
с
оранжевыми

,
но

Senseit
P8
это

.
Размер

–
132х68х20
мм
,

без

антенны

–

ее
,

6.5

.
С

почти
20

,

,

.
Сравение
с
RugGear
P860
На

-
крепление

,
оно

,

.
Случайно

с

,

–

.
Специально

в

,

,
и

,

.
Если

не
планируете
использовать

,
то

и
убрать

,

,

.
Вес

180

,

.
Но

защиты
только
IP56
,
топить

,
он

этого
.
Его
можно
уронить
в
воду
,
тут
же

–
проблем

не
принесет
,
но

не

.
На
верхнем

кнопка
SOS
,

ее

можно

отправку

различные

,
которые

введете
в
телефон
.
Кнопка
тугая
,
и

.
На
левой

,

-

.
На
правой
стороне
-
две
кнопки

,
а

рации
.
Задняя

,

.
Крышка
прилегает

,
даже

,
ее
придется

с
боку
,
в

,

.
Внутри

,

,

утопить

,

.
Но

IP67

,

.
Поэтому

воде

.
Вынимать

,

.
Вынув

,

,

,

4
ГБ
.
У

–

одной
стороны
,

,
здесь

второй
аккумулятор
,

,

она

.
С
другой

,

,
и

.
Но
есть

–

,

.
Это

,

и

.
На

,

-

,

.

Дисплей
В

2.4

240х320

.
По

,

,

.
Мне

,

,

,

.
На

выцветает
(
TFT
)
,

.
Для

аппаратов
экран

,

.
К
содержанию
>
>
>
Клавиатура
Кнопки
выполнены
из
пластика
,
довольно
большие
по
размеру
,
с
хорошим
ходом
.
Случайных
нажатий
можно
избежать
за
счет
блокировки
кнопок
,
но
в
кармане
при
сильном
сжатии
возможны
нажатия
кнопок
с
цифрами
,
в
то
время
как
функциональные
клавиши
жестче
-
своего
рода
защита
от
случайных
нажатий
.
Подсветка
клавиш
белого
цвета
,
средней
яркости
,
хорошо
видна
в
разных
условиях
.
Обозначения
на
кнопках
двуязычные
,
читаются
хорошо
.
Мне
клавиатура
нравится
,
она
удобная
и
не
вызывает
ни
малейшего
дискомфорта
.
К
содержанию
>
>
>
Аккумулятор
В
комплекте
поставки
два
аккумулятора
,
каждый
емкостью
2000
мАч
.
Учитывая
,

является

,
его

–
производитель

время
в
режиме
ожидания
до
400
часов
,
в
режиме
разговора
до
12
часов
.
В
условиях
умеренного

аппарата
(
30

день
,
несколько
SMS
,
до
часа
музыки
)
он
спокойно
работает
6-7
дней
.
В
то
же
время
,
использование
рации
в

условиях
резко

время

телефона
–
в
городе
это
7-8
часов
,
за
городом
до
одного
дня
.
Объяснить
это
легко
тем
,
что
в
городских
условиях
используется
большое
число
раций
на
том
же
диапазоне
частот
и
аппарат
тратит
больше
энергии
,
да
и
вы
слушаете
чужие
разговоры
,
пусть
и
невольно
–
на
это
уходит
заряд
аккумулятора
.
Время
зарядки
аккумулятора
-
около
2
часов
.

Обычная

,

.
Можно

.

,

Ниже

.

.

,

.
Подключение

Работоспособность

,

с
телефоном
.
Bluetooth
.

,

.

.

.

)

)
Object

)

)

)

В

,

.
В

Меню
,
интерфейс
Обычный
аппарат
на
платформе
MTK
(
MT6235
)
,
в
нем
неплохо
проведена
локализация
.
Главное
меню
представлено
матрицей
2х4
,
это
сделано
из-за
ориентации

.
В
режиме
ожидания
показываются
время
,
давление
и
дата
,
а
также
строка
с
иконками

.
Из
особых

стоит

,

,
высотомер
.
К
сожалению
,
шагомер
не
работает
в
фоновом
режиме
,
что

удобным
.
Другие
функции
работают
нормально
.
Ниже
опишем
основные
возможности
аппарата
,
а
начнем
с
рации
.
Рация
.
Рация
работает
в
разрешенном
диапазоне
PMR446
,
имеет
8
частотных
каналов
и
мощность
до
0.5
Вт
.
В
настройках
можно
указать
мощность
(
нормальная
или
высокая
)
,
от

,
как
плеер
,
радио
работает
в
фоновом
режиме
,
и
на
экране
standby
отображается
название
станции
.
Звук
можно
вывести
как
на
наушники
,
так
и
на
динамик
телефона
.
Диктофон
аппарата
может
использоваться
для
записи
телефонного
разговора
.
Запись
можно
поставить
на
паузу
,
а
потом
возобновить
.
Время
записи
ограничено
лишь
местом
хранения
(
телефон
или
карта
памяти
)
.
Можно
выбрать
кодек
,
посредством
которого
будет
производиться
запись
:
AMR
,
WAV
,
AWB
.
К
содержанию
>
>
>
Впечатления
Очень
громкий
звонок
,
который
хорошо
различим
даже
в
шумных
условиях
.
Такой
же
динамик
и
неплохая
вибра
.
По
качеству
связи
никаких
нареканий
к
аппарату
нет
.
Из
недостатков
стоит
отметить
,
что
при
неустановленной
антенне
рации
,
заглушка
разъема
может
вылетать
.
Плотная
крышка
-
это
и
плюс
,
и
минус
,
быстро
сменить
аккумулятор
не
выйдет
,
но
наличие
второй
батареи
в
комплекте
радует
.
Модель
уникальна
наличием
рации
,
и
прямых
конкурентов
у
нее
не
было
на
момент
появления
,
но
чуть
позднее
в
России
появился
аппарат
RugGear
Explorer
P860
,
который
также
имеет
встроенную
рацию
.
Отличие
моделей
в
том
,
что
в
P860
нет
внешней
антенны
,
аккумулятор
1900
мАч
,
такая
же
камера
и
та
же
платформа
MTK
.
По
большому
счету
,
с
точки
зрения
функциональности
модели
схожи
,
в
RugGear
сохранили
GPS-приемник
,
но
использовать
его
на
практике
нельзя
–
карты
от
китайского
поставщика
,
на
них
Россия
представлена
очень
схематично
.
Поставить
свои
приложения
с
картографией
также
не
получится
.
При
этом
в
стандартном
комплекте
поставки
есть
игры
,
чего
нет
на
Senseit
P8
.
С
другой
стороны
,
оба
аппарата
поддерживают
Java
и
установку
игр
.
Главное
отличие
RugGear
в
том
,
что
уровень
защиты
IP67
,
а
не
IP56
,
как
в
рассматриваемом
аппарате
.
По
большому
счету
,
это
и
является
самой
большой
разницей
,
кроме
цены
.
Стоимость
RugGear
составляет
около
8
990
рублей
(
Связной
,
на
момент
написания
обзора
)
,
а
цена
на
Senseit
P8
7
490
рублей
(
тот
же
магазин
)
.
Разница
в
полторы
тысячи
рублей
довольно
ощутима
,
основной
вопрос
,
который
встает
,
какой
уровень
защиты
вам
нужен
–
что
вытекает
напрямую
из
того
,
как
вы
собираетесь
применять
аппарат
.
Если
это
сплав
по
реке
,
то
лучше
выбирать
модель
с
максимальным
уровнем
защиты
,
если
рыбалка
,
то
будет
достаточно
и
IP56
.
В
любом
случае
,
аппаратов
со
встроенной
рацией
пока
нет
в
достатке
,
и
выбор
только
между
двумя
моделями
.
