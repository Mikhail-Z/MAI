Инструкция
Позиционирование
В
сегменте
недорогих
и
защищенных
кнопочных
телефонов
почти
нет
предложения
,
единичные
модели
,
каждая
из
которых
привлекает
внимание
.
Подобные
аппараты
всегда
берут
как
телефоны
только
для
звонков
,
сообщений
,
обращают
внимание
на
защитные
свойства
корпуса
.
Это
осознанный
выбор
тех
,
кто
отправляется
в
лес
,
на
реку
,
море
,
иногда
тех
,
кто
работает
в
сложных
условиях
.
Требования
к
таким
телефонам
почти
всегда
одинаковы
–
минимальная
цена
,
чтобы
долго
работал
от
одной
зарядки
,
имел
защиту
от
воды
на
уровне
IP67
или
выше
.
До
недавнего
времени
в
России
в
этом
сегменте
была
модель
RugGear
Mariner
RG128
,
она
отличалась
двумя
аккумуляторами
в
комплекте
поставки
,
с
маленьким
аппарат
даже
умел
не
тонуть
,
плавать
по
поверхности
.
Еще
можно
вспомнить
недорогой
Texet
TM511R
,
который
также
имеет
неплохой
уровень
защиты
и
стоит
недорого
.
Texet
TM511R
С
появлением
Senseit
P101
выбор
становится
еще
шире
,
и
теперь
пользователи
могут
остановить
свое
внимание
на
том
,
что
им
необходимо
,
а
не
выбирать
из
одного-двух
аппаратов
.
К
содержанию
>
>
>
Дизайн
,
размеры
,
управляющие
элементы
Обычно
защищенные
телефоны
выходят
небольшими
тиражами
,
как
следствие
,
их
выпускают
только
в
одном
цветовом
решении
,
никакого
выбора
не
существует
.
В
Senseit
P101
решили
отойти
от
канонов
и
запустили
в
продажу
сразу
три
цветовых
решения
–
оранжевое
(
как
у
нас
в
обзоре
,
классика
жанра
)
,
темно-зеленое
и
черное
.
Мне
нравятся
все
цветовые
решения
,
тут
каждый
найдет
свой
вариант
.
Телефон
относительно
небольшой
по
размеру
–
125х58х19
мм
,
вес
–
145
грамм
.
Отлично
лежит
в
руке
,
боковые
вставки
из
резины
,
не
проскальзывает
,
даже
когда
руки
мокрые
.
На
верхнем
торце
есть
фонарик
,
он
довольно
яркий
,
но
ничего
особенного
не
предлагает
.
На
клавиатуре
есть
кнопка
быстрого
включения
для
фонарика
.
Разъем
microUSB
спрятан
за
заглушкой
,
привинченной
к
левой
стороне
телефона
,
тут
придется
использовать
только
кабель
из
комплекта
,
слишком
глубоко
запрятан
разъем
,
обычный
штекер
microUSB
не
дотянется
.
На
задней
стороне
можно
видеть
объектив
VGA-камеры
,
серьезно
говорить
о
ее
качестве
не
стоит
,
по
современным
меркам
это
ничто
.
Впрочем
,
большинство
аналогичных
аппаратов
обладают
такими
же
камерами
,
это
бюджетные
решения
.
Задняя
крышка
закручивается
двумя
винтами
,
в
комплекте
есть
“
отвертка
”
,
чтобы
вывернуть
их
,
но
прекрасно
подойдет
и
монетка
.
Винты
имеют
большую
высоту
,
крутить
придется
довольно
долго
–
за
защиту
надо
платить
,
в
том
числе
и
так
.
Вы
явно
не
будете
раскручивать
эти
аппараты
постоянно
,
в
этом
не
будет
нужды
.
Из
минусов
отмечу
то
,
что
поддеть
крышку
проще
сбоку
,
выступ
,
сделанный
по
центру
неудобен
для
пальцев
,
об
него
можно
обломать
ногти
.
Впрочем
,
и
отвертка
из
комплекта
тут
не
поможет
,
надо
приноровиться
открывать
телефон
.
Внутри
вы
увидите
батарею
,
а
под
ней
–
два
слота
для
полноразмерных
SIM-карт
,
а
также
разъем
для
microSD-карты
(
до
32
ГБ
)
.
Качество
сборки
у
аппарата
отличное
,
есть
держатель
для
ремешка
,
то
есть
можно
будет
телефон
подвесить
на
шею
или
пояс
.
Стандарт
защиты
–
IP68
,
то
есть
аппарат
можно
смело
кидать
в
воду
,
он
не
утонет
.
Проверка
показала
,
что
вода
внутрь
не
попадает
,
в
реальных
условиях
этот
телефон
наверняка
будет
вести
себя
ровно
так
же
,
главное
–
закрыть
заглушку
и
затянуть
винты
.
К
содержанию
>
>
>
Дисплей
Диагональ
экрана
–
2.4
дюйма
,
разрешение
–
240х320
точек
,
что
неплохо
для
бюджетной
модели
.
Экран
,
по
современным
меркам
,
не
самый
лучший
,
но
в
бюджетных
защищенных
аппаратах
он
выгодно
отличается
большей
диагональю
и
разрешением
(
в
том
же
Mariner
–
2.2
дюйма
,
176х220
точек
)
.
На
ярком
солнце
экран
слепнет
,
что
также
типично
для
аналогичных
продуктов
.
Сказать
что-то
особенное
про
него
не
выйдет
.
Все
ожидаемо
.
К
содержанию
>
>
>
Клавиатура
Клавиатура
сделана
из
пластика
,
клавиши
имеют
яркую
подсветку
белого
цвета
,
она
хорошо
различима
как
на
улице
,
так
и
в
помещении
.
Ход
кнопок
средний
,
номер
легко
набрать
как
в
обычных
перчатках
,
так
и
без
них
.
Клавиатура
по
эргономике
хороша
,
никаких
замечаний
к
ней
нет
.
К
содержанию
>
>
>
Аккумулятор
Для
простого
кнопочного
телефона
батарея
Li-Ion
емкостью
1800
мАч
означает
длительную
работу
от
одной
зарядки
.
Производитель
заявляет
до
10
часов
работы
в
режиме
разговора
и
до
680
часов
в
режиме
ожидания
.
На
практике
аппарат
легко
работает
неделю
при
3-4
часах
разговоров
всего
.
В
режиме
ожидания
при
меньшем
числе
звонков
это
около
2
недель
.
Как
видите
,
это
очень
неплохо
,
с
другой
стороны
,
это
только
звонки
и
SMS
.
Полное
время
зарядки
–
около
2
часов
.
К
содержанию
>
>
>
Радио
и
проигрыватель
мр3
–
карта
памяти
При
желании
можно
докупить
карту
памяти
любого
объема
(
до
32
ГБ
)
,
записать
на
нее
мр3
книги
,
музыку
или
подкасты
.
Но
думаю
,
что
большинство
будет
использовать
радио
,
которое
имеет
встроенную
антенну
и
отлично
работает
в
режиме
громкой
связи
.
Этакая
музыкальная
шкатулка
.
Автопоиск
станций
,
переключение
в
списке
между
станциями
.
Все
предельно
просто
и
понятно
.
К
содержанию
>
>
>
Меню
Обычный
аппарат
с
фирменным
меню
от
Senseit
,
набор
функций
типичен
,
остановимся
на
них
.
Телефонная
книга
.
Отображается
как
из
памяти
аппарата
,
так
и
с
SIM-карты
.
На
одно
имя
можно
записать
только
один
номер
,
всего
до
300
номеров
в
памяти
телефона
.
Есть
три
списка
вызовов
–
пропущенные
,
набранные
,
полученные
–
указываются
время
звонка
и
его
продолжительность
.
Сообщения
.
Поддерживаются
не
только
SMS
,
но
и
MMS
,
хотя
последние
вряд
ли
актуальны
для
большинства
пользователей
этого
аппарата
.
Есть
шаблоны
SMS
,
счетчик
сообщений
,
при
вводе
сообщений
можно
задействовать
Т9
.
В
памяти
можно
сохранить
до
100
SMS
сообщений
,
50
MMS
.
Медиа
.
Здесь
содержится
доступ
к
камере
,
можно
получать
фотографии
640х480
точек
(
качество
плохое
даже
на
экране
)
,
записывать
видео
до
176х144
точки
,
тут
же
галерея
для
просмотра
всего
этого
.
Также
плеер
мр3
,
радио
плюс
диктофон
(
во
время
разговора
он
не
работает
)
.
Чтобы
слушать
музыку
,
ее
надо
сохранить
в
определенную
папку
Music
на
карте
памяти
.
Органайзер
.
Здесь
находятся
калькулятор
,
календарь
(
нет
ввода
событий
,
только
отображение
календаря
)
,
будильник
(
до
5
разных
будильников
)
,
файловый
менеджер
.
Настройки
.
Профили
позволяют
настроить
то
,
как
телефон
будет
вибрировать
,
звонить
,
какие
мелодии
будут
на
сообщениях
и
так
далее
,
все
привычно
.
Выбрать
время
и
яркость
подсветки
,
переназначить
клавиши
быстрого
доступа
(
их
4
–
соответствуют
направлениям
навигационной
кнопки
)
.
К
сожалению
,
в
телефоне
нет
резервной
батареи
,
и
каждый
раз
при
разряде
или
при
вынимании
батареи
вам
придется
настраивать
время
и
дату
–
автоматической
установки
этих
параметров
нет
.
Впечатления
Наличие
двух
SIM-карт
и
одного
радиомодуля
говорит
о
том
,
что
использовать
обе
карты
можно
,
но
стоит
приготовиться
к
тому
,
что
одна
может
быть
недоступна
во
время
разговора
по
второй
.
С
качеством
связи
нет
никаких
проблем
,
аппарат
ловит
хорошо
,
как
в
городе
,
так
и
вне
его
,
хотя
и
не
может
похвастаться
чем-то
особенным
.
Громкость
звонка
хорошая
,
виброзвонок
также
чувствительный
,
хотя
из
карманов
верхней
одежды
его
можно
и
пропустить
.
Цена
этого
аппарата
–
3
999
рублей
,
что
аналогично
стоимости
RugGear
Mariner
RG128
.
На
стороне
новинки
от
Senseit
большая
емкость
батареи
и
время
работы
,
но
она
в
комплекте
одна
,
да
и
телефон
не
умеет
плавать
.
Так
ли
это
надо
?
Сомневаюсь
,
что
это
очень
востребованные
функции
,
но
кому-то
приглянется
вторая
батарея
в
комплекте
.
В
минус
Mariner
стоит
записать
меньший
экран
,
в
остальном
функциональность
сравнима
.
Обзор
защищенного
телефона
RugGear
RG128
Mariner
За
ту
же
сумму
можно
взять
Texet
TM-511R
,
у
которого
меньше
экран
,
но
заметно
больше
батарея
,
стандарт
защиты
IP67
.
По
этому
аппарату
есть
довольно
много
нареканий
на
качество
,
попадается
брак
.
Но
при
прочих
равных
он
выглядит
интереснее
,
чем
конкуренты
–
в
первую
очередь
,
за
счет
большой
батареи
.
Как
ни
странно
,
одним
из
плюсов
Senseit
P101
выступает
выбор
разных
цветов
корпуса
–
хотите
яркий
оранжевый
,
а
хотите
спокойный
черный
или
зеленый
.
Тут
все
будет
зависеть
от
ваших
предпочтений
.
В
остальном
же
,
аппарат
вышел
вполне
неплохим
для
своего
класса
.
С
учетом
подвижки
ценовых
сегментов
на
российском
рынке
теперь
самые
недорогие
защищенные
модели
стоят
порядка
4
000
рублей
.
Хотя
раньше
они
начинались
по
цене
почти
в
два
раза
ниже
.
