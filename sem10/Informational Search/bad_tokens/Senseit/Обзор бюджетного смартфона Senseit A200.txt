Позиционирование
Бюджетный
смартфон
с
самым
популярным
форм-фактором
и
,
соответственно
,
экраном
в
5
дюймов
.
Позиционирование
таких
устройств
лишено
изысков
,
это
необходимый
минимум
за
свои
деньги
,
причем
в
этот
минимум
уже
входит
поддержка
4G
,
а
также
HD-экран
.
В
данном
ценовом
сегменте
на
полках
магазинов
не
так
много
устройств
,
обычно
они
работают
на
старых
версиях
Android
,
не
имеют
поддержки
4G
.
Это
уже
не
устройства
первой
цены
,
как
операторские
смартфоны
,
например
,
Senseit
A109
,
созданный
для
Билайн
,
имеет
ту
же
диагональ
экрана
,
но
похвастать
высоким
разрешением
не
может
.
Поэтому
стоит
рассматривать
Senseit
A200
как
бюджетный
,
но
отнюдь
не
самый
дешевый
смартфон
с
4G
,
это
решение
для
тех
,
кто
уже
смотрит
,
а
что
именно
может
предложить
телефон
.
Во
многом
эту
модель
можно
назвать
рабочей
лошадкой
,
так
как
в
ней
довольно
типичные
характеристики
.
Дизайн
,
размеры
,
управляющие
элементы
От
бюджетных
смартфонов
никто
не
ждет
чего-то
особенного
в
области
дизайна
,
большинство
из
них
скучны
и
однообразны
.
В
Senseit
A200
попытались
немного
поиграть
с
дизайном
,
так
,
с
нижнего
торца
динамик
вынесли
на
лицевую
панель
,
оформили
его
решеточкой
,
также
как
и
разговорный
.
Первое
впечатление
,
что
в
смартфоне
стереодинамики
,
к
сожалению
,
это
не
так
.
Во
время
разговора
действует
верхний
динамик
,
звук
звонка
или
музыку
воспроизводит
нижний
.
Но
впечатление
от
дизайна
иное
.
Цветовое
решение
у
корпуса
ровно
одно
-
черная
лицевая
панель
,
темно-серый
корпус
на
задней
части
телефона
,
тут
он
имеет
мелкоячеистую
структуру
.
Очень
приятная
на
ощупь
текстура
,
вроде
бы
мелочь
,
но
видно
,
что
пытались
сделать
так
,
чтобы
корпус
не
затирался
и
оставался
долгое
время
таким
же
,
как
в
первый
день
после
покупки
.
Традиционно
для
Senseit
камера
выделена
сзади
красным
ободком
,
что
дает
небольшой
акцент
и
чуточку
оживляет
привычный
дизайн
.
На
правой
боковой
стороне
находятся
спаренная
клавиша
регулировки
громкости
,
кнопка
включения/выключения
.
На
верхнем
торце
-
разъем
3.5
мм
,
на
нижнем
торце
-
разъем
microUSB
.
Корпус
разборный
,
его
легко
снять
и
обнаружить
внутри
два
слота
для
microSIM
,
а
также
слот
для
карты
памяти
.
Аккумулятор
также
сделан
съемным
.
Размеры
смартфона
-
143.5х71.4х8.95
мм
,
вес
-
139
грамм
.
Он
отлично
лежит
в
руке
,
не
вызывает
затруднений
в
переднем
кармане
джинсов
.
Компактная
модель
для
своей
диагонали
,
типичный
представитель
бюджетного
класса
.
По
сборке
никаких
нареканий
нет
,
все
отлично
.
Дисплей
При
диагонали
в
5

экран
имеет
HD-разрешение
(
1280х720

,
294

)
,
нужно

,
что
есть
датчик

яркости
экрана
.
Качество
картинки
хорошее
,
экран
яркий
,
подстраивается
под
освещение
,
что

.
Для
своей

модель
имеет

выше

,
что
можно
только

,
он
остается

на

.
Аккумулятор
Батарея
Li-Ion
2500

,

–

,

.
Время

–

.
Время

HD-видео
в
MX
Player
составляет
5.5

на

яркости
,
что
примерно
в
два
раза
больше
,
чем
у
аналогичных
по
стоимости
бюджетных

.
При
умеренном
использовании
аппарат
работает
полный
световой
день
,
это
примерно
50
минут
разговоров
,
до
полутора

в

,
а

социальных

.
Уверенный
день
работы
,
этого
вполне

для
такой

.
Для
сравнения
,
Senseit
A109
(
батарея
1950

,
но

меньше

)
,

,
причем

так
на
30
.
Хотя
и
его
можно

работать

день
,
другое
дело
,
что
сделаете
вы
на
аппарате
за
это

меньше

.

,

,

,

.
Телефон

,

.

.

Смартфон

6.0.1
,

,

.
Есть

,

,

,

.
По

и

.
Обзор
операционной
системы
Android
6.0
Marshmallow
Из

программ
отмечу
диспетчер
файлов
,

.
Встроенное
FM-радио
работает
только
с
гарнитурой
,
оно
вполне
типично
,
хорошо
играет
.
Использование
телефона
в
качестве
музыкальной
шкатулки
не
всегда
оправдано
,
динамик
на

громкости
может
давать
хрипотцу
.
В
наушниках
звук
вполне
приемлемый
,
но
назвать
его
особенным
нельзя
,
это
бюджетная
модель
со
всеми
вытекающими
.
Впечатления
По
качеству
связи

нареканий
к

нет
,
громкость
звонка
средняя
,

по

.
Виброзвонок
заметен
в

.
Цена
этого

-
6
490
рублей
,

.
Отличие
только

,

на

5.1
и

.
Насколько

,

,

,

что

бодрее
.
Других

вы
не

.
Работа

,

,
чем

,

ее

остается
средним
.
Конкурентов

,

,

же

A109
выглядит

,

у

экрана
,

,

.

.
