.

,

.
Как

,

–

,

,

.
Но

,

не

–
эксклюзивность
и
продажи
через
одного

означают
высокую
стартовую
цену
,
отсутствие

и
гибкости
цены
.
Модель
может
продержаться
на
одном

–

продаж
,
несмотря
на
то
,
что
весь
рынок
уже
уйдет
в
другом
направлении
и
аналогичные

стоить
заметно
меньше
.
В
такой
стабильности
,
пожалуй
,
больше

для
потребителей
,
а
выгоды
неясны
.
Эти

полной
мере
применимы
к
Idol
2S
,
эта

стоит
заметно
больше
,
чем
ее
аналоги
от
Alcatel
.
Тут
надо
отметить
,
что

на
борту
LTE
,
но
так
ли
это
важно
для
целевой
аудитории
от
Alcatel
?
На
мой
взгляд
,
совсем
не
важно
–
люди
,
покупающие
Alcatel
,
не
нуждаются
в
LTE
.
Поймите

,
дело
не
в
том
,
что
им

не
нужен
LTE
,
это
приятная

,
если

–
но
он
не
нужен
за
счет

в
15-20

,
для

Alcatel
критичны

и

.
Аудиторией
Idol
2s

абоненты

,

LTE-смартфоне
.
То
есть
,
это
ниша
,
и

.
Также
хочу
отметить
,
что

,
что
Idol
2

2s

,

.
У

больше

,

,

,

.
Они

,

.
Этот

,

.
Все

,

,

.

Дизайн
,
размеры
,
управляющие
элементы
Внешне
перед
нами
типичный
Alcatel
-
это
означает
приятный
корпус
из
пластика
,
конструкция
моноблочная
.
Производитель
предлагает
модель
в
разных
цветах
,
но
,
как
правило
,
в
рознице
можно
найти
темно-серебристый
аппарат
.
Размер
телефона
–
136.5
x
69.7
x
7.45
мм
,
вес
–
126
грамм
.
Компактный
,
хорошо
ложится
в
руку
.
Лицевая
панель
одноцветная
(
зависит
от
цвета
корпуса
,
так
,
в
белом
исполнении
она
также
белая
)
.
По
лицевой
стороне
этот
аппарат
сложно
отличить
от
других
моделей
,
он
начинает
играть
,
если
перевернуть
его
–
корпус
с
небольшим
изгибом
,
приятный
пластик
,
который
устойчив
к
истиранию
.
На
задней
поверхности
можно
видеть
большой
динамик
,
вверху
–
светодиодную
вспышку
,
а
также
8-мегапиксельную
камеру
с
автофокусом
.
На
левой
боковой
стороне
находится
слот
для
SIM-карты
,
а
также
для
карт
памяти
microSD
,
на
правой
–
спаренная
клавиша
регулировки

и
кнопка

.
Разъем
3.5

вынесли
на

,
а

–
на

.
На
торцах

видеть
отверстия

,
их
тут
два
.
Над
экраном
есть
индикатор
освещенности
,
фронтальная
камера
,
а
также
светодиод
,
его
можно
настроить
на
работу
при
пропущенных
звонках
или
событиях
.
В
целом
,
качество

не
вызывает
вопросов
,
но
при
сильном
сжатии
задняя
панель
потрескивает
.
Кому-то
это
может
не
понравиться
–
на
надежности
аппарата
это
никак
не
сказывается
,
но
конструкция
такова
,
что
скрипит
.
Неаккуратность
при
проектировании
аппарата
,
которая
дает
такой
эффект
.
В
реальной
жизни
вы
не
обратите
на
это
внимания
и
вряд
ли
будете
сжимать
так

.

Дисплей
В

стоит
IPS-матрица

5

и

720х1080

.
Мультитач

5

.
Запас

,

,

–

,

.
Возможно
,

,

,

на

.
На

выцветает
,
но
остается

.
К

>
>
>

В

.
По

,

На

.
При

,

,

.
Подобные

,

.
То

,

,

.
Полное

.

Телефон

(

)

Программные
особенности
–

4.3
В
Idol
2s
,
также
как
в
других
моделях
,
компания
Alcatel
использует
свою
программную
оболочку
,
что
может
быть
воспринято
многими
неоднозначно
,
так

зачастую
люди
предпочитают
«
голый
»
Android
и
вполне
удовлетворены
его
возможностями
.
Многие
функции
сильно
переработаны
,
в
Alcatel
добавили

программ
,
которые
можно
загрузить
бесплатно
и
так
,
но
то
,

,

.
Обзор

Android
4.2
Новый
Google
Nexus
7
на
Android
4.3
Без
разблокировки

вы

посмотреть
прогноз
погоды
на
текущий
день
.
Пролистнем
экран
вправо
,
тут

календарь
и

дня
(

)
.
Также
можно
посмотреть
статусную
строку
,
она

и

,

ключа

или

кода
.
Кардинальное
отличие

Alcatel
от

оболочек
в

,

.
На

,

.
Это

.
Листаем

,
и

,

.
Идем

,
и

,

.
К

.
В

.

,

.
В

,

,

,

.
Далее

,

,

.
Это

,

,

,

,

.
Поэтому

.
Еще

,

,

.
В

,

1080p
,

.
