Мини-обзор
карманного
компьютера
MiTAC
Mio
339
Комплект
поставки
:
КПК
Док-станция
Зарядное
устройство
Чехол
из
кожзаменителя
Диск
с
программным
обеспечением
Документация
Пульт
для
управления
воспроизведением
MP3
Наушники
Мы
не
будем
подробно
рассматривать
все
функции
данного
наладонника
,
поскольку
все
они
были
рассмотрены
в
обзоре
карманного
компьютера
Rover
P4
,
который
построен
на
платформе
от
компании
MiTAC
.
Остановимся
лишь
на
отличиях
:
на
встроенной
камере
и
воспроизведение
MP3
(
наличие
пульта
для
управления
MP3
в
комплекте
,
что
встречается
крайне
редко
на
рынке
Pocket
PC
)
.
Камера
Для
КПК
на
базе
Palm
OS
встроенная
камера
–
привычное
дело
.
Практически
все
топовые
модели
Sony
оснащены
камерой
,
компания
PalmOne
имеет
в
своем
активе
КПК
Palm
Zire
71
.
Но
на
рынке
Pocket
PC
камера
–
пока
редкость
.
Модель
Mio339
–
первый
и
пока
фактически
единственный
КПК
Pocket
PC
(
не
смартфон
)
со
встроенной
камерой
.
Можно
еще
вспомнить
Toshiba
e550c
,
но
дальше
японского
рынка
модель
не
ушла
.
Камера
расположена
на
задней
стенке
устройства
рядом
с
верхней
частью
стилуса
,
поскольку
она
не
прикрыта
ничем
,
то
при
извлечении
пера
большой
риск
задеть
«
объектив
»
,
испачкав
или
поцарапав
.
Чуть
ниже
–
маленькое
зеркальце
для
автопортрета
.
Отдельной
кнопки
,
отвечающей
за
включение
камеры
,
нет
(
сама
программа
называется
Mio
Camera
)
,
но
на
ее
запуск
можно
назначить
одну
из
стандартных
клавиш
(
мы
назначили
на
нее
кнопку
диктофона
)
.
Программа
стартует
около
четырех
секунд
.
После
чего
можно
работать
с
камерой
,
делать
снимки
или
видео
.
Экран
при
этом
разворачивается
в
альбомный
режим
,
и
все
управление
камерой
осуществляется
пятипозиционной
клавишей
,
которая
удобно
ложится
под
большой
палец
правой
руки
(
в
настройках
можно
переориентировать
дисплей
под
левую
руку
)
.
Впрочем
,
управлять
можно
и
стилусом
.
Дисплей
КПК
при
этом
служит
в
качестве
видоискателя
,
отображая
и
дополнительную
информацию
:
куда
сохраняются
изображения
(
основная
память
или
на
SD
карту
)
,
категория
фото
,
режим
съемки
,
дату
и
время
,
заряд
батареи
,
оставшееся
количество
фотографий
.
Изображение
на
экран
с
камеры
передается
без
задержки
,
а
не
с
запаздыванием
на
пару
секунд
,
как
,
например
,
в
модели
Palm
Zire
71
.
Отметим
один
негативный
момент
–
с
момента
нажатия
на
центр
джойстика
(
играет
роль
кнопки
спуска
)
до
фотографирования
проходит
несколько
секунд
.
Из-за
этого
иногда
получаются
смазанные
изображения
или
не
получаются
вовсе
(
нажимаешь
на
спуск
и
убираешь
камеру
,
думая
,
что
фото
сделано
)
,
но
к
этому
моменту
можно
вполне
со
временем
привыкнуть
.
В
итоге
получается
,
что
у
камеры
в
целом
довольно
продолжительное
время
старта
(
от
включения
до
реального
фото
проходит
как
минимум
6
секунд
)
.
Неожиданную
сцену
вполне
можно
и
упустить
,
не
успев
сфотографировать
.
Имеется
встроенная
вспышка
!
Это
все
еще
большая
редкость
в
КПК
и
смартфонах
(
на
вскидку
вспоминается
только
Sony
NZ90
и
Sendo
X
)
.
Если
делать
снимки
в
ночное
время
на
улице
или
в
неосвещенном
помещении
,
то
вспышка
здесь
плохой
помощник
–
нормально
освещает
пространство
на
расстоянии
только
до
полуметра
.
Если
снимать
на
большем
расстоянии
в
темноте
,
то
на
фото
объекты
получаются
плохо
различимыми
(
примеры
фотографий
в
темноте
со
вспышкой
:
на
большом
расстоянии
,
на
маленьком
расстоянии
)
.
Вспышку
можно
активировать
постоянно
,
тогда
она
будет
работать
как
фонарик
,
позволяет
снимать
видео
в
темноте
.
Расскажем
о
настройках
камеры
.
Камера
снимает
в
разрешении
VGA
(
640х480
точек
)
или
QVGA
(
320х240
точек
)
.
Можно
выставить
контраст
вручную
или
автоматически
,
аналогично
–
с
балансом
белого
.
Можно
отключить
или
включить
щелчок
при
нажатии
на
спуск
.
В
режиме
ожидания
отклонений
джойстика
влево/вправо
управляет
цифровым
зумом
(
максимум
–
2х
)
.
В
общем
,
для
встроенной
в
КПК
камеры
–
количество
настроек
приятно
удивляет
.
Если
говорить
о
качестве
снимков
,
то
оно
среднее
для
VGA
камер
.
Хвалить
не
за
что
,
но
и
придраться
особо
не
к
чему
.
Наиболее
оптимально
камера
подойдет
для
съемки
при
дневном
свете
на
расстоянии
1-2
метров
(
фото
друзей
,
близких
,
несложные
архитектурные
пейзажи
)
.
Впрочем
,
вы
сами
можете
оценить
получившиеся
фотографии
,
которые
размещены
на
отдельной
страничке
.
Примеры
фоторгафий
Видеоролики
снимаются
в
двух
разрешениях
:
это
320х240
точек
или
176х144
точек
.
Продолжительность
записываемого
ролика
зависит
от
свободного
места
в
памяти
устройства
.
Качество
низкое
,
даже
при
воспроизведении
с
КПК
.
На
настольном
компьютере
сразу
просмотреть
видео
не
получилось
,
даже
несмотря
на
то
,
что
файлы
сохраняются
c
расширением
.mpg
,
пришлось
скачивать
кодек
с
сайта
MiTAC
.
Пример
видео
(
320х240
точек
,
129
Кб
)
Пример
видео
(
176х144
точек
,
99
Кб
)
Аудио
плеер
Одна
из
интересных
особенностей
данного
КПК
–
оснащенность
пультом
управления
музыкой
.
Его
можно
прикрепить
к
верхней
одежде
,
и
не
доставать
КПК
всякий
раз
,
когда
нужно
нажать
на
паузу
или
прокрутить
на
следующую
композицию
.
Отметим
минус
–
команды
с
пульта
«
понимает
»
только
встроенный
плеер
,
который
носит
характерное
название
–
MP3Player
.
Сторонние
программные
MP3
плееры
«
не
видят
»
пульт
,
то
есть
назначить
на
клавиши
пульта
какие-либо
действия
в
них
нельзя
.
Качество
воспроизведение
музыки
в
наушниках
неплохое
(
благо
во
встроенном
плеере
есть
эквалайзер
)
,
но
максимальный
уровень
громкости
мог
быть
и
больше
(
в
очень
шумных
местах
музыка
заглушается
,
ее
практически
не
слышно
)
.
Когда
уровень
заряда
батареи
КПК
опускается
до
20
%
,
плеер
прекращает
воспроизведение
,
ссылаясь
на
низкий
уровень
заряда
аккумулятора
.
Итоги
При
цене
в
$
330-350
КПК
выглядит
неплохим
приобретением
,
если
на
минимальную
сумму
нужен
«
развлекательный
»
центр
,
сочетающий
в
себе
камеру
,
mp3
плеер
и
полноценный
карманный
компьютер
.
Из
других
бюджетных
КПК
со
встроенной
камерой
можно
отметить
Palm
Zire
71
и
Sony
TJ37
,
но
они
не
являются
прямыми
конкурентами
КПК
от
MiTAC
,
так
как
работают
на
другой
операционной
системе
.
Если
потребности
в
камере
нет
(
например
,
уже
куплена
отдельная
цифровая
камера
или
есть
встроенная
в
телефоне
)
,
то
предпочитаемыми
окажутся
конкурирующие
модели
(
такие
как
Toshiba
e350
,
iPaq
1940
,
Rover
P4
)
в
основном
за
счет
меньшей
цены
.
Об
остальных
достоинствах
и
недостатках
модели
Mio339
можно
прочитать
в
обзоре
Rover
P4
,
а
в
данной
статье
мы
рассказали
об
отличительных
особенностях
.
