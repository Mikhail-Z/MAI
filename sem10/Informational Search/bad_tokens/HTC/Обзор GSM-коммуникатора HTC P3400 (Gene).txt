Аппарат
имеет
в
целом
классическую
внешность
.
Ожидать
от
бесклавиатурного
коммуникатора
неординарных
идей
в
плане
дизайна
нет
смысла
.
Хотя
нельзя
не
отметить
,
что
дизайнерам
HTC
уже
в
который
раз
удалось
привнести
в
стандартную
обертку
новые
идеи
.
HTC
P3400
имеет
обтекаемую
форму
.
Лицевая
часть
единственная
,
которая
сделана
строго
,
ребра
и
торцы
не
имеют
точно
очерченных
граней
и
плавно
переходят
в
заднюю
крышку
.
Интересно
,
что
большинство
людей
,
которые
знакомились
с
коммуникатором
впервые
,
отмечали
то
,
что
его
удобно
держать
в
руке
.
Аппарат
производит
хорошее
впечатление
.
Это

важно

в
свете
позиционирования
модели
.
Судя

,

,

хорошее
впечатление
на
людей
,

с
коммуникаторами
.
Дизайн
прост
и
приятен
.
Частично
это

за
счет
формы
,

–

пластика
.
В
HTC
P3400
используется
пластик
с

(
soft-touch
)
,
что

удобство
.
Из
этого
пластика
выполнена
большая
часть
коммуникатора
,
за

лицевой
панели
.
Аппарат
не
скользит
в
руках
,
на
нем

остаются

и
потертости
.
К

,

подобного

в

аппарата
.
Так

,
что

в

.
Именно
этот
стереотип

.
Габариты
устройства
(
109x58x17.65
мм
,

126

)

.
За

руке
.
Носить

,
так

рубашки
.
Хотя

.
На

,
разговорный

.
По

,

.
Один

(

)
,
а

о
работе

(
зеленый
)
.
Второй

активность
модуля
Bluetooth
(
синий

)
.
Рассмотрим
навигационный
блок
.
Он
включает
в
себя
шесть
кнопок
и
пятипозиционную
навигационную

в
центре
.
Для
продуктов
HTC

навигационных
клавиш

стал
стандартным
.
Причем

него
входит
необходимый
минимум
кнопок
для
выполнения
простейших
действий

использования
пера
.
Речь
идет
,
в
первую

,
о
клавише
«
Ok
»
,
находящейся

от
навигационной

,
с
ее

можно

к

Today
.
Слева

.
Верхний

.
Клавиши

и

.
Выглядит

.
Говоря

HTC
,

у

.
Например
,

Quanta

.
В

O2
XDA
Atom
клавиши

в
меню
Today
нет
(
на
ее

)
,
в
Glofiish
M700
нижнего

нет

,
только
отбой\прием
вызова
и
софт-кнопки
,
в
Orsio
n725
на

находится

меню
почты
и

,
есть

.
Радует
,

новых

свернули
c

.
Стандартный

удобен
.
В
ближайшем

,

.
Из

клавиш

только

–
прием
и
отбой

.
Соответственно
зеленым
и

.
Центральная

находится
на
общем

всем
блоком

,
немного

.
Весь

,
центральная

.
Кнопки

,

.
Каждое

.
Единственное

–

.
За
счет
того
,
что
они

левого
и
правого
блоков
,

нажатиях

задеваются
навигационные
кнопки
.
Центральная
клавиша
в
целом
удобна
,
хотя
к

.
На

и

,

.
В

ютится

.
Ее

несколько
непривычно
в
первое
время
работы
.
Нажимать
ее
удобнее

указательным

.
Правое

голосового

.
Все

одном

,
выступая

миллиметр
.
Это
не

на
удобстве
.
Нажатия
клавиш
не

трудностей
.
В
верхней
части
правого

разъем

.
Перо

в

.
По

P3600
.
Размеры

.
Если

к

перу
,

.
Верхняя

,
это
позволяет
не
глядя

его

.
Для
извлечения

же

.
В

.
Наконец-то

.
Для

.
Короткий

.
Верхнее

под

памяти
формата
SD/MMC
(
SDIO
)
.
Это
самый

,
его

.
Карты

4
Гб

120

США
(
3000

)
.
Разъем

.
Это

.
Карта
вставляется
до
щелчка
,
после
чего

в
разъеме
.
Для
того

ее
вынуть
,
достаточно

на
верхнее
ребро
карты
.
Нижний
торец
расстраивает

сильно
,
как
радует
верхний
.
Радость

общения
с
HTC
X7500
была
недолгой
.
В
P3400
нам
снова
предлагается
единственный
разъем
под
зарядное
устройство
,
кабель
и
аксессуары
.
И
,
конечно
же
,
это
фирменный
ExtUSB
.
Разговоры
о
том
,
что
компания
слишком
долго
приходит
к
пониманию
того
,
что
разъем
не
слишком
удобен
,
начинать
не
имеет
смысла
.
Мы
говорили
об
этом
во
многих
обзорах
,
посвященных
продуктам

.
Пока
ситуация
неизменна
.
Отчасти
проблема

,
что
в
Москве

переходник
ExtUSB
–
3.5
мм
компании
Brando
.
Стоит
он

17
долларов
США
(
450
рублей
)
.
Разъем
поддерживает

USB
1.1
,

.
Проще

+

.
Слева

,

–

устройства
.
Обратимся
к
задней

.
Сверху

окошко
2
Мп
камеры
.
Объектив
немного
выступает
над
корпусом
и
ничем
не
защищен
.
С
другой
стороны
,
его
удобно

,
в

,

и

корпус
.
Справа

,
он
прикрыт

.
Всю

.
Крепление

.
По

.
Люфт

,

.
Если

,

.
Под

.
Оно

.
Карта
вставляется
в
паз
,
а
вынимается
с

.
Экран
Экран

с

аналогов
,

что

ознакомления
с
ними
возможностей
программы
более
чем
достаточно
.
Java
–
Эмулятор
для
запуска
программ
,
написанных
на
Java
.
Возможности
его
ограниченны
,
нет
поддержки
3D
Java
.
В
большинстве
ситуаций
поддержка
работы
Java-приложений
для
смартфонов
и
коммуникаторов
не
востребованна
,
программ
для
Windows
Mobile
сегодня
в
избытке
.
Исключением
для
некоторых
пользователей
может
стать
браузер
OperaMini
,
рекордсмен
по
экономии
трафика
.
Еще
один
пример
–
любители
Java-игр
,
которым
также
необходим
эмулятор
.
Здесь
стоит
оговориться
,
что
штатный
эмулятор
не
поддерживает
дополнительные
библиотеки
,
используемые
при
создании
Java-приложений
под
телефоны
Nokia
,
Motorola
и
некоторые
другие
.
Запуск
и
последующая
нормальная
работа
подобных
программ
не
гарантирована
.
Audio
Manager
–
приложение
,
аналогичное
стандартному
Media
Player
.
По
своим
возможностям
программа
не
отличается
от
встроенного
проигрывателя
ничем
,
кроме
внешнего
вида
,
полностью
дублирует
его
.
Ее
наличие
в
системе
не
совсем
понятно
с
учетом
отсутствия
каких-либо
отличий
от
Media
Player
.
В
плюсы
этой
программе
можно
записать
разве
что
приятный
интерфейс
.
Zip–
простая
программа
для
работы
с
ZIP-архивами
.
Позволяет
упаковывать
файлы
в
архив
,
распаковывать
их
,
а
также
просто
просматривать
находящиеся
на
коммуникаторе
архивы
.
Медиа-хаб
–
оригинальная
программа
от
тайваньцев
.
Назвать
это
полноценным
приложением
нельзя
,
скорее
,
перед
нами
своеобразная
оболочка
.
Запустив
ее
,
мы
получаем
доступ
к
анимированному
экрану
с
иконками
мультимедийных
приложений
.
Это
проигрыватель
Audio
Manager
,
программа
для
просмотра
картинок
Мои
фото
(
стандартное
приложение
фото
и
видео
)
и
для
просмотра
видео
также
она
.
Последним
пунктом
идет
приложение
Аудио-помощник
.
Аудио-помощник
–
программа
для
настройки
звука
.
Активировать
ее
можно
исключительно
при
подключенной
гарнитуре
или
наушниках
.
Из
настроек
доступны
регулировки
высоких
частот
и
басов
,
а
также
10
полосный
эквалайзер
.
Для
тех
,
кто
не
хочет
разбираться
с
этими
параметрами
вручную
предусмотрено
25
предустановленных
настроек
эквалайзера
.
Выводы
,
заключение
По
качеству
приема
сигнала
аппарат
не
имеет
нареканий
.
Этот
параметр
с
каждым
днем
играет
все
меньшую
роль
.
Громкость
вызывного
динамика
находится
на
среднем
уровне
,
во
время
разговора
собеседника
также
слышно
нормально
.
В
условиях
шумных
помещений
или
на
улице
вблизи
оживленной
магистрали
удобнее
использовать
гарнитуру
.
В
этих
условиях
громкости
аппарата
будет
недостаточно
для
комфортного
разговора
.
Вибровызов
–
это
мелкая
и
очень
частая
дробь
.
Ощущается
не
только
сама
вибрация
,
которая
очень
сильна
,
но
и
постоянный
шум
,
вызываемый
ею
.
Разработчик
немного
перестарался
с
мощностью
вибро
.
В
итоге
при
звонке
с
включенной
вибрацией
коммуникатор
начинает
жутко
трясти
,
при
этом
действо
сопровождается
шумом
,
издаваемым
корпусом
.
В
плюс
занесем
тот
факт
,
что
не
заметить
вибровызов
очень
сложно
.
Модель
появилась
на
рынке
в
середине
марта
.
Сегодня
средняя
цена
на
аппарат
составляет
500
долларов
США
(
13
000
рублей
)
.
Можно
говорить
о
том
,
что
HTC
P3400
становится
одним
из
наиболее
важных
игроков
на
рынке
.
Сочетание
зарекомендовавшего
себя
за
полтора
года
качества
продуктов
HTC
и
разумная
ценовая
политика
могут
позволить
коммуникатору
HTC
P3400
стать

продаж
среди
Windows
Mobile-устройств
.
Особенно
любопытно
то
,
что
HTC
–

и
пока
единственный
производитель
,
решивший

сегмент
.
Ни
HP
,
ни
E-Ten
(
Glofiish
)
,
ни
i-Mate

не

.
Продукты

,
в

,

для

PC
,
не

и
не

.
А
в

коммуникатор
должна
представить
компания
i-Mate
,
это

для
HTC
P3400
.
