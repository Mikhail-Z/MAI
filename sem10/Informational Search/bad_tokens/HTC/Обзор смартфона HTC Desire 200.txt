Дизайн

,

В

.
На

,

,

,

,

»
,

.
Лицевая

,

есть

.
Ничего

,

же

никаких
лишних

и

–
смартфон

и

,

есть

,
особенно

его
в
один

с
конкурентами
по
характеристикам
от
китайских

.
Вот
,

видны

,

.

Элементы

Под

:

Назад
»

»

Действий

.
Присутствует

.
Клавиша

.
Здесь

3.5

и

.
Разъем
microUSB

левом

.
Клавиша

–

.
На

,

.
В

,

)
,

–

.
Карточка

.

В

”

Качество

.

В

5

Обсуждать

.

,

.
Пример

(

,

)

В

1230

.
Если

30-40

,

,

,

Если

,

.
Но

целом
,

,
не

)

.

,

.

,

,

Заключение
К

.

Вызывной

,

,

.

.
Стоимость

.
Конечно

.
При

.

.

.
Во-вторых
,

.
Смартфон

,

,

,

.
В-третьих
,
HTC
Sense
.
Для

,

или

,

,

.
И

Однако

,

000

,

,

.

Впрочем

,
