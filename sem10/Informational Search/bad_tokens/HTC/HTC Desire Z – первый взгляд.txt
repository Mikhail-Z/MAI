.

их

.
Дизайн

.
И

,

,

,

,

.

,

.
Пластик

.
Нельзя

,

.
Аппарат

,

,

Сравните

Z
(
14,2

)

,

О

В

Это

Управление
В

Еще

.
Сам

Под

)

(

,

,

.
Эта

.
Клавиши

.
Возле

,

Разъемы
3.5

-

.
Клавиатура

По

Напомню

Клавиатура

Я

.
Ведь
