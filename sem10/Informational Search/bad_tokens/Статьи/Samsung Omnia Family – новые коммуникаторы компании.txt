Несмотря

,

.
Не

,

тому
(

)
стал
т.н
.

:

всегда

,

.
Среднего

,

к
2008

,
однако

и

.
Для

производителей

не
было
.
Компании

м
огромными

такие

,

Windows

этот

.
Вопрос

:

таким
же

,

,

,

,

иных

,

.
Из

:

Acer

(

)
.
Последний

,

.
Но

.
Во

,

.
План

,

,

.

,

Samsung

стоимость\функционал
.

,

,

.
